36      /**
37       * Creates a PackageDescriptor for the named package using the 
38       * specified ClassLoader to load the PackageInfo or Package.
39       */
40      public static PackageDescriptor forName(String packageName,
41                                              ClassLoader classLoader) {
42          
43          PackageDescriptor pd = new PackageDescriptor(packageName);
44          if (classLoader == null) {
45              classLoader = ClassLoader.getSystemClassLoader();
46          }
47          
48          if (packageName == null || packageName.trim().length() == 0) {
49              return pd;
50          }
51          
52          String packageInfoClassName = packageName;
53  
54          if (!packageInfoClassName.endsWith(".")) {
55              packageInfoClassName = packageInfoClassName + ".";
56          }
57          
58          packageInfoClassName = packageInfoClassName + "PackageInfo";
59  
60  
61          Class packageInfoClass = null;
62          
63          try {
64              if (classLoader != null) {
65                  packageInfoClass = 
66                      classLoader.loadClass(packageInfoClassName);
67              }
68              else {
69                  packageInfoClass = Class.forName(packageInfoClassName);
70              }
71          }
72          catch (Throwable t) {
73          }
74  
75          if (packageInfoClass != null) {
76              try {
77                  initFromPackageInfo(pd, packageInfoClass);
78              }
79              catch (Throwable t) {
80              }
81          }
82          else {
83  
84              Package pkg = null;
85              
86              if (classLoader != null) {
87                  // Get around getPackage's protected access
88                  PackageLoader packageLoader = 
89                      new PackageLoader(classLoader);
90                  
91                  pkg = packageLoader.getPackage(packageName);
92              }
93              
94              if (pkg == null) {
95                  pkg = Package.getPackage(packageName);
96              }                        
97  
98              if (pkg != null) {
99                  initFromPackage(pd, pkg);
100             }
101         }
102 
103         return pd;
104     }

188     // Initialize the PackageDescriptor from the Package
189     private static void initFromPackage(PackageDescriptor pd,
190                                         Package pkg) {
191 
192         pd.setExists(true);
193 
194         String specificationTitle = pkg.getSpecificationTitle();
195         String specificationVersion = pkg.getSpecificationVersion(); 
196         String specificationVendor = pkg.getSpecificationVendor();
197 
198         String implementationTitle = pkg.getImplementationTitle();
199         String implementationVersion = pkg.getImplementationVersion(); 
200         String implementationVendor = pkg.getImplementationVendor();
201 
202         if (implementationTitle == null) {
203             implementationTitle = specificationTitle;
204         }
205 
206         if (implementationVersion == null) {
207             implementationVersion = specificationVersion;
208         }
209 
210         if (implementationVendor == null) {
211             implementationVendor = specificationVendor;
212         }
213 
214 
215         pd.setSpecificationTitle(specificationTitle);
216         pd.setSpecificationVersion(specificationVersion);
217         pd.setSpecificationVendor(specificationVendor);
218 
219         pd.setImplementationTitle(implementationTitle);
220         pd.setImplementationVersion(implementationVersion);
221         pd.setImplementationVendor(implementationVendor);
222 
223         pd.setProduct(implementationTitle);
224         pd.setProductVersion(implementationVersion);
225     }
