49      public Asset getAsset(String path) {
50          // validate base path exists
51          if (basePath != null && !basePath.isEmpty() && 
52              !path.startsWith(basePath)) {
53              return null;
54          }
55          
56          // trim base path from path
57          path = path.substring(basePath.length());
58          
59          // get extension
60          int idx = path.lastIndexOf('.');
61          if (idx < 0) { return null; }
62          String extension = path.substring(idx + 1);
63          
64          // validate mime type
65          String mimeType = mimeTypes.get(extension);
66          if (mimeType == null) {
67              Syslog.warn("unauthorized access to restricted mime type: " + 
68                          this.name + ":" + path);
69              
70              return null;
71          }
72          
73          // check if cached for better efficiency
74          InputStream input = null;
75          AssetFactory cached = this.assets.get(path);
76          if (cached != null) {
77              input = cached.getAsset(path);
78          }
79          
80          // otherwise, find first valid factory
81          else {
82              for (AssetFactory factory : factories) {
83                  input = factory.getAsset(path);
84                  if (input != null) {
85                      // cache for better efficiency
86                      this.assets.put(path, factory);
87                      
88                      // break out of loop (factory found)
89                      break; 
90                  }
91              }
92          }
93          
94          // verify input found
95          if (input == null) {
96              Syslog.debug("unable to find resource: ".concat(path));
97              return null;
98          }
99          
100         // return found resource
101         return new Asset(path, mimeType, input);
102     }
