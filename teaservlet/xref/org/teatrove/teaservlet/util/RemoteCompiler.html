39  /**
40   * RemoteCompiler compiles tea source files by reading them from a remote 
41   * location specified by a URL. The compiled code can be written as class files
42   * to a given destination directory, they can be passed to a ClassInjector, or
43   * they can be sent to both.
44   *
45   * <p>When given a URL, RemoteCompiler compiles all files with the
46   * extension ".tea". If a destination directory is used, tea files that have a
47   * matching class file that is more up-to-date will not be compiled, unless
48   * they are forced to be re-compiled.
49   *
50   * @author Jonathan Colwell
51   * @version
52   * <!--$$Revision:--> 16 <!-- $--> 21 <!-- $$JustDate:-->  3/19/04 <!-- $-->
53   * @see ClassInjector
54   */
55  public class RemoteCompiler extends AbstractCompiler {

57      public static final String TEMPLATE_LOAD_PROTOCOL = "http://";

59      private String[] mRemoteSourceDirs;
60      private Map<String, TemplateSourceInfo> mTemplateMap;
61      private Map<String, SocketFactory> mSocketFactories;
62      private long mTimeout;

64      public RemoteCompiler(String[] rootSourceDirs,
65                            String rootPackage,
66                            File rootDestDir,
67                            ClassInjector injector,
68                            String encoding,
69                            long timeout,
70                            long precompiledTolerance) {
71                            	
72          super(rootPackage, rootDestDir, injector, encoding, precompiledTolerance);
73          mRemoteSourceDirs = rootSourceDirs;
74          mTimeout = timeout;
75          mSocketFactories = new HashMap<String, SocketFactory>();
76          mTemplateMap = retrieveTemplateMap();
77      }

90      /**
91       * Overrides the method from Compiler to allow timestamp synchronization
92       * after the actual compilation has taken place.
93       */
94      public String[] compile(String[] names) throws IOException {
95          String[] compiled = super.compile(names);
96          if (getErrorCount() == 0) {
97              for (int j=0;j<compiled.length;j++) {
98                  syncSources(compiled[j]);
99              }
100         }
101         return compiled;
102     }

108     /**
109      * sets the template classes to have the same timestamp as the sources
110      * to account for differences between the machine clocks.
111      */
112     private void syncSources(String name) {
113         TemplateSourceInfo info = (TemplateSourceInfo)mTemplateMap.get(name);
114         Unit compUnit = (Unit)this.getCompilationUnit(name,null);
115         File destFile = compUnit.getDestinationFile();
116         if (destFile != null) {
117             destFile.setLastModified(info.timestamp);
118         }
119     }

121     /**
122      * returns a socket connected to a host running the TemplateServerServlet
123      */
124     HttpClient getTemplateServerClient(String remoteSource) throws IOException {
125         SocketFactory factory = (SocketFactory)mSocketFactories.get(remoteSource);
126 
127         if (factory == null) {
128             int port = 80;
129             String host = remoteSource.substring(TEMPLATE_LOAD_PROTOCOL.length());
130 
131             int portIndex = host.indexOf("/");
132         
133             if (portIndex >= 0) {
134                 host = host.substring(0,portIndex);                         
135             }
136             portIndex = host.indexOf(":");
137             if (portIndex >= 0) {
138                 try {
139                     port = Integer.parseInt(host.substring(portIndex+1));
140                 }
141                 catch (NumberFormatException nfe) {
142                     System.out.println("Invalid port number specified");
143                 }
144                 host = host.substring(0,portIndex);
145             }
146             factory = new PooledSocketFactory
147                 (new PlainSocketFactory(InetAddress.getByName(host), port, mTimeout));
148             
149             mSocketFactories.put(remoteSource, factory);
150         }
151         return new HttpClient(factory);
152     }

154     /**
155      * turns a template name and a servlet path into a  
156      */
157     String createTemplateServerRequest(String servletPath,String templateName) {
158         String pathInfo = servletPath.substring(servletPath.indexOf("/",TEMPLATE_LOAD_PROTOCOL.length()));
159         if (templateName != null) {            
160             pathInfo = pathInfo + templateName;
161         }
162         return pathInfo;
163     }

165     /**
166      * creates a map relating the templates found on the template server
167      * to the timestamp on the sourcecode
168      */
169     private Map<String, TemplateSourceInfo> retrieveTemplateMap() {
170         Map<String, TemplateSourceInfo> templateMap = 
171         	new TreeMap<String, TemplateSourceInfo>();
172         for (int j=(mRemoteSourceDirs.length-1); j >= 0; j--) {
173             String remoteSource = mRemoteSourceDirs[j];
174             if (!remoteSource.endsWith("/")) {
175                 remoteSource = remoteSource + "/";
176             }
177             
178             try {
179                 HttpClient tsClient = getTemplateServerClient(remoteSource);
180 
181                 HttpClient.Response response = tsClient.setURI(createTemplateServerRequest(remoteSource,null))
182                     .setPersistent(true).getResponse(); 
183 
184                 if (response != null && response.getStatusCode() == 200) {
185 
186                     Reader rin = new InputStreamReader
187                         (new BufferedInputStream(response.getInputStream()));
188                 
189                     StreamTokenizer st = new StreamTokenizer(rin);
190                     st.resetSyntax();
191                     st.wordChars('!','{');
192                     st.wordChars('}','}');
193                     st.whitespaceChars(0,' ');
194                     st.parseNumbers();
195                     st.quoteChar('|');
196                     st.eolIsSignificant(true);
197                     String templateName = null; 
198                     int tokenID = 0;
199                     // ditching the headers by looking for "\r\n\r\n"
200                     /* 
201                      * no longer needed now that HttpClient is being used but leave
202                      * in for the moment.
203                      *
204                      * while (!((tokenID = st.nextToken()) == StreamTokenizer.TT_EOL 
205                      *       && st.nextToken() == StreamTokenizer.TT_EOL) 
206                      *       && tokenID != StreamTokenizer.TT_EOF) {
207                      * }
208                      */
209                     while ((tokenID = st.nextToken()) != StreamTokenizer.TT_EOF) {
210                         if (tokenID == '|' || tokenID == StreamTokenizer.TT_WORD) {
211                         
212                             templateName = st.sval;
213                         }
214                         else if (tokenID == StreamTokenizer.TT_NUMBER 
215                                  && templateName != null) {
216                             templateName = templateName.substring(1);
217                             //System.out.println(templateName);
218                             templateMap.put(templateName.replace('/','.'),
219                                             new TemplateSourceInfo(
220                                                                    templateName,
221                                                                    remoteSource,
222                                                                    (long)st.nval));
223                             templateName = null;
224                         }
225                     }
226                 }
227             }
228             catch (IOException ioe) {
229                 ioe.printStackTrace();
230             }
231         }
232         //System.out.println("retrieving templateMap");
233         return templateMap;
234     }

251     public class Unit extends AbstractUnit {
252 
253         Unit(String name, Compiler compiler) {
254         	super(name, compiler);
255             TemplateSourceInfo info = 
256             	(TemplateSourceInfo) mTemplateMap.get(mDotPath);
257 
258             if (info==null) {
259                 mSourceUrl = getSourceFileName();
260             } else if(info.server.trim().endsWith("/")) {
261                 mSourceUrl = info.server.trim() + getSourceFileName();
262             } else {
263                 mSourceUrl = info.server.trim() + "/" + getSourceFileName();
264             }
265         }
266 
267         public boolean shouldCompile() {
268             long remoteTimeStamp = ((TemplateSourceInfo) mTemplateMap.get(mDotPath)).timestamp;
269 
270             boolean isWindows = System.getProperty("os.name").startsWith("Windows");
271             if (!mForce &&
272                 mDestFile != null &&
273                 mDestFile.exists() && 
274                 ((isWindows && mDestFile.lastModified() >= remoteTimeStamp) ||
275                 (!isWindows && Math.max(mDestFile.lastModified(), remoteTimeStamp) - 
276                     Math.min(mDestFile.lastModified(), remoteTimeStamp) < 1000L))) {
277 
278                 return false;
279             } else
280             // Try to defer to the template repository (and possibly precompiled templates)
281             return shouldCompile(remoteTimeStamp);
282         }

284         /**
285          * get a input stream containing the template source data.
286          */
287         @Override
288         protected InputStream getTemplateSource(String templateSourceName) 
289         	throws IOException {
290         	
291             TemplateSourceInfo tsInfo = (TemplateSourceInfo)mTemplateMap
292                                                 .get(templateSourceName);
293 
294             HttpClient client = getTemplateServerClient(tsInfo.server);
295             HttpClient.Response response = client
296                 .setURI(createTemplateServerRequest(tsInfo.server,tsInfo.name + ".tea"))
297                 .setPersistent(true).getResponse();            
298             InputStream in = response.getInputStream();
299             return in;
300         }
