17  package org.teatrove.teaservlet.stats;
18  
19  import java.util.Map;
20  import java.util.concurrent.ConcurrentHashMap;
21  import java.util.Collection;
22  
23  import org.teatrove.trove.util.PropertyMap;
24  
25  /**
26   * This class functions as the mediator for handling all http request
27   * level statistics gathering and derivation.
28   * 
29   * This class logs template requests with the TemplateStats object
30   * associated with a given template.
31   * 
32   * @author Scott Jappinen
33   */
34  public class TeaServletRequestStats {
35  
36  	public static final int DEFAULT_RAW_WINDOW_SIZE = 100;
37  	public static final int DEFAULT_AGGREGATE_WINDOW_SIZE = 50;
38  
39      private static TeaServletRequestStats mInstance = null;
40      private Map<String, TemplateStats> mStatsMap = new ConcurrentHashMap<String, TemplateStats>(100);
41  
42      private int mRawWindowSize = DEFAULT_RAW_WINDOW_SIZE;
43      private int mAggregateWindowSize = DEFAULT_AGGREGATE_WINDOW_SIZE;
44  
45      /**
46       * Returns a static instance of TeaServletRequestStats.
47       * 
48       * @return the tea servlet request stats.
49       */
50      public static TeaServletRequestStats getInstance() {
51          if (mInstance == null)
52              mInstance = new TeaServletRequestStats();
53          return mInstance;
54      }
55      
56      /**
57       * This method applies properties from a stats block
58       * in the tea servlet config.
59       * 
60       * @param properties
61       */
62      public void applyProperties(PropertyMap properties) {
63      	if (properties != null) {
64  	    	mRawWindowSize = properties.getInt
65  	    		("rawWindowSize", DEFAULT_RAW_WINDOW_SIZE);
66  	    	mAggregateWindowSize = properties.getInt
67  	    		("aggregateWindowSize", DEFAULT_AGGREGATE_WINDOW_SIZE);
68  	    	reset();
69      	}
70      }
71      
72      /**
73       * Returns the template stats for a given template name.
74       * 
75       * @param fullTemplateName the full name of the template with '.'s as
76       *    delimeters.
77       * @return the template stats
78       */
79      public TemplateStats getStats(String fullTemplateName) {
80          TemplateStats stats = (TemplateStats) mStatsMap.get(fullTemplateName);
81          if (stats == null) {
82              stats = new TemplateStats(fullTemplateName, mRawWindowSize, mAggregateWindowSize);
83              mStatsMap.put(fullTemplateName, stats);
84          }
85          return stats;
86      }
87      
88      /**
89       * Returns an array of template stats.
90       * 
91       * Returns the template raw and aggregate statistics so as to
92       * better understand the performance of templates through time.
93       * 
94       * @return the template stats for this given template.
95       */
96      public TemplateStats[] getTemplateStats() {
97      	Collection<TemplateStats> collection = mStatsMap.values();
98      	TemplateStats[] result = null;
99      	if (collection.size() > 0) {
100     		result = collection.toArray(new TemplateStats[collection.size()]);
101     	}
102     	return result;
103     }
104 
105     /**
106      * Logs an invokation of an http template invocation.
107      * 
108      * @param fullTemplateName the name of the template
109      * @param startTime the time the template was invoked.
110      * @param stopTime the time the template completed.
111      * @param contentLength the content length of the result of the template.
112      * @param params the parameter values the template was invoked with.
113      */
114     public void log(String fullTemplateName, long startTime, long stopTime, long contentLength, Object[] params) {
115     	//System.out.println(fullTemplateName + ", " + (stopTime-startTime) + ", " + contentLength);
116         TemplateStats stats = (TemplateStats) mStatsMap.get(fullTemplateName);
117         if (stats == null) {
118             stats = new TemplateStats(fullTemplateName, mRawWindowSize, mAggregateWindowSize);
119             mStatsMap.put(fullTemplateName, stats);
120         }
121         stats.log(startTime, stopTime, contentLength, params);
122     }
123 
124     /**
125      * Returns the template name -> template stats map.
126      * 
127      * @return the mapping of template names to stats.
128      */
129     public Map<String, TemplateStats> getStatisticsMap() {
130         return mStatsMap;
131     }
132     
133     /**
134      * Sets the raw window size.
135      * 
136      * Resets all statistics.
137      * 
138      * @param rawWindowSize
139      */
140     public void setRawWindowSize(int rawWindowSize) {
141     	mRawWindowSize = rawWindowSize;
142     	reset();
143     }
144     
145     /**
146      * Returns the raw window size.
147      * 
148      * This defines the number of raw data entries that will be
149      * kept in memory for all templates.
150      * 
151      * @return the raw window size.
152      */
153     public int getRawWindowSize() {
154     	return mRawWindowSize;
155     }
156     
157     /**
158      * Returns the aggregate window size.
159      * 
160      * This defines the number of aggregate intervals that will be
161      * kept in memory for all templates.
162      * 
163      * @return
164      */
165     public int getAggregateWindowSize() {
166     	return mAggregateWindowSize;
167     }
168     
169     /**
170      * Sets the aggregate interval size.
171      * 
172      * Resets all statistics.
173      * 
174      * @param aggregateWindowSize
175      */
176     public void setAggregateWindowSize(int aggregateWindowSize) {
177     	mAggregateWindowSize = aggregateWindowSize;
178     	reset();
179     }
180 
181     /**
182      * Resets all raw data and aggregate interval statistics.
183      */
184     public void reset() {
185         mStatsMap = new ConcurrentHashMap<String, TemplateStats>(100);
186     }
187 
188     /**
189      * Resets all raw data and aggregate interval statistics.
190      * 
191      * @param initialMapSize the initial size of the template name / stats map
192      */
193     public void reset(int initialMapSize) {
194         mStatsMap = new ConcurrentHashMap<String, TemplateStats>(initialMapSize);
195     }
196 }
