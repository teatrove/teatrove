26      public String capWords(String value) {
27          StringTokenizer st = new StringTokenizer(value);
28          StringBuffer buffer = new StringBuffer();
29          int count = st.countTokens();
30          for (int i = 0; i < count; i++) {
31              StringBuffer tempBuffer = new StringBuffer();
32              String word = st.nextElement().toString();
33              // first letter
34              tempBuffer.append(word.substring(0, 1).toUpperCase());
35              // rest of sentence
36              tempBuffer.append( word.substring(1, word.length()).toLowerCase());
37              buffer.append(tempBuffer.append(" "));
38          }
39          return buffer.toString().trim();
40      }

47      public String[] splitString(String text,int maxChars,String[] splitValues) {
48          String restOfText = text;
49          List<String> list = new ArrayList<String>();
50          while (restOfText.length() > maxChars) {
51              String thisLine = restOfText.substring(0,maxChars);
52              int lineEnd = 0;
53              for (int i = 0; i < splitValues.length;i++) {
54                  int lastIndex = thisLine.lastIndexOf(splitValues[i]);
55                  if (lastIndex > lineEnd) {
56                      lineEnd = lastIndex;
57                  }
58              }
59              if (lineEnd == 0) {
60                  list.add(thisLine);
61                  restOfText = restOfText.substring(maxChars);
62              }
63              else {
64                  list.add(thisLine.substring(0,lineEnd));
65                  restOfText = restOfText.substring(lineEnd);
66              }
67  
68          }
69          list.add(restOfText);
70          return (String[]) list.toArray(new String[list.size()]);
71      }

73      public String replace(String str, String pattern, String replace) {
74          int s = 0;
75          int e = 0;
76          StringBuffer result = new StringBuffer();
77  
78          while ((e = str.indexOf(pattern, s)) >= 0) {
79              result.append(str.substring(s, e));
80              result.append(replace);
81              s = e + pattern.length();
82          }
83          result.append(str.substring(s));
84          return result.toString();
85      }

87      public String[] tokenize(String string, String token) {
88          String[] result;
89          try {
90              StringTokenizer st = new StringTokenizer(string, token);
91              List<String> list = new ArrayList<String>();
92              while (st.hasMoreTokens()) {
93                  list.add(st.nextToken());
94              }
95              result = (String[]) list.toArray(new String[list.size()]);
96          } catch (Exception e) {
97              result = null;
98          }
99          return result;
100     }

116     public String[] removeFromStringArray(String[] strArray, String str) {
117         String[] result = null;
118         if (strArray != null && strArray.length > 0) {
119             List<String> list = new ArrayList<String>();
120             for (int i=0; i < strArray.length; i++) {
121                 if (!strArray[i].equals(str)) {
122                     list.add(strArray[i]);
123                 }
124             }
125             int size = list.size();
126             if (size > 0) {
127                 result = (String[]) list.toArray(new String[size]);
128             }
129         }
130         return result;
131     }

133     public String[] removeFromStringArray(String[] strArray, int index) {
134         String[] result = null;
135         if (strArray != null && index > 0 && index < strArray.length) {
136             result = new String[strArray.length - 1];
137             for (int i=0, j=0; i < strArray.length; i++) {
138                 if (i != index) {
139                     result[j++] = strArray[i];
140                 }
141             }
142         }
143         return result;
144     }
