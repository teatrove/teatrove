163     public void execute() throws MojoExecutionException, MojoFailureException {
164 
165         final Log logger = getLog();
166 
167         // Create the Helper for the Context Class Builder
168         ContextClassBuilderHelper helper =
169                 new DefaultContextClassBuilderHelper(
170                         session,
171                         new TeaCompilerExpressionEvaluator(session, translator, project),
172                         logger, container, this.getClass().getClassLoader(), rootPackage
173                 );
174 
175         // Merge the contexts
176         final Class contextClass;
177         try {
178             if(this.context == null) {
179                 if(contextClassBuilder == null) {
180                     throw new MojoExecutionException("Either context or contextClassBuilder parameter is required.");
181                 }
182                 contextClass = contextClassBuilder.getContextClass(helper);
183             } else {
184                 contextClass = Class.forName(this.context);
185             }
186         } catch (ContextClassBuilderException e) {
187             throw new MojoExecutionException("Unable to find or create the Context.", e);
188         } catch (ClassNotFoundException e) {
189             throw new MojoExecutionException("Unable to load the Context.", e);
190         }
191 
192         final File realOutputDirectory = new File(outputDirectory, rootPackage.replace('.', File.separatorChar));
193 
194         realOutputDirectory.mkdirs();
195 
196         if (sourceDirectories == null || sourceDirectories.length == 0) {
197             sourceDirectories = new File[]{defaultSourceDirectory};
198         } else {
199             // Filter out any that don't exist
200             List<File> existing = new ArrayList<File>(sourceDirectories.length);
201             for (File sourceDirectory : sourceDirectories) {
202                 if (sourceDirectory.exists()) {
203                     existing.add(sourceDirectory);
204                 } else if (logger.isDebugEnabled()) {
205                     logger.debug("Removing source directory because it does not exist. [" + sourceDirectory + "].");
206                 }
207             }
208             sourceDirectories = existing.toArray(new File[existing.size()]);
209         }
210 
211         final FileCompiler compiler =
212                 new FileCompiler(sourceDirectories, rootPackage, realOutputDirectory, null, encoding);
213 
214         compiler.setClassLoader(contextClass.getClassLoader());
215         compiler.setRuntimeContext(contextClass);
216         compiler.setForceCompile(force);
217         compiler.addErrorListener(new ErrorListener() {
218             public void compileError(ErrorEvent e) {
219                 logger.error(e.getDetailedErrorMessage());
220             }
221         });
222         compiler.setExceptionGuardianEnabled(guardian);
223 
224         final String[] names;
225         try {
226             if (includes == null || includes.length == 0) {
227                 names = compiler.compileAll(true);
228             } else {
229                 names = compiler.compile(includes);
230             }
231         } catch (IOException e) {
232             throw new MojoExecutionException("I/O error while compiling templates.", e);
233         }
234 
235         final int errorCount = compiler.getErrorCount();
236 
237         if (errorCount > 0) {
238             String msg = errorCount + " error" + (errorCount != 1 ? "s" : "");
239             if(failOnError) {
240                 throw new MojoFailureException(msg);
241             } else if(logger.isWarnEnabled()){
242                 logger.warn(msg);
243             }
244         }
245 
246         if (logger.isInfoEnabled()) {
247             logger.info("Compiled the following templates:");
248             Arrays.sort(names);
249             for (String name : names) {
250                 logger.info(name);
251             }
252         }
253     }
