17  package org.teatrove.tea.util;
18  
19  import java.io.IOException;
20  import java.io.PrintStream;
21  import java.io.BufferedReader;
22  import org.teatrove.tea.compiler.SourceInfo;
23  import org.teatrove.tea.compiler.CompilationUnit;
24  import org.teatrove.tea.compiler.ErrorEvent;
25  import org.teatrove.tea.compiler.ErrorListener;
26  import org.teatrove.trove.io.LinePositionReader;
27  
28  /**
29   * ConsoleErrorReporter takes ErrorEvents and prints
30   * detailed messages to a PrintStream. When no longer needed, close the
31   * ConsoleErrorReporter to ensure all open resources (except the PrintStream)
32   * are closed. 
33   *
34   * @author Brian S O'Neill
35   */
36  public class ConsoleErrorReporter implements ErrorListener {
37      private PrintStream mOut;
38      private LinePositionReader mPositionReader;
39      private CompilationUnit mPositionReaderUnit;
40  
41      public ConsoleErrorReporter(PrintStream out) {
42          mOut = out;
43      }
44  
45      /**
46       * Closes all open resources.
47       */
48      public void close() throws IOException {
49          if (mPositionReader != null) {
50              mPositionReader.close();
51          }
52  
53          mPositionReader = null;
54          mPositionReaderUnit = null;
55      }
56  
57      public void compileError(ErrorEvent e) {
58          mOut.println(e.getDetailedErrorMessage());
59  
60          SourceInfo info = e.getSourceInfo();
61          CompilationUnit unit = e.getCompilationUnit();
62  
63          try {
64              if (unit != null && info != null) {
65                  int line = info.getLine();
66                  int start = info.getStartPosition();
67                  int end = info.getEndPosition();
68  
69                  if (mPositionReader == null ||
70                      mPositionReaderUnit != unit ||
71                      mPositionReader.getLineNumber() >= line) {
72  
73                      mPositionReaderUnit = unit;
74                      mPositionReader = new LinePositionReader
75                          (new BufferedReader(unit.getReader()));
76                  }
77  
78                  mPositionReader.skipForwardToLine(line);
79                  int position = mPositionReader.getNextPosition();
80  
81                  String lineStr = mPositionReader.readLine();
82                  lineStr = LinePositionReader.cleanWhitespace(lineStr);
83                  mOut.println(lineStr);
84  
85                  int indentSize = start - position;
86                  String indent =
87                      LinePositionReader.createSequence(' ', indentSize);
88  
89                  int markerSize = end - start + 1;
90                  String marker =
91                      LinePositionReader.createSequence('^', markerSize);
92  
93                  mOut.print(indent);
94                  mOut.println(marker);
95                  mOut.println();
96              }
97          }
98          catch (IOException ex) {
99              Thread t = Thread.currentThread();
100             t.getThreadGroup().uncaughtException(t, ex);
101         }
102     }
103 
104     protected void finalize() throws Throwable {
105         close();
106     }
107 }
