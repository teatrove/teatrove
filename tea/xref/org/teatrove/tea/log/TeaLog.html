29  /**
30   * The TeaLog class extends a regular log.  It simplifies stack traces for the
31   * TeaServlet so it is easier for template authors to understand.  It strips off
32   * irrelevant stack trace elements in the stack trace leaving only the template
33   * and application elements.  These are the only elements that are relevant
34   * unless there is a bug in the Tea hosting environment - which never happens! 
35   * ;-)
36   *
37   * This TeaLog is designed to be used in hosting environments such as the
38   * TeaServlet and StaticTea.  These hosts use templates as the controller so
39   * any stack traces elements before the templates are irrelevant.  Authors of
40   * systems that don't use templates as the controller will probably not want to
41   * use the log as their stack traces will not return to them enough information.
42   *
43   * Some examples are listed below.
44   *
45   * The Trove Log class displays exceptions as:
46   *    java.lang.NullPointerException
47   *  	at org.teatrove.teaservlet.template.Test1NullPointer.execute(Test1NullPointer.tea:2)
48   *  	at sun.reflect.NativeMethodAccessorImpl.invoke0(Native Method)
49   *  	at sun.reflect.NativeMethodAccessorImpl.invoke(NativeMethodAccessorImpl.java:39)
50   *  	at sun.reflect.DelegatingMethodAccessorImpl.invoke(DelegatingMethodAccessorImpl.java:25)
51   *  	at java.lang.reflect.Method.invoke(Method.java:324)
52   *  	at org.teatrove.tea.runtime.TemplateLoader$TemplateImpl.execute(TemplateLoader.java:270)
53   *  	at org.teatrove.tea.engine.TemplateSourceImpl$TemplateImpl.execute(TemplateSourceImpl.java:838)
54   *  	at org.teatrove.teaservlet.TeaServlet.processTemplate(TeaServlet.java:594)
55   *  	at org.teatrove.teaservlet.TeaServlet.doGet(TeaServlet.java:259)
56   *  	at org.teatrove.barista.servlet.TQTeaServlet.doGet(TQTeaServlet.java:117)
57   *  	at javax.servlet.http.HttpServlet.service(HttpServlet.java:740)
58   *  	at javax.servlet.http.HttpServlet.service(HttpServlet.java:853)
59   *  	at org.teatrove.barista.http.HttpServletDispatcher$ServletFilterChainTransaction.doFilter(HttpServletDispatcher.java:1436)
60   *  	at org.teatrove.barista.http.HttpServletDispatcher$ServletFilterChainTransaction.service(HttpServletDispatcher.java:1409)
61   *  	at org.teatrove.trove.util.tq.TransactionQueue$Worker.run(TransactionQueue.java:671)
62   *  	at org.teatrove.trove.util.ThreadPool$PooledThread.run(ThreadPool.java:676)
63   *  
64   * The TeaLog class displays exceptions as:
65   *    java.lang.NullPointerException
66   *  	at line 2 of template Test1NullPointer.tea
67   *
68   * @author Reece Wilton
69   */

72      private static final long serialVersionUID = 1L;

74      private final String TEA_EXCEPTION = ".tea:";

76      /**
77       * @param name
78       * @param parent
79       */
80      public TeaLog(Log parent) {
81          super(parent.getName(), parent);
82      }

84      public synchronized void info(Throwable t) {
85          TeaStackTraceLine[] lines = getTeaStackTraceLines(t);
86          if (lines == null) {
87              super.info(t);
88          }
89          else {
90              if (isEnabled() && isInfoEnabled()) {
91                  dispatchLogTeaStackTrace(new TeaLogEvent(this, LogEvent.INFO_TYPE,
92                                                           lines));
93              }
94          }
95      }
96  
97      public synchronized void error(Throwable t) {
98          TeaStackTraceLine[] lines = getTeaStackTraceLines(t);
99          if (lines == null) {
100             super.error(t);
101         }
102         else {
103             if (isEnabled() && isErrorEnabled()) {
104                 dispatchLogTeaStackTrace(new TeaLogEvent(this, LogEvent.ERROR_TYPE,
105                                                          lines));
106             }
107         }
108     }
109 
110     public synchronized void warn(Throwable t) {
111         TeaStackTraceLine[] lines = getTeaStackTraceLines(t);
112         if (lines == null) {
113             super.warn(t);
114         }
115         else {
116             if (isEnabled() && isWarnEnabled()) {
117                 dispatchLogTeaStackTrace(new TeaLogEvent(this, LogEvent.WARN_TYPE,
118                                                          lines));
119             }
120         }
121     }
122 
123     public synchronized void debug(Throwable t) {
124         TeaStackTraceLine[] lines = getTeaStackTraceLines(t);
125         if (lines == null) {
126             super.debug(t);
127         }
128         else {
129             if (isEnabled() && isDebugEnabled()) {
130                 dispatchLogTeaStackTrace(new TeaLogEvent(this, LogEvent.DEBUG_TYPE,
131                                                          lines));
132             }
133         }
134     }

136     private void dispatchLogTeaStackTrace(TeaLogEvent e) {
137         int size = mListeners.size();
138         try {
139             for (int i=0; i<size; i++) {
140                 LogListener listener = (LogListener) mListeners.get(i);
141                 if (listener instanceof TeaLogListener) {
142                     ((TeaLogListener)listener).logTeaStackTrace(e);
143                 }
144                 else {
145                     e.setMessage(printTeaStackTraceLines(e.getLines()));
146                     listener.logMessage(e);
147                 }
148             }
149         }
150         catch (IndexOutOfBoundsException ex) {
151         }
152     }

154     /**
155      * Prints the stack trace lines to a String.
156      *
157      * @param lines
158      * @return  the output of the stack trace lines as a string
159      */
160     private String printTeaStackTraceLines(TeaStackTraceLine[] lines) {
161 
162         String result = "";
163         for (int line = 0; line < lines.length; line++) {
164             if (line > 0) {
165                 result += '\n';
166             }
167             result += lines[line].toString();
168         }
169 
170         return result;
171     }

173     /**
174      * Splits the stack trace into separate lines and extracts the template
175      * name and line number.
176      *
177      * @param t
178      * @return  the separated stack trace lines
179      */
180     private TeaStackTraceLine[] getTeaStackTraceLines(Throwable t) {
181 
182         // grab the existing stack trace
183         StringWriter stackTraceGrabber = new StringWriter();
184         t.printStackTrace(new PrintWriter(stackTraceGrabber));
185         String stackTrace = stackTraceGrabber.toString();
186         int extensionIndex = stackTrace.lastIndexOf(TEA_EXCEPTION);
187         boolean isTeaException = extensionIndex != -1;
188         if (isTeaException) {
189 
190             // trim off all lines after the last template exception
191             int endIndex = stackTrace.indexOf('\n', extensionIndex);
192             int endRIndex = stackTrace.indexOf('\r', extensionIndex);
193 
194             if(endRIndex>-1 && endRIndex<endIndex)
195                 endIndex=endRIndex;
196 
197             if (endIndex <= 0) {
198                 endIndex = stackTrace.length();
199             }
200 
201             stackTrace = stackTrace.substring(0, endIndex);
202 
203             // parse each line
204             List<TeaStackTraceLine> teaStackTraceLines =
205                 new ArrayList<TeaStackTraceLine>();
206             StringTokenizer tokenizer = new StringTokenizer(stackTrace, "\n");
207             while (tokenizer.hasMoreElements()) {
208                 String line = (String)tokenizer.nextElement();
209                 if (line.indexOf(TEA_EXCEPTION) != -1) {
210                     /*
211                     TODO: make sure this works for lines that don't have
212                           line numbers.  Look in ESPN logs for examples.
213                     at org.teatrove.teaservlet.template.schedule.substitute(schedule.tea:78)
214                     at org.teatrove.teaservlet.template.shell.story.frame.substitute(shell/story/frame.tea)
215                     */
216                     String tempLine = line;
217 
218                     int bracket = tempLine.indexOf('(');
219                     tempLine = tempLine.substring(bracket + 1);
220                     bracket = tempLine.indexOf(')');
221                     tempLine = tempLine.substring(0, bracket);
222                     int colonIndex = tempLine.indexOf(':');
223                     String templateName = null;
224                     Integer lineNumber = null;
225                     if (colonIndex >= 0) {
226                         templateName = tempLine.substring(0, colonIndex);
227                         lineNumber = new Integer(tempLine.substring(colonIndex + 1));
228                     }
229                     else {
230                         templateName = tempLine;
231                         lineNumber = null;
232                     }
233                     teaStackTraceLines.add(new TeaStackTraceLine(templateName,
234                                                                  lineNumber,
235                                                                  line));
236                 }
237                 else {
238                     teaStackTraceLines.add(new TeaStackTraceLine(null, null, line));
239                 }
240             }
241             return (TeaStackTraceLine[]) teaStackTraceLines.toArray(
242                               new TeaStackTraceLine[teaStackTraceLines.size()]);
243         }
244         else {
245             return null;
246         }
247     }

249     /**
250      * The TeaStackTraceLine class contains the values of a stack trace line.
251      * It contains the template name and line number.  It also includes the
252      * original stack trace line from the JVM.
253      *
254      * @author Reece Wilton
255      */
256     public class TeaStackTraceLine {
257 
258         private String mTemplateName;
259         private Integer mLineNumber;
260         private String mLine;
261 
262         public TeaStackTraceLine(String templateName,
263                                  Integer lineNumber,
264                                  String line) {
265             mTemplateName = templateName;
266             mLineNumber = lineNumber;
267             mLine = line;
268         }
269 
270         /**
271          * @return  the original stack trace line
272          */
273         public String getLine() {
274             return mLine;
275         }
276 
277         /**
278          * @return  the Tea template line number.  May be null even if a
279          * template name exists.
280          */
281         public Integer getLineNumber() {
282             return mLineNumber;
283         }
284 
285         /**
286          * @return  the template name.  Will be null if the line isn't for a
287          * Tea template.
288          */
289         public String getTemplateName() {
290             return mTemplateName;
291         }
292 
293         public String toString() {
294             if (mLineNumber != null) {
295                 return "\tat " +
296                        ((mLineNumber != null) ? "line " + mLineNumber + " of "
297                                                 : "unknown line of ")
298                        + "template " + mTemplateName;
299             }
300             else {
301                 return mLine;
302             }
303         }
304     }
