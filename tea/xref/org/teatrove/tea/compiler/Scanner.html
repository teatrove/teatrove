24  /**
25   * A Scanner breaks up a source file into its basic elements, called
26   * {@link Token Tokens}. Add an {@link ErrorListener} to capture any syntax
27   * errors detected by the Scanner.
28   *
29   * @author Brian S O'Neill
30   */
31  public class Scanner {
32      private SourceReader mSource;
33      private CompilationUnit mUnit;
34  
35      private boolean mEmitSpecial;
36  
37      /** StringBuffer for temporary use. */
38      private StringBuilder mWord = new StringBuilder(20);
39  
40      /** The scanner supports any amount of lookahead. */
41      private Stack<Token> mLookahead = new Stack<Token>();
42      
43      private Token mEOFToken;
44  
45      private Vector<ErrorListener> mListeners = new Vector<ErrorListener>(1);
46      private int mErrorCount = 0;
47  
48      private MessageFormatter mFormatter;
49  
50      public Scanner(SourceReader in) {
51          this(in, null);
52      }
53  
54      public Scanner(SourceReader in, CompilationUnit unit) {
55          mSource = in;
56          mUnit = unit;
57          mFormatter = MessageFormatter.lookup(this);
58      }
59  
60      public void addErrorListener(ErrorListener listener) {
61          mListeners.addElement(listener);
62      }
63  
64      public void removeErrorListener(ErrorListener listener) {
65          mListeners.removeElement(listener);
66      }
67  
68      private void dispatchParseError(ErrorEvent e) {
69          mErrorCount++;
70  
71          synchronized (mListeners) {
72              for (int i = 0; i < mListeners.size(); i++) {
73                  mListeners.elementAt(i).compileError(e);
74              }
75          }
76      }
77  
78      private void error(String str, SourceInfo info) {
79          dispatchParseError
80              (new ErrorEvent(this, mFormatter.format(str), info, mUnit));
81      }
82  
83      private void error(String str) {
84          error(str, new SourceInfo(mSource.getLineNumber(),
85                                    mSource.getStartPosition(),
86                                    mSource.getEndPosition()));
87      }

89      /**
90       * Passing true causes Scanner to emit additional tokens that should not
91       * be bassed into a Parser. These are {@link Token.COMMENT},
92       * {@link Token.ENTER_CODE}, and {@link Token.ENTER_TEXT}. By default,
93       * these special tokens are not emitted.
94       */
95      public void emitSpecialTokens(boolean enable) {
96          mEmitSpecial = enable;
97      }

99      /**
100      * Returns EOF as the last token.
101      */
102     public synchronized Token readToken() throws IOException {
103         if (mLookahead.empty()) {
104             return scanToken();
105         }
106         else {
107             return mLookahead.pop();
108         }
109     }
110 
111     /** 
112      * Returns EOF as the last token.
113      */
114     public synchronized Token peekToken() throws IOException {
115         if (mLookahead.empty()) {
116             return mLookahead.push(scanToken());
117         }
118         else {
119             return mLookahead.peek();
120         }
121     }
122 
123     public synchronized void unreadToken(Token token) throws IOException {
124         mLookahead.push(token);
125     }
126 
127     public void close() throws IOException {
128         mSource.close();
129     }

131     public int getErrorCount() {
132         return mErrorCount;
133     }

135     private Token scanToken() throws IOException {
136         int c;
137         int peek;
138         
139         int startPos;
140 
141         while ((c = mSource.read()) != -1) {
142             switch (c) {
143 
144             case SourceReader.ENTER_TEXT:
145                 Token enter;
146                 if (mEmitSpecial) {
147                     enter = makeStringToken(Token.ENTER_TEXT,
148                                             mSource.getEndTag());
149                 }
150                 else {
151                     enter = null;
152                 }
153 
154                 Token t = scanText(c);
155 
156                 if (mEmitSpecial) {
157                     if (t.getStringValue().length() > 0) {
158                         mLookahead.push(t);
159                     }
160                     return enter;
161                 }
162 
163                 if (t.getStringValue().length() == 0) {
164                     continue;
165                 }
166 
167                 return t;
168 
169             case SourceReader.ENTER_CODE:
170                 // Entering code while in code is illegal. Just let the parser
171                 // deal with it.
172                 return makeStringToken(Token.ENTER_CODE,
173                                        mSource.getBeginTag());
174 
175             case '(':
176                 return makeToken(Token.LPAREN);
177             case ')':
178                 return makeToken(Token.RPAREN);
179 
180             case '{':
181                 return makeToken(Token.LBRACE);
182             case '}':
183                 return makeToken(Token.RBRACE);
184 
185             case '[':
186                 return makeToken(Token.LBRACK);
187             case ']':
188                 return makeToken(Token.RBRACK);
189                 
190             case ';':
191                 return makeToken(Token.SEMI);
192                 
193             case ',':
194                 return makeToken(Token.COMMA);
195                 
196             case ':':
197                 return makeToken(Token.COLON);
198                 
199             case '?':
200                 return makeToken(Token.QUESTION);
201 
202             case '.':
203                 peek = mSource.peek();
204 
205                 if (peek >= '0' && peek <= '9') {
206                     error("number.decimal.start");
207                     return scanNumber(c);
208                 }
209                 else if (peek == '.') {
210                     startPos = mSource.getStartPosition();
211                     // read the second '.'
212                     mSource.read();
213 
214                     peek = mSource.peek();
215                     if (peek == '.') {
216                         // read the third '.'
217                         mSource.read();
218                         return makeToken(Token.ELLIPSIS, startPos);
219                     }
220                     else {
221                         return makeToken(Token.DOTDOT, startPos);
222                     }
223                 }
224                 else {
225                     return makeToken(Token.DOT);
226                 }
227                 
228             case '#':
229                 peek = mSource.peek();
230                 
231                 if (peek == '#') {
232                     startPos = mSource.getStartPosition();
233                     mSource.read();
234                     return makeToken(Token.DOUBLE_HASH, startPos);
235                 }
236                 else {
237                     return makeToken(Token.HASH);
238                 }
239                 
240             case '!':
241                 if (mSource.peek() == '=') {
242                     startPos = mSource.getStartPosition();
243                     mSource.read();
244                     return makeToken(Token.NE, startPos);
245                 }
246                 else {
247                     return makeStringToken(Token.UNKNOWN,
248                                            String.valueOf((char)c));
249                 }
250                 
251             case '<':
252                 if (mSource.peek() == '=') {
253                     startPos = mSource.getStartPosition();
254                     mSource.read();
255                     if (mSource.peek() == '>') {
256                         mSource.read();
257                         return makeToken(Token.SPACESHIP, startPos);
258                     }
259                     else {
260                         return makeToken(Token.LE, startPos);
261                     }
262                 }
263                 else {
264                     return makeToken(Token.LT);
265                 }
266                 
267             case '>':
268                 if (mSource.peek() == '=') {
269                     startPos = mSource.getStartPosition();
270                     mSource.read();
271                     return makeToken(Token.GE, startPos);
272                 }
273                 else {
274                     return makeToken(Token.GT);
275                 }
276 
277             case '=':
278                 if (mSource.peek() == '=') {
279                     startPos = mSource.getStartPosition();
280                     mSource.read();
281                     return makeToken(Token.EQ, startPos);
282                 }
283                 if (mSource.peek() == '>') {
284                     startPos = mSource.getStartPosition();
285                     mSource.read();
286                     return makeToken(Token.EQUAL_GREATER, startPos);
287                 }
288                 else {
289                     return makeToken(Token.ASSIGN);
290                 }
291 
292             case '&':
293                 return makeToken(Token.CONCAT);
294 
295             case '+':
296                 return makeToken(Token.PLUS);
297                 
298             case '-':
299                 return makeToken(Token.MINUS);
300                 
301             case '*':
302                 if (mSource.peek() == '.') {
303                     startPos = mSource.getStartPosition();
304                     mSource.read();
305                     return makeToken(Token.SPREAD, startPos);
306                 }
307                 else {
308                     return makeToken(Token.MULT);
309                 }
310                 
311             case '%':
312                 return makeToken(Token.MOD);
313                 
314             case '/':
315                 startPos = mSource.getStartPosition();
316                 peek = mSource.peek();
317 
318                 if (peek == '*') {
319                     mSource.read();
320                     mSource.ignoreTags(true);
321                     t = scanMultiLineComment(startPos);
322                     mSource.ignoreTags(false);
323                     if (mEmitSpecial) {
324                         return t;
325                     }
326                     else {
327                         continue;
328                     }
329                 }
330                 else if (peek == '/') {
331                     mSource.read();
332                     t = scanOneLineComment(startPos);
333                     if (mEmitSpecial) {
334                         return t;
335                     }
336                     else {
337                         continue;
338                     }
339                 }
340                 else {
341                     return makeToken(Token.DIV);
342                 }
343                 
344             case '\"':
345             case '\'':
346                 mSource.ignoreTags(true);
347                 t = scanString(c);
348                 mSource.ignoreTags(false);
349                 return t;
350                 
351             case '0': case '1': case '2': case '3': case '4': 
352             case '5': case '6': case '7': case '8': case '9':
353                 return scanNumber(c);
354                 
355             case 'a': case 'b': case 'c': case 'd': case 'e':
356             case 'f': case 'g': case 'h': case 'i': case 'j':
357             case 'k': case 'l': case 'm': case 'n': case 'o':
358             case 'p': case 'q': case 'r': case 's': case 't':
359             case 'u': case 'v': case 'w': case 'x': case 'y':
360             case 'z':
361             case 'A': case 'B': case 'C': case 'D': case 'E':
362             case 'F': case 'G': case 'H': case 'I': case 'J':
363             case 'K': case 'L': case 'M': case 'N': case 'O':
364             case 'P': case 'Q': case 'R': case 'S': case 'T':
365             case 'U': case 'V': case 'W': case 'X': case 'Y':
366             case 'Z': case '_':
367                 return scanIdentifier(c);
368 
369             case ' ': 
370             case '\0':
371             case '\t': 
372             case '\r': 
373             case '\n':
374                 continue;
375 
376             default:
377                 if (Character.isWhitespace((char)c)) {
378                     continue;
379                 }
380 
381                 if (Character.isLetter((char)c)) {
382                     return scanIdentifier(c);
383                 }
384                 else {
385                     return makeStringToken(Token.UNKNOWN,
386                                            String.valueOf((char)c));
387                 }
388             }
389         }
390 
391         if (mEOFToken == null) {
392             mEOFToken = makeToken(Token.EOF);
393         }
394         
395         return mEOFToken;
396     }

398     // The ENTER_TEXT code has already been scanned when this is called.
399     private Token scanText(int c) throws IOException {
400         // Read first character in text so that source info does not include
401         // tags.
402         c = mSource.read();
403 
404         int startLine = mSource.getLineNumber();
405         int startPos = mSource.getStartPosition();
406         int endPos = mSource.getEndPosition();
407         StringBuilder buf = new StringBuilder(256);
408 
409         while (c != -1) {
410             if (c == SourceReader.ENTER_CODE) {
411                 if (mEmitSpecial) {
412                     mLookahead.push(makeStringToken(Token.ENTER_CODE,
413                                                     mSource.getBeginTag()));
414                 }
415                 break;
416             }
417             else if (c == SourceReader.ENTER_TEXT) {
418                 buf.append(mSource.getEndTag());
419             }
420             else {
421                 buf.append((char)c);
422             }
423 
424             if (mSource.peek() < 0) {
425                 endPos = mSource.getEndPosition();
426             }
427 
428             c = mSource.read();
429         };
430 
431         if (c == -1) {
432             // If the last token in the source file is text, trim all trailing
433             // whitespace from it.
434 
435             int length = buf.length();
436             
437             int i;
438             for (i = length - 1; i >= 0; i--) {
439                 if (buf.charAt(i) > ' ') {
440                     break;
441                 }
442             }
443 
444             buf.setLength(i + 1);
445         }
446 
447         String str = buf.toString();
448         return new StringToken(startLine, startPos, endPos,
449                                Token.STRING, str);
450     }

452     private Token scanString(int delimiter) throws IOException {
453         int c;
454         int startLine = mSource.getLineNumber();
455         int startPos = mSource.getStartPosition();
456         mWord.setLength(0);
457         
458         while ( (c = mSource.read()) != -1 ) {
459             if (c == delimiter) {
460                 break;
461             }
462 
463             if (c == '\n' || c == '\r') {
464                 error("string.newline");
465                 break;
466             }
467             
468             if (c == '\\') {
469                 int next = mSource.read();
470                 switch (next) {
471                 case '0':
472                     c = '\0';
473                     break;
474                 case 'b':
475                     c = '\b';
476                     break;
477                 case 't':
478                     c = '\t';
479                     break;
480                 case 'n':
481                     c = '\n';
482                     break;
483                 case 'f':
484                     c = '\f';
485                     break;
486                 case 'r':
487                     c = '\r';
488                     break;
489                 case '\\':
490                     c = '\\';
491                     break;
492                 case '\'':
493                     c = '\'';
494                     break;
495                 case '\"':
496                     c = '\"';
497                     break;
498                 default:
499                     error("escape.code");
500                     c = next;
501                     break;
502                 }
503             }
504             
505             mWord.append((char)c);
506         }
507 
508         if (c == -1) {
509             error("string.eof");
510         }
511 
512         Token t = new StringToken(startLine,
513                                   startPos, 
514                                   mSource.getEndPosition(),
515                                   Token.STRING,
516                                   mWord.toString()); 
517         
518         return t;
519     }

521     // The first character has already been scanned when this is called.
522     private Token scanNumber(int c) throws IOException {
523         int startLine = mSource.getLineNumber();
524         int startPos = mSource.getStartPosition();
525         mWord.setLength(0);
526         
527         int errorPos = -1;
528 
529         // 0 is decimal int, 
530         // 1 is hex int, 
531         // 2 is decimal long, 
532         // 3 is hex long, 
533         // 4 is float, 
534         // 5 is double,
535         // 6 is auto-double by decimal
536         // 7 is auto-double by exponent ('e' or 'E')
537         int type = 0; 
538 
539         if (c == '0') {
540             if (mSource.peek() == 'x' || mSource.peek() == 'X') {
541                 type = 1;
542                 mSource.read(); // absorb the 'x'
543                 c = mSource.read(); // get the first digit after the 'x'
544             }
545         }
546 
547         for (; c != -1; c = mSource.read()) {
548             if (c == '.') {
549                 int peek = mSource.peek();
550                 if (peek == '.') {
551                     mSource.unread();
552                     break;
553                 }
554                 else {
555                     if (peek < '0' || peek > '9') {
556                         error("number.decimal.end");
557                     }
558 
559                     mWord.append((char)c);
560 
561                     if (type == 0) {
562                         type = 6;
563                     }
564                     else if (errorPos < 0) {
565                         errorPos = mSource.getStartPosition();
566                     }
567 
568                     continue;
569                 }
570             }
571 
572             if (c >= '0' && c <= '9') {
573                 mWord.append((char)c);
574 
575                 if (type == 2 || type == 3 || type == 4 || type == 5) {
576                     if (errorPos < 0) {
577                         errorPos = mSource.getStartPosition();
578                     }
579                 }
580 
581                 continue;
582             }
583 
584             if ((c >= 'a' && c <= 'f') || (c >= 'A' && c <= 'F')) {
585                 if (type == 1) {
586                     mWord.append((char)c);
587                     continue;
588                 }
589 
590                 if (c == 'f' || c == 'F') {
591                     if (type == 0 || type == 6 || type == 7) {
592                         type = 4;
593                         continue;
594                     }
595                 }
596                 else if (c == 'd' || c == 'D') {
597                     if (type == 0 || type == 6 || type == 7) {
598                         type = 5;
599                         continue;
600                     }
601                 }
602                 else if (c == 'e' || c == 'E') {
603                     if (type == 0 || type == 6) {
604                         mWord.append((char)c);
605                         type = 7;
606                         int peek = mSource.peek();
607                         if (peek == '+' || peek == '-') {
608                             mWord.append((char)mSource.read());
609                         }
610                         continue;
611                     }
612                 }
613 
614                 mWord.append((char)c);
615 
616                 if (errorPos < 0) {
617                     errorPos = mSource.getStartPosition();
618                 }
619 
620                 continue;
621             }
622 
623             if (c == 'l' || c == 'L') {
624                 if (type == 0) {
625                     type = 2;
626                 }
627                 else if (type == 1) {
628                     type = 3;
629                 }
630                 else {
631                     mWord.append((char)c);
632                     if (errorPos < 0) {
633                         errorPos = mSource.getStartPosition();
634                     }
635                 }
636 
637                 continue;
638             }
639 
640             if (Character.isLetterOrDigit((char)c)) {
641                 mWord.append((char)c);
642 
643                 if (errorPos < 0) {
644                     errorPos = mSource.getStartPosition();
645                 }
646             }
647             else {
648                 mSource.unread();
649                 break;
650             }
651         }
652         
653         String str = mWord.toString();
654         int endPos = mSource.getEndPosition();
655         Token token;
656 
657         if (errorPos >= 0) {
658             token = new StringToken
659                 (startLine, startPos, endPos, errorPos, Token.NUMBER, str);
660         }
661         else {
662             try {
663                 switch (type) {
664                 case 0:
665                 default:
666                     try {
667                         token = new IntToken
668                             (startLine, startPos, endPos,
669                              Integer.parseInt(str));
670                     }
671                     catch (NumberFormatException e) {
672                         token = new LongToken
673                             (startLine, startPos, endPos, Long.parseLong(str));
674                     }
675                     break;
676                 case 1:
677                     try {
678                         token = new IntToken
679                             (startLine, startPos, endPos, parseHexInt(str));
680                     }
681                     catch (NumberFormatException e) {
682                         token = new LongToken
683                             (startLine, startPos, endPos, parseHexLong(str));
684                     }
685                     break;
686                 case 2:
687                     token = new LongToken
688                         (startLine, startPos, endPos, Long.parseLong(str));
689                     break;
690                 case 3:
691                     token = new LongToken
692                         (startLine, startPos, endPos, parseHexLong(str));
693                     break;
694                 case 4:
695                     token = new FloatToken
696                         (startLine, startPos, endPos, Float.parseFloat(str));
697                     break;
698                 case 5:
699                 case 6:
700                 case 7:
701                     token = new DoubleToken
702                         (startLine, startPos, endPos, Double.parseDouble(str));
703                     break;
704                 }
705             }
706             catch (NumberFormatException e) {
707                 token = new IntToken(startLine, startPos, endPos, 0);
708                 error("number.range", token.getSourceInfo());
709             }
710         }
711 
712         return token;
713     }

715     private int parseHexInt(String str) {
716         if (str.length() > 8) {
717             // Strip off any leading zeros.
718             while (str.charAt(0) == '0') {
719                 str = str.substring(1);
720             }
721         }
722 
723         try {
724             return Integer.parseInt(str, 16);
725         }
726         catch (NumberFormatException e) {
727             if (str.length() == 8) {
728                 return (int)Long.parseLong(str, 16);
729             }
730             else {
731                 throw e;
732             }
733         }
734     }
735 
736     private long parseHexLong(String str) {
737         if (str.length() > 16) {
738             // Strip off any leading zeros.
739             while (str.charAt(0) == '0') {
740                 str = str.substring(1);
741             }
742         }
743 
744         try {
745             return Long.parseLong(str, 16);
746         }
747         catch (NumberFormatException e) {
748             if (str.length() == 16) {
749                 long v1 = Long.parseLong(str.substring(0, 8), 16);
750                 long v2 = Long.parseLong(str.substring(8), 16);
751                 return v1 << 32 + v2 & 0xffffffffL;
752             }
753             else {
754                 throw e;
755             }
756         }
757     }

759     // The first character has already been scanned when this is called.
760     private Token scanIdentifier(int c) throws IOException {
761         int startLine = mSource.getLineNumber();
762         int startPos = mSource.getStartPosition();
763         int endPos = mSource.getEndPosition();
764         mWord.setLength(0);
765         
766         mWord.append((char)c);
767 
768     loop:
769         while ( (c = mSource.peek()) != -1 ) {
770             switch (c) {
771             case 'a': case 'b': case 'c': case 'd': case 'e':
772             case 'f': case 'g': case 'h': case 'i': case 'j':
773             case 'k': case 'l': case 'm': case 'n': case 'o':
774             case 'p': case 'q': case 'r': case 's': case 't':
775             case 'u': case 'v': case 'w': case 'x': case 'y':
776             case 'z':
777             case 'A': case 'B': case 'C': case 'D': case 'E':
778             case 'F': case 'G': case 'H': case 'I': case 'J':
779             case 'K': case 'L': case 'M': case 'N': case 'O':
780             case 'P': case 'Q': case 'R': case 'S': case 'T':
781             case 'U': case 'V': case 'W': case 'X': case 'Y':
782             case 'Z': case '_': case '$':
783             case '0': case '1': case '2': case '3': case '4': 
784             case '5': case '6': case '7': case '8': case '9':
785                 mSource.read();
786                 endPos = mSource.getEndPosition();
787                 mWord.append((char)c);
788                 continue loop;
789             }
790                 
791             if (Character.isLetterOrDigit((char)c)) {
792                 mSource.read();
793                 endPos = mSource.getEndPosition();
794                 mWord.append((char)c);
795             }
796             else {
797                 break;
798             }
799         }
800         
801         int id = Token.findReservedWordID(mWord);
802         
803         Token t;
804 
805         if (id != Token.UNKNOWN) {
806             t = new Token(startLine, startPos, endPos, id);
807         }
808         else {
809             t = new StringToken(startLine, startPos, endPos,
810                                 Token.IDENT, mWord.toString());
811         }
812 
813         mWord.setLength(0);
814         return t;
815     }

817     // The two leading slashes have already been scanned when this is
818     // called.
819     private Token scanOneLineComment(int startPos) throws IOException {
820         int c;
821         int startLine = mSource.getLineNumber();
822         int endPos = mSource.getEndPosition();
823         mWord.setLength(0);
824         mWord.append('/').append('/');
825 
826         while ( (c = mSource.peek()) != -1 ) {
827             if (c == '\r' || c == '\n') {
828                 break;
829             }
830             
831             mSource.read();
