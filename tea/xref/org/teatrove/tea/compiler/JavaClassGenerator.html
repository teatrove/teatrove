175     /**
176      * If a template has any substitution blocks that it passes in a
177      * call expression, then the generated code is in a different format.
178      *
179      * Ordinarily, the generated class looks like this:
180      *
181      * public final class a.b.c.Template {
182      *     public static void execute(Context, params ...)
183      *     throws Exception {
184      *         ...
185      *     }
186      *
187      *     private Template() {
188      *         super();
189      *     }
190      * }
191      *
192      * When mSubBlockCount is greater than zero, the generated class looks
193      * like this:
194      *
195      * public final class a.b.c.Template implements Substitution {
196      *     public static void execute(Context, params ...)
197      *     throws Exception {
198      *         new Template(Context, params ...).substitute();
199      *     }
200      *
201      *     private int blockId;
202      *     private Context context;
203      *
204      *     private Template(Context, params ...) {
205      *         super();
206      *         this.context = Context;
207      *         this.params = params;
208      *     }
209      *
210      *     public void substitute() throws Exception {
211      *         if (this.context == null) {
212      *             throw new UnsupportedOperationException();
213      *         }
214      *         substitute(this.context);
215      *     }
216      *
217      *     public void substitute(Context) throws Exception {
218      *         switch (this.blockId) {
219      *         case 0:
220      *             ...
221      *         case 1:
222      *             ...
223      *         case n:
224      *             ...
225      *         }
226      *     }
227      *
228      *     public Object getIdentifier() {
229      *         return new SubstitutionId(this, blockId);
230      *     }
231      *
232      *     public Substitution detach() {
233      *         Substitution sub = (Substitution)clone();
234      *         sub.context = null;
235      *         return sub;
236      *     }
237      * }
238      */

241     private List<String> mCallerToSubNoList = new ArrayList<String>();

255     public void writeTo(OutputStream out) throws IOException {
256         String className = mUnit.getName();
257         String targetPackage = mUnit.getTargetPackage();
258         if (targetPackage != null) {
259             className = targetPackage + '.' + className;
260         }
261 
262         ClassFile classFile = new ClassFile(className);
263         classFile.getModifiers().setFinal(true);
264 
265         String sourceFile = mUnit.getSourceFileName();
266         if (sourceFile != null) {
267             classFile.setSourceFile(sourceFile);
268         }
269 
270         final Template t = getParseTree();
271 
272         t.accept(new TreeWalker() {
273             public Object visit(FunctionCallExpression node) {
274                 if (node.getSubstitutionParam() != null) {
275                     mSubBlockCount++;
276                     mGenerateSubFormat = true;
277                 }
278                 return super.visit(node);
279             }
280 
281             public Object visit(TemplateCallExpression node) {
282                 if (node.getSubstitutionParam() != null) {
283                     mSubBlockCount++;
284                     mGenerateSubFormat = true;
285                 }
286                 return super.visit(node);
287             }
288 
289             public Object visit(Variable node) {
290                 if (node.isField()) {
291                     mGenerateSubFormat = true;
292                 }
293                 return super.visit(node);
294             }
295         });
296 
297         generateTemplate(t, className, classFile);
298 
299         classFile.writeTo(out);
300         out.flush();
301     }

303     protected void generateTemplateParameters(Template t, ClassFile classFile) {
304 
305         // Build the static getTemplateParameterNames method.
306 
307         Variable[] params = t.getParams();
308         int paramCount = params.length;
309 
310         Modifiers pubstat = new Modifiers();
311         pubstat.setPublic(true);
312         pubstat.setStatic(true);
313 
314         TypeDesc stringArrayDesc = TypeDesc.STRING.toArrayType();
315 
316         MethodInfo mi = classFile.addMethod
317             (pubstat, PARAMETER_METHOD_NAME, stringArrayDesc);
318 
319         CodeBuilder builder = new CodeBuilder(mi);
320         builder.loadConstant(paramCount);
321         builder.newObject(stringArrayDesc);
322 
323         // Populate the array.
324         for (int i=0; i<paramCount; i++) {
325             builder.dup();
326             builder.loadConstant(i);
327             builder.loadConstant(params[i].getName());
328             builder.storeToArray(TypeDesc.STRING);
329         }
330 
331         builder.returnValue(TypeDesc.OBJECT);
332 
333         // Done building the static getTemplateParameterNames method.
334     }

336     protected void generateTemplate(Template t, String className,
337                                     ClassFile classFile) {
338         Class<?> subClass;
339         Method subMethod, subContextMethod, subIdMethod, subDetachMethod;
340         if (mGenerateSubFormat) {
341             subClass = Substitution.class;
342             classFile.addInterface(subClass);
343             classFile.addInterface(Cloneable.class);
344             classFile.addInterface(java.io.Serializable.class);
345             try {
346                 subMethod = subClass.getMethod("substitute");
347                 subContextMethod = subClass.getMethod
348                     ("substitute", new Class[]{Context.class});
349                 subIdMethod = subClass.getMethod("getIdentifier");
350                 subDetachMethod = subClass.getMethod("detach");
351             }
352             catch (NoSuchMethodException e) {
353                 throw new InternalError(e.toString());
354             }
355         }
356         else {
357             subClass = null;
358             subMethod = null;
359             subContextMethod = null;
360             subIdMethod = null;
361             subDetachMethod = null;
362         }
363 
364         // Build the static getTemplateParameterNames method.
365 
366         Variable[] params = t.getParams();
367         int paramCount = params.length;
368 
369         generateTemplateParameters(t, classFile);
370 
371         // Done building the static getTemplateParameterNames method.
372 
373         // Build the static execute method.
374 
375         Modifiers pubstat = new Modifiers();
376         pubstat.setPublic(true);
377         pubstat.setStatic(true);
378 
379         Variable[] allParams;
380         if (t.hasSubstitutionParam()) {
381             allParams = new Variable[paramCount + 2];
382             Type type = new Type(Substitution.class);
383             Variable subParam = new Variable(null, SUB_PARAM_NAME, type);
384             allParams[paramCount + 1] = subParam;
385         }
386         else {
387             allParams = new Variable[paramCount + 1];
388         }
389 
390         Type type = new Type(mUnit.getCompiler().getRuntimeContext());
391         allParams[0] = new Variable(null, CONTEXT_PARAM_NAME, type);
392         allParams[0].setTransient(true);
393 
394         for (int i=0; i<paramCount; i++) {
395             allParams[i + 1] = params[i];
396         }
397 
398         TypeDesc[] paramTypes = new TypeDesc[allParams.length];
399 
400         for (int i=0; i<allParams.length; i++) {
401             if (mGenerateSubFormat) {
402                 allParams[i].setField(true);
403             }
404             paramTypes[i] = makeDesc(allParams[i]);
405         }
406 
407         Type returnType = t.getReturnType();
408         TypeDesc returnTypeDesc;
409         if (returnType == null) {
410             returnTypeDesc = null;
411         }
412         else {
413             returnTypeDesc = makeDesc(returnType);
414         }
415 
416         MethodInfo mi = classFile.addMethod(pubstat, EXECUTE_METHOD_NAME,
417                                             returnTypeDesc, paramTypes);
418 
419         mi.addException("java.lang.Exception");
420         CodeBuilder builder = new CodeBuilder(mi);
421 
422         LocalVariable[] localVars = builder.getParameters();
423         // Apply names to the passed in parameters, even though this step
424         // is not required.
425         for (int i=0; i<localVars.length; i++) {
426             localVars[i].setName(allParams[i].getName());
427         }
428 
429         // inject template body profiling bytecode - start marker
430         TypeDesc methodObserverType = makeDesc(MergedClass.InvocationEventObserver.class);
431         boolean profilingEnabled = isProfilingEnabled();
432         if (profilingEnabled) {
433             mGlobalTime = builder.createLocalVariable("startTime", makeDesc(long.class));
434             builder.invokeStatic(mUnit.getRuntimeContext().getName(),
435                 "getInvocationObserver", methodObserverType);
436             builder.invokeInterface(methodObserverType.getFullName(), "currentTime",
437                TypeDesc.forClass(long.class));
438             builder.storeLocal(mGlobalTime);
439         }
440 
441         if (!mGenerateSubFormat) {
442             Visitor gen = new Visitor(allParams, builder.getParameters());
443             gen.allowInitializerStatements();
444             gen.generateNormalFormat(builder, t);
445 
446         }
447         else {
448             boolean doReturnValue = !(Type.VOID_TYPE.equals(returnType));
449 
450             // new <this>(params ...).substitute();
451             TypeDesc thisType = makeDesc(className);
452             builder.newObject(thisType);
453             builder.dup();
454             for (int i=0; i<localVars.length; i++) {
455                 builder.loadLocal(localVars[i]);
456             }
457             builder.invokeConstructor(paramTypes);
458 
459             if (doReturnValue) {
460                 // We'll need the object again to load the field.
461                 builder.dup();
462             }
463 
464             builder.invoke(subMethod);
465 
466             Visitor gen = new Visitor(allParams);
467             gen.allowInitializerStatements();
468 
469             // Generate Substitution.substitute().
470             mi = classFile.addMethod(subMethod);
471             CodeBuilder subBuilder = new CodeBuilder(mi);
472             gen.generateContext(subBuilder);
473             Label gotContext = subBuilder.createLabel();
474             subBuilder.ifNullBranch(gotContext, false);
475             String unSupExName =
476                 UnsupportedOperationException.class.getName();
477             subBuilder.newObject(makeDesc(unSupExName));
478             subBuilder.dup();
479             subBuilder.invokeConstructor(unSupExName);
480             subBuilder.throwObject();
481             gotContext.setLocation();
482             subBuilder.loadThis();
483             gen.generateContext(subBuilder);
484             subBuilder.invoke(subContextMethod);
485             subBuilder.returnVoid();
486 
487             // Generate Substitution.substitute(Context).
488             mi = classFile.addMethod(subContextMethod);
489             subBuilder = new CodeBuilder(mi);
490 
491             String[] subFieldNames = gen.generateSubFormat(subBuilder, t);
492 
493             // inject template body profiling bytecode - end and generate event
494             if (profilingEnabled) {
495                 builder.invokeStatic(mUnit.getRuntimeContext().getName(),
496                     "getInvocationObserver", methodObserverType);
497                 builder.loadConstant(mUnit.getName());
498                 builder.loadConstant(null);
499                 builder.invokeStatic(mUnit.getRuntimeContext().getName(),
500                     "getInvocationObserver", methodObserverType);
501                 builder.invokeInterface(methodObserverType.getFullName(), "currentTime",
502                     makeDesc(long.class));
503                 builder.loadLocal(mGlobalTime);
504                 builder.math(Opcode.LSUB);
505                 builder.invokeInterface(methodObserverType.getFullName(), "invokedEvent", null,
506                     new TypeDesc[] { makeDesc(String.class), makeDesc(String.class),
507                         makeDesc(long.class) });
508             }
509 
510             // Finish execute method.
511             if (doReturnValue) {
512                 // If template returns something, return it here by reading
513                 // from a special field.
514                 builder.loadField(subFieldNames[1], returnTypeDesc);
515                 builder.returnValue(makeDesc(returnType.getNaturalClass()));
516             }
517             else {
518                 builder.returnVoid();
519             }
520 
521             // Generate getIdentifier method.
522             mi = classFile.addMethod(subIdMethod);
523             subBuilder = new CodeBuilder(mi);
524 
525             TypeDesc td = makeDesc(SubstitutionId.class);
526             subBuilder.newObject(td);
527             subBuilder.dup();
528             subBuilder.loadThis();
529             subBuilder.loadThis();
530             subBuilder.loadField(subFieldNames[0], TypeDesc.INT);
531             subBuilder.invokeConstructor
532                 (td.getRootName(),
533                  new TypeDesc[]{TypeDesc.OBJECT, TypeDesc.INT});
534             subBuilder.returnValue(td);
535 
536             // Generate detach method.
537             mi = classFile.addMethod(subDetachMethod);
538             subBuilder = new CodeBuilder(mi);
539             subBuilder.loadThis();
540             subBuilder.invokeVirtual("clone", TypeDesc.OBJECT);
541             subBuilder.checkCast(thisType);
542             LocalVariable sub = subBuilder.createLocalVariable(null, thisType);
543             subBuilder.storeLocal(sub);
544             subBuilder.loadLocal(sub);
545             subBuilder.loadConstant(null);
546             subBuilder.storeField(gen.mContextParam.getVariable().getName(),
547                                   paramTypes[0]);
548             if (t.hasSubstitutionParam()) {
549                 // Also detach internal substitution parameter.
550                 subBuilder.loadLocal(sub);
551                 subBuilder.loadLocal(sub);
552                 String subFieldName = gen.mSubParam.getVariable().getName();
553                 TypeDesc subType = makeDesc(subClass);
554                 subBuilder.loadField(subFieldName, subType);
555                 subBuilder.invoke(subDetachMethod);
556                 subBuilder.storeField(subFieldName, subType);
557             }
558             subBuilder.loadLocal(sub);
559             subBuilder.returnValue(TypeDesc.OBJECT);
560         }
561 
562         // Done building the static execute method.
563 
564         // Build the private constructor.
565 
566         Modifiers pvt = new Modifiers();
567         pvt.setPrivate(true);
568 
569         if (mGenerateSubFormat) {
570             mi = classFile.addConstructor(pvt, paramTypes);
571         }
572         else {
573             mi = classFile.addConstructor(pvt);
574         }
575 
576         builder = new CodeBuilder(mi);
577         builder.loadThis();
578         builder.invokeSuperConstructor();
579 
580         if (mGenerateSubFormat) {
581             // Copy all the params to fields.
582 
583             localVars = builder.getParameters();
584 
585             for (int i=0; i<localVars.length; i++) {
586                 builder.loadThis();
587                 builder.loadLocal(localVars[i]);
588                 builder.storeField(allParams[i].getName(), paramTypes[i]);
589             }
590         }
591 
592         builder.returnVoid();
593 
594         // Build static initializer, if required.
595 
596         if (mInitializerStatements.size() > 0) {
597             mi = classFile.addInitializer();
598             builder = new CodeBuilder(mi);
599 
600             Visitor gen = new Visitor(new Variable[0]);
601 
602             for (int i=0; i<mInitializerStatements.size(); i++) {
603                 Statement stmt = (Statement)mInitializerStatements.get(i);
604                 gen.generateNormalFormat(builder, stmt);
605             }
606 
607             builder.returnVoid();
608         }
609 
610         // Done building static initializer, if required.
611 
612         // Create any necessary private fields, after all methods have been
613         // defined.
614         Iterator<Variable> it = mFields.values().iterator();
615         Modifiers flags = new Modifiers();
616         flags.setPrivate(true);
617         while (it.hasNext()) {
618             Variable v = (Variable)it.next();
619             flags.setStatic(v.isStatic());
620             flags.setTransient(v.isTransient());
621             TypeDesc td = makeDesc(v);
622             classFile.addField(flags, v.getName(), td).markSynthetic();
623         }
624     }

720             mBuilder.loadLocal(param);
721             mBuilder.checkCast(makeDesc(contextVar));
722             mBuilder.storeLocal(param);
723

725             // this template.
726             storeToVariable(contextVar, new Runnable() {
727                 public void run() {
728                     mBuilder.loadLocal(param);
729                 }
730             });
731             contextVar.setField(false);
732 
733             // Create ReturnValue field if template returns a value.
734             Type returnType = node.getReturnType();
735             if (!(Type.VOID_TYPE.equals(returnType))) {
736                 Variable returnValue =
737                     new Variable(null, "returnValue", returnType);
738                 returnValue.setField(true);
739                 declareVariable(returnValue);
740 
741                 mReturnValue = new VariableRef(null, returnValue.getName());
742                 mReturnValue.setVariable(returnValue);
743             }

746             Type blockIdType = Type.INT_TYPE;
747             Variable blockId = new Variable(null, "blockId", blockIdType);
748             blockId.setField(true);
749             declareVariable(blockId);
750

752             mBlockId.setVariable(blockId);

760             Label[] switchLabels = new Label[mSubBlockCount + 1];
761             mCaseNodes = new ArrayList<Node>(mSubBlockCount + 1);
762

764                 cases[i] = i;

766             }

769 
770             mBuilder.switchBranch(cases, switchLabels, defaultLabel);
771

773             int newSize;

775             while ( (newSize = mCaseNodes.size()) > size) {

779                     }

781

784                         TypeDesc.forClass(MergedClass.InvocationEventObserver.class);

789                         TypeDesc.forClass(long.class));

791                         mBuilder.invokeStatic(mContextParam.getVariable().getType().getObjectClass().getName(),
792                             "getInvocationObserver", methodObserverType);
793                         mBuilder.invokeInterface(methodObserverType.getFullName(), "currentTime",
794                            TypeDesc.forClass(long.class));
795                         mBuilder.storeLocal(startTime);
796                     }
797 
798                     generate((Node)mCaseNodes.get(size));
799 
800                     // Inject post-call profiling bytecode.
801                     if (profilingEnabled && size > 0) {
802                         mBuilder.invokeStatic(mContextParam.getVariable().getType().getObjectClass().getName(),
803                             "getInvocationObserver", methodObserverType);
804                         mBuilder.loadConstant(mUnit.getName());
805                         mBuilder.loadConstant((String) mCallerToSubNoList.get(size - 1));
806                         mBuilder.invokeStatic(mContextParam.getVariable().getType().getObjectClass().getName(),
807                             "getInvocationObserver", methodObserverType);
808                         mBuilder.invokeInterface(methodObserverType.getFullName(), "currentTime",
809                             TypeDesc.forClass(long.class));
810                         mBuilder.loadLocal(startTime);
811                         mBuilder.math(Opcode.LSUB);
812                         mBuilder.invokeInterface(methodObserverType.getFullName(), "invokedEvent", null,
813                             new TypeDesc[] { TypeDesc.forClass(String.class), TypeDesc.forClass(String.class),
814                                 TypeDesc.forClass(long.class) });
815                     }
816 
817                 }
818             }

822 
823             generateExceptionHandlers();

826                 mBlockId.getName(),

844         private void generate(Node node) {
845             try {
846                 setLineNumber(node.getSourceInfo());
847                 if (!(node instanceof Expression)) {
848                     node.accept(this);
849                     return;
850                 }
851 
852                 // Expressions require special attention for type conversion.
853 
854                 Expression expr = (Expression)node;
855                 List<Conversion> conversions = expr.getConversionChain();
856 
857                 // Iterate conversion chain backwards at first.
858                 ListIterator<Conversion> it =
859                     conversions.listIterator(conversions.size());
860 
861                 while (it.hasPrevious()) {
862                     typeConvertBegin(it.previous());
863                 }
864 
865                 expr.accept(this);
866 
867                 while (it.hasNext()) {
868                     typeConvertEnd(it.next());
869                 }
870             }
871             catch (DetailException e) {
872                 throw e;
873             }
874             catch (RuntimeException e) {
875                 throw new DetailException(e, "(near line " + mLastLine + ')');
876             }
877         }

879         private void generateExceptionHandlers() {
880             if (mExceptionGuardHandlers == null) {
881                 return;
882             }
883 
884             int size = mExceptionGuardHandlers.size();
885             if (size == 0) {
886                 return;
887             }
888 
889             Label dumpException = mBuilder.createLabel();
890 
891             for (int i=0; i<size; i++) {
892                 GuardHandler gh = (GuardHandler)mExceptionGuardHandlers.get(i);
893                 mBuilder.exceptionHandler(gh.tryStart, gh.tryEnd,
894                                           "java.lang.RuntimeException");
895                 // Subroutine will pop exception object off stack.
896                 mBuilder.jsr(dumpException);
897 
898                 if (gh.replacement != null) {
899                     generate(gh.replacement);
900                 }
901 
902                 mBuilder.branch(gh.tryEnd);
903             }
904 
905             dumpException.setLocation();
906 
907             // Generate code to pass caught exception to
908             // ThreadGroup.uncaughtException.
909 
910             TypeDesc throwableDesc = makeDesc(Throwable.class);
911             TypeDesc threadDesc = makeDesc(Thread.class);
912             TypeDesc threadGroupDesc = makeDesc(ThreadGroup.class);
913 
914             LocalVariable exception =
915                 mBuilder.createLocalVariable("e", throwableDesc);
916             LocalVariable retAddr =
917                 mBuilder.createLocalVariable("addr", TypeDesc.OBJECT);
918             LocalVariable thread =
919                 mBuilder.createLocalVariable("t", threadDesc);
920 
921             // Capture return address.
922             mBuilder.storeLocal(retAddr);
923 
924             // Assume caller has placed exception on stack.
925             mBuilder.storeLocal(exception);
926 
927             mBuilder.invokeStatic("java.lang.Thread", "currentThread",
928                                   threadDesc);
929             mBuilder.storeLocal(thread);
930             mBuilder.loadLocal(thread);
931             mBuilder.invokeVirtual("java.lang.Thread", "getThreadGroup",
932                                    threadGroupDesc);
933             mBuilder.loadLocal(thread);
934             mBuilder.loadLocal(exception);
935             mBuilder.invokeVirtual
936                 ("java.lang.ThreadGroup", "uncaughtException", null,
937                  new TypeDesc[]{threadDesc, throwableDesc});
938 
939             mBuilder.ret(retAddr);
940         }

1019             VariableRef lvalue = node.getLValue();

1021             Variable v = lvalue.getVariable();
1022             LocalVariable local = getLocalVariable(v);
1023 
1024             final Expression rvalue = node.getRValue();
1025             Type rtype = rvalue.getType();
1026 
1027             // Special optimization to convert a = a + n to a += n
1028             if (local != null &&
1029                 ltype.getNaturalClass() == int.class &&
1030                 rtype.getNaturalClass() == int.class &&
1031                 rvalue instanceof ArithmeticExpression) {
1032

1034                 int ID = arith.getOperator().getID();

1036                     Expression left = arith.getLeftExpression();
1037                     Expression right = arith.getRightExpression();

1040                         right.isValueKnown()) {
1041                         if ( ((VariableRef)left).getVariable() == v ) {
1042                             amount = (Integer)right.getValue();
1043                         }
1044                     }
1045                     else if (right instanceof VariableRef &&
1046                              left.isValueKnown() &&
1047                              ID != Token.MINUS) {

1050                         }

1057                         }

1344         public Object visit(NewArrayExpression node) {
1345             ExpressionList list = node.getExpressionList();
1346             Expression[] exprs = list.getExpressions();
1347 
1348             Type initialType = node.getInitialType();
1349             Class<?> initialClass = initialType.getObjectClass();
1350 
1351             // Check if this array is composed entirely of constants,
1352             // and if so, make the array a static field, and populate
1353             // it in a static initializer.
1354             if (mAllowInitializerStatements && node.isAllConstant()) {
1355                 SourceInfo info = node.getSourceInfo();
1356 
1357                 // Create a variable...
1358                 Variable var = new Variable(info, "array", initialType);
1359                 var.setStatic(true);
1360                 generate(var);
1361                 // Create an assignment statement...
1362                 VariableRef ref = new VariableRef(info, "array");
1363                 ref.setVariable(var);
1364 
1365                 // Clone the NewArrayExpression so that the type can be
1366                 // changed back to the underlying array type. This prevents
1367                 // unecessary casting in the assignment statement.
1368                 Expression clonedNode = (Expression)node.clone();
1369                 clonedNode.setType(initialType);
1370 
1371                 AssignmentStatement assn =
1372                     new AssignmentStatement(info, ref, clonedNode);
1373 
1374                 // Move this statement into the static initializer
1375                 mInitializerStatements.add(assn);
1376 
1377                 // Substitute a field access for a NewArrayExpression
1378                 TypeDesc td = makeDesc(initialClass);
1379                 mBuilder.loadStaticField(var.getName(), td);
1380             }
1381             else if (node.isAssociative()) {
1382                 // Constuct LinkedHashMap to support associative array.
1383                 TypeDesc td = makeDesc(LinkedHashMap.class);
1384                 mBuilder.newObject(td);
1385                 mBuilder.dup();
1386 
1387                 int capacity = exprs.length;
1388                 if (capacity == 0) {
1389                     // Should probably use a JDK1.3 EMPTY_MAP, and not wrap
1390                     // with unmodifiableMap.
1391                 }
1392                 else if (capacity == 2) {
1393                     // Should probably use a JDK1.3 singletonMap, and not wrap
1394                     // with unmodifiableMap.
1395                     capacity = 1;
1396                 }
1397                 else {
1398                     // Initial capacity at least 2n + 1.
1399                     BigInteger cap = BigInteger.valueOf(capacity + 1);
1400                     // Ensure capacity is prime to reduce hash collisions.
1401                     while (!cap.isProbablePrime(100)) {
1402                         cap = cap.add(BigInteger.valueOf(2));
1403                     }
1404                     capacity = cap.intValue();
1405                 }
1406 
1407                 mBuilder.loadConstant(capacity);
1408                 mBuilder.invokeConstructor
1409                     (LinkedHashMap.class.getName(), cIntParam);
1410 
1411                 Method putMethod;
1412                 try {
1413                     putMethod = LinkedHashMap.class.getMethod
1414                         ("put", new Class[] {Object.class, Object.class});
1415                 }
1416                 catch (NoSuchMethodException e) {
1417                     throw new RuntimeException(e.toString());
1418                 }
1419 
1420                 boolean doPop = (putMethod.getReturnType() != void.class);
1421 
1422                 // Perform put operations to populate the Map.
1423                 for (int i = 0; i < exprs.length; i += 2) {
1424                     mBuilder.dup(); // duplicate the Map
1425                     generate(exprs[i]); // generate the key
1426                     generate(exprs[i + 1]); // generate the value
1427                     mBuilder.invoke(putMethod);
1428                     if (doPop) {
1429                         mBuilder.pop();
1430                     }
1431                 }
1432 
1433                 // Prevent map modification by rogue Java functions.
1434                 if (node.isAllConstant()) {
1435                     try {
1436                         Method unmodifiable = Collections.class.getMethod
1437                             ("unmodifiableMap", new Class[] {Map.class});
1438                         mBuilder.invoke(unmodifiable);
1439                     }
1440                     catch (NoSuchMethodException e) {
1441                     }
