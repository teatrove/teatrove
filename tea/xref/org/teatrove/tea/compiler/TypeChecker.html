89   * {@link Parser}, filling in type information while it checks the validity of
90   * the whole template. After a template has been type-checked and there are no

92   * {@link ErrorListener} to capture any semantic errors detected by the
93   * TypeChecker.

196     private Class<?> loadClass(String name) throws ClassNotFoundException {
197         while (true) {
198             try {
199                 if (mClassLoader == null) {
200                     return Class.forName(name);
201                 }
202                 else {
203                     return mClassLoader.loadClass(name);
204                 }
205             }
206             catch (ClassNotFoundException e) {
207                 int index = name.lastIndexOf('.');
208                 if (index < 0) {
209                     throw e;
210                 }
211 
212                 // Search for inner class.
213                 name = name.substring(0, index) + '$' +
214                     name.substring(index + 1);
215             }
216         }
217     }

223     public void typeCheck(Visitor visitor) {
224 
225         Template template = mUnit.getParseTree();
226         Name name = template.getName();
227 
228         if (name != null) {
229             if ( !(mUnit.getShortName().equals(name.getName())) ) {
230                 error("template.name", mUnit.getShortName(), name);
231             }
232         }
233 
234         // Convert some ExpressionStatements into ReturnStatements.
235         template.accept(new ReturnConvertor());
236 
237         // Reduce the amount of concatenation operations.
238         // It is important that this step be performed after using the
239         // ReturnConvertor and before the main type checking. Otherwise, a
240         // return may only capture a partial result or an expression may be
241         // set to a type which is only appropriate for concatenation.
242         //
243         // Note: Decreasing the amount of concatenations can have a slight
244         // impact on the language semantics depending on how the context is
245         // defined. Generally, reducing the amount of concatenations improves
246         // performance. If reducing the amount of concatenations hinders
247         // performance, then the data buffering and/or output strategy in the
248         // context may need to be improved.
249         //
250         template.accept(new ConcatenationReducer());
251 
252         // Increase the amount of concatenation operations.
253         // It is important that this step be performed after using the
254         // ReturnConvertor and before the main type checking. Otherwise, a
255         // return may capture too large of a result or an expression may be
256         // set to a type which is only appropriate for output.
257         //
258         // Note: Increasing the amount of concatenations can have a profound
259         // impact on the language semantics depending on how the context is
260         // defined. Generally, increasing the amount of concatenations hinders
261         // performance. If increasing the amount of concatenations improves
262         // performance, then the data buffering and/or output strategy in the
263         // context may need to be improved.
264         //
265         //template.accept(new ConcatenationIncreaser());
266 
267         // Perform major type checking.
268         visitor.check(template);
269 
270         if (mErrorCount == 0 && mExceptionGuardian) {
271             // Equip the parse tree with ExceptionGuard statements.
272             template.accept(new ExceptionGuardian());
273         }
274     }

299         private void defineVariable(VariableRef ref, Type type) {
300             Variable v = ref.getVariable();
301             boolean staticallyTyped = v != null ? v.isStaticallyTyped() : false;
302 
303             Variable newVar = null;
304             if (v == null) {
305                 newVar = new Variable(ref.getSourceInfo(), ref.getName(), type,
306                                       staticallyTyped);
307             }
308             else {
309                 newVar = new Variable(ref.getSourceInfo(), ref.getName(),
310                                       v.getTypeName(), staticallyTyped);
311                 newVar.setType(type);
312             }
313 
314             mScope.declareVariable(newVar);
315             if (!mScope.bindToVariable(ref)) {
316                 // Shouldn't happen.
317                 error("variable.undefined", ref.getName(), ref);
318             }
319         }

331         public Object visit(Template node) {
332             enterScope();
333 
334             if (node.getDirectives() != null) {
335                 for (Directive directive : node.getDirectives()) {
336                     check(directive);
337                 }
338             }
339 
340             Variable[] declared = node.getParams();
341             if (declared != null) {
342                 for (int i=0; i<declared.length; i++) {
343                     check(declared[i]);
344                 }
345             }
346 
347             Statement stmt = node.getStatement();
348             if (stmt != null) {
349                 check(stmt);
350             }
351 
352             Type returnType = mReturnType;
353             if (returnType == null) {
354                 returnType = Type.VOID_TYPE;
355             }
356             node.setReturnType(returnType);
357 
358             exitScope();
359             return null;
360         }

366         protected Class<?> lookupType(String name, int dim, Node node) {
367             StringBuffer fb = new StringBuffer(name);
368             // strip array descriptors from class names, we know its an array from the type node.
369             if (name != null && !name.isEmpty() && name.charAt(0) == '[') {
370                 fb.deleteCharAt(name.lastIndexOf('[') + 1);
371                 int i = 0;
372                 while ((i = fb.indexOf("[")) != -1)
373                     fb.deleteCharAt(i);
374                 if ((i = fb.indexOf(";")) != -1)
375                     fb.deleteCharAt(i);
376             }
377             String checkName = fb.toString();
378             String fixedName = checkName;
379             String errorMsg = null;
380 
381             boolean resolved = false;
382             Class<?> resolvedClass = null;
383             String resolvedPackage = mImports[0];
384             for (int i=-1; i<mImports.length; i++) {
385                 if (i >= 0) {
386                     checkName = mImports[i] + '.' + fixedName;
387                 }
388 
389                 try {
390                     Class<?> clazz = null;
391                     if (checkName.equals("boolean"))
392                         clazz = Boolean.TYPE;
393                     else if (checkName.equals("int"))
394                         clazz = Integer.TYPE;
395                     else if (checkName.equals("double"))
396                         clazz = Double.TYPE;
397                     else if (checkName.equals("long"))
398                         clazz = Long.TYPE;
399                     else if (checkName.equals("short"))
400                         clazz = Short.TYPE;
401                     else
402                         clazz = loadClass(checkName);
403                     checkAccess(clazz, node);
404 
405                     // Found class, check if should be array.
406                     if (dim > 0) {
407                         clazz =
408                             Array.newInstance(clazz, new int[dim]).getClass();
409                     }
410 
411                     errorMsg = null;
412                     if (! resolved) {
413                         if (! mHasImports) {
414                             return clazz;
415                         }
416 
417                         resolved = true;
418                         resolvedClass = clazz;
419                         if (i > 0) {
420                             resolvedPackage = mImports[i];
421                         }
422                         continue;
423                     }
424                     else {
425                         if (i >= 0) {
426                             error("typename.package.conflict", name,
427                                   resolvedPackage, mImports[i], node);
428                         }
429                         return clazz;
430                     }
431                 }
432                 catch (ClassNotFoundException e) {
433                     if (errorMsg == null) {
434                         errorMsg = mFormatter.format("typename.unknown", name);
435                     }
436                 }
437                 catch (RuntimeException e) {
438                     error(e.toString(), node);
439                     return null;
440                 }
441                 catch (LinkageError e) {
442                     error(e.toString(), node);
443                     return null;
444                 }
445             }
446 
447             // Fall through to here only if class wasn't found.
448             if (!resolved && errorMsg != null && node != null) {
449                 error(errorMsg, node);
450             }
451 
452             // return resolved if found
453             return resolvedClass;
454         }

467         protected void checkGenericTypeNames(Class<?> rawType, TypeName node) {
468             // Check that generic types are valid java classes
469             TypeName[] genericTypes = node.getGenericTypes();
470             if (genericTypes != null) {
471                 java.lang.reflect.Type[] arguments =
472                     new java.lang.reflect.Type[genericTypes.length];
473 
474                 for (int i = 0; i < genericTypes.length; i++) {
475                     this.check(genericTypes[i]);
476                     Type result = genericTypes[i].getType();
477                     if (result != null) {
478                         arguments[i] = result.getGenericClass();
479                     } else {
480                         arguments[i] = Object.class;
481                     }
482                 }
483 
484                 node.setType(new Type(rawType, new ParameterizedTypeImpl(rawType, arguments)));
485             }
486             else {
487                 node.setType(new Type(rawType));
488             }
489         }

560         public Object visit(AssignmentStatement node) {
561             VariableRef lvalue = node.getLValue();
562             String lname = lvalue.getName();
563             if(mLoopVariables.contains(lname)) {
564                 error("foreach.loopvar.nomodify", lname, node.getSourceInfo());
565                 return null;
566             }
567 
568             Expression rvalue = node.getRValue();
569             check(rvalue);
570 
571             Type type = rvalue.getType();
572 
573             // Start mod for declarative typing
574             if (type != null) {
575                 Class<?> rclass = type.getObjectClass();
576                 Variable dvar = mScope.getDeclaredVariable(lname);
577                 if (dvar == null)
578                     dvar = lvalue.getVariable();
579                 if (dvar != null && dvar.isStaticallyTyped()) {
580                     check(dvar.getTypeName());
581                     if (dvar.getType() == null) // Added this line
582                         dvar.setType(dvar.getTypeName().getType());
583                     if (dvar.getType() == null)
584                         return null;   // Bad type
585                     Class<?> lclass = dvar.getType().getObjectClass();
586                     if (lvalue.getVariable() == null)
587                         lvalue.setVariable(dvar);
588                     if (!type.isPrimitive())
589                         type = dvar.getType();
590                     if (rclass != null && lclass != null && rclass.isAssignableFrom(lclass) && rclass != lclass) {
591                         type = dvar.getType();
592                         if (rvalue instanceof NullLiteral && type.isPrimitive()) {
593                             error("variable.primitive.uninitialized", type.getNaturalClass().getName(), node);
594                             return null;
595                         }
596                         rvalue.convertTo(type.toNullable());
597                     }
598                     else if (rclass == lclass) {
599                         // nothing to convert (already same type)
600                     }
601                     else if (lclass != null && rclass != null && lclass.isAssignableFrom(rclass)) {
602                         rvalue.convertTo(dvar.getType().toNullable());
603                     }
604                     else if (rclass != lclass) {
605                         error("assignmentstatement.cast.invalid", rclass.getName(), lclass.getName(), node);
606                         return null;
607                     }
608                 }
609             }
610            //End mod
611 
612             if (type != null) {
613                 if (mExceptionGuardian &&
614                     rvalue.isExceptionPossible() &&
615                     type.isNonNull()) {
616 
617                     // Since the expression may throw an exception and be
618                     // guarded, an assignment of null needs to work.
619 
620                     rvalue.convertTo(type.toNullable());
621                     type = rvalue.getType();
622                 }
623                 defineVariable(lvalue, type);
624             }
625 
626             return null;
627         }

647         public Object visit(ForeachStatement node) {
648 
649             ++mForeachCount;
650 
651             VariableRef loopVar = node.getLoopVariable();
652             String varName = loopVar.getName();
653             if(mLoopVariables.contains(varName)) {
654                 error("foreach.loopvar.noreuse", varName,
655                       loopVar.getSourceInfo());
656                 return null;
657             }
658             mLoopVariables.push(varName);
659 
660             Expression range = node.getRange();
661             Expression endRange = node.getEndRange();
662             Block body = node.getBody();
663 
664             check(range);
665             Type rangeType = range.getType();
666 
667             Type endRangeType;
668             if (endRange == null) {
669                 endRangeType = null;
670             }
671             else {
672                 check(endRange);
673                 endRangeType = endRange.getType();
674             }
675 
676             if (rangeType == null ||
677                 (endRange != null && endRangeType == null)) {
678 
679                 // Some error must have been detected in the range, so just
680                 // check everything else possible and bail out.
681 
682                 enterScope();
683                 check(body);
684                 exitScope();
685 
686                 mLoopVariables.pop();
687                 --mForeachCount;
688                 return null;
689             }
690 
691             Class<?> rangeClass = rangeType.getObjectClass();
692             Type elementType;
693 
694             // If there is no end range, then range expression type must be
695             // an array or Collection.
696 
697             if (endRangeType == null) {
698                 if (loopVar.getVariable() != null && loopVar.getVariable().isStaticallyTyped()) {
699                     check(loopVar.getVariable().getTypeName());
700                     try {
701                         if (rangeType.getIterationElementType() == null)
702                             error("foreach.as.collection.required", rangeType.getSimpleName(), range);
703                     }
704                     catch (IntrospectionException e) {
705                         error(e.toString(), node);
706                         return null;
707                     }
708 
709                     elementType = loopVar.getVariable().getTypeName().getType();
710                     if (rangeClass.isArray() && (!rangeClass.getComponentType().isAssignableFrom(
711                             elementType.getNaturalClass())
712                             && !elementType.getNaturalClass().isAssignableFrom(
713                             rangeClass.getComponentType()))) {
714                         try {
715                             error("foreach.as.not.assignable", elementType.getSimpleName(),
716                                 rangeType.getArrayElementType().getSimpleName(), range);

719                             error(e.toString(), node);
720                             return null;
721                         }
722                     }

724                 else {

727                     }

730                         return null;
731                     }
732                 }

736                     error("foreach.iteration.not.supported",
737                           rangeType.getSimpleName(), range);
738                 }
739                 else {

741 
742                     if (node.isReverse() &&
743                         !rangeType.isReverseIterationSupported()) {
744                         error("foreach.reverse.not.supported",
745                               rangeType.getSimpleName(), range);
746                     }
747                 }

753                     error("foreach.range.start",
754                           rangeType.getSimpleName(), range);
755                     range.setType(elementType);
756                 }

764                 if (!Number.class.isAssignableFrom(endRangeClass)) {
765                     error("foreach.range.end",
766                           endRangeType.getSimpleName(), endRange);
767                     endRange.setType(elementType);
768                 }
769                 else {
770                     if (Long.class.isAssignableFrom(endRangeClass)) {
771                         elementType = Type.LONG_TYPE;
772                     }
773                 }
774 
775                 range.convertTo(elementType);
776                 endRange.convertTo(elementType);
777             }
778 
779             // Enter a new scope because variable declarations local
780             // to a foreach loop might never get a value assigned because
781             // there is no guarantee that the body will ever execute.
782             Scope bodyScope = enterScope();
783 
784             if (elementType != null) {
785                 defineVariable(loopVar, elementType);
786             }
787 
788             check(body);
789             exitScope();
790

792 
793             if (vars.length > 0) {
794                 // Since the body can be executed multiple times, variables
795                 // must be of the correct type before entering the scope.
796                 node.setInitializer
797                     (createConversions(mScope, vars, node.getSourceInfo()));
798 
799                 // Apply any conversions at the end of the body as well in
800                 // order for the variables to be of the correct type when
801                 // accessed again at the start of the body.
802                 body.setFinalizer
803                     (createConversions(bodyScope, vars, body.getSourceInfo()));
804 
805                 // Declare all promoted variables outside body scope.
806                 mScope.declareVariables(vars);
807 
808                 // Re-check body to account for variable declaration changes.
809                 if (mErrorCount == 0) {
810                     bodyScope.delete();
811                     bodyScope = enterScope();
812 
813                     if (elementType != null) {
814                         defineVariable(loopVar, elementType);
815                     }
816 
817                     check(body);
818                     exitScope();
819                 }
820             }
821 
822             mLoopVariables.pop();
823             --mForeachCount;
824             return null;
825         }

827         public Object visit(IfStatement node) {
828             Expression condition = node.getCondition();
829             Block thenPart = node.getThenPart();
830             Block elsePart = node.getElsePart();
831             Variable[] thenCasts = null;
832             Variable[] elseCasts = null;
833 
834             int preCondErrorCount = mErrorCount;
835             check(condition);
836 
837             // NOTE: there is no need to check the condition type or convert
838             // to a boolean since the generator will use the truthful checker
839             // to ensure the condition is met
840 
841             if (preCondErrorCount == mErrorCount) {
842                 IsaDetector detector = new IsaDetector();
843                 condition.accept(detector);
844                 thenCasts = detector.getThenCasts();
845                 elseCasts = detector.getElseCasts();
846             }
847 
848             Scope thenScope = null;
849             Scope elseScope = null;
850 
851             thenScope = enterScope();
852             if (thenPart != null) {
853                 Statement stmt = createCasts
854                     (thenScope, thenCasts, thenPart.getSourceInfo());
855                 // Don't typecheck any inserted casts, they are already correct
856                 check(thenPart);
857                 thenPart.setInitializer(stmt);
858             }
859             exitScope();
860 
861             elseScope = enterScope();
862             if (elsePart != null) {
863                 Statement stmt = createCasts
864                     (elseScope, elseCasts, elsePart.getSourceInfo());
865                 // Don't typecheck any inserted casts, they are already correct
866                 check(elsePart);
867                 elsePart.setInitializer(stmt);
868             }
869             exitScope();
870 
871             // Merge then and else scopes
872             Variable[] vars = thenScope.intersect(elseScope);
873 
874             if (vars.length == 0) {
875                 node.setMergedVariables(vars);
876             }
877             else {
878                 if (mExceptionGuardian && condition.isExceptionPossible()) {
879                     // If condition could throw an exception, then the
880                     // exception handler needs to ensure that these variables
881                     // are assigned null. Therefore, the variables need to be
882                     // able to accept null.
883                     for (int i=0; i<vars.length; i++) {
884                         Variable v = vars[i];
885                         Type t = v.getType();
886                         if (t.isNonNull()) {
887                             v = (Variable)v.clone();
888                             v.setType(t.toNullable());
889                             vars[i] = v;
890                         }
891                     }
892                 }
893 
894                 node.setMergedVariables(vars);
895 
896                 // Add corrective code as a result of merging.
897                 if (thenPart == null) {
898                     thenPart = new Block(node.getSourceInfo());
899                     node.setThenPart(thenPart);
900                 }
901                 Statement fin = createConversions
902                     (thenScope, vars, thenPart.getSourceInfo());
903                 thenPart.setFinalizer(fin);
904 
905                 if (elsePart == null) {
906                     elsePart = new Block(node.getSourceInfo());
907                     node.setElsePart(elsePart);
908                 }
909                 fin = createConversions
910                     (elseScope, vars, elsePart.getSourceInfo());
911                 elsePart.setFinalizer(fin);
912 
913                 // Re-declare all new variables with new types for future
914                 // variable references.
915                 mScope.declareVariables(vars);
916             }
917 
918             return null;
919         }

922          * Create variable casting assignment statements.
923          *
924          * @param scope scope of statement
925          * @param newVars optional new variables to assign to
926          * @param info source info to apply to created assignments
927          * @return statement with assignments or null if none
928          */
929         private Statement createCasts(Scope scope,
930                                       Variable[] newVars,
931                                       SourceInfo info) {
932             if (newVars == null) {
933                 return null;
934             }
935 
936             int length = newVars.length;
937             if (length == 0) {
938                 return null;
939             }
940 
941             List<Statement> newStatements = new ArrayList<Statement>(length);
942 
943             for (int i=0; i<length; i++) {
944                 Variable newVar = newVars[i];
945                 String name = newVar.getName();
946                 Variable oldVar = scope.getDeclaredVariable(name);
947 
948                 // It is important that the variable be declared private.
949                 newVar = scope.declareVariable(newVar, true);
950 
951                 if (newVar == oldVar) {
952                     continue;
953                 }
954 
955                 VariableRef lvalue = new VariableRef(info, name);
956                 scope.bindToVariable(lvalue);
957                 lvalue.setVariable(newVar);
958

960                 scope.bindToVariable(rvalue);
961                 rvalue.setVariable(oldVar);

964                 }
965                 catch (IllegalArgumentException e) {
966                     if (mErrorCount == 0) {
967                         throw e;
968                     }
969                 }

971                 newStatements.add
972                     (new AssignmentStatement(info, lvalue, rvalue));
973             }
974 
975             if (newStatements.size() == 0) {
976                 return null;
977             }
978             else {
979                 Statement[] stmts = new Statement[newStatements.size()];
980                 stmts = newStatements.toArray(stmts);
981                 return new StatementList(info, stmts);
982             }

1388         /**
1389          * Returns false if initial checks failed.
1390          */
1391         private boolean initialCallExpressionCheck(CallExpression node) {
1392             // handle expressions
1393             Expression expr = node.getExpression();
1394 
1395             // if expression is a variable handle as either a variable or as a
1396             // reference to a fully-qualified application
1397             if (expr instanceof VariableRef) {
1398                 String name = ((VariableRef) expr).getName();
1399 
1400                 // if variable not defined, attempt to lookup as a
1401                 // fully-qualifed application or type reference
1402                 if (mScope.getDeclaredVariable(name) == null) {
1403                     // check if type reference
1404                     Class<?> type = lookupType(name, 0, null);
1405                     if (type != null) {
1406                         expr = checkForTypeExpression(expr);
1407                         if (expr == null) {
1408                             return false;
1409                         }
1410 
1411                         node.setExpression(expr);
1412                     }
1413 
1414                     // otherwise, assume fully-qualified application target
1415                     else {
1416                         Name target = node.getTarget();
1417                         String tname = name + '.' + target.getName();
1418 
1419                         expr = null;
1420                         node.setExpression(null);
1421                         node.setTarget(new Name(target.getSourceInfo(), tname));
1422                     }
1423                 }
1424 
1425                 // otherwise, assume standard function on the variable
1426                 else {
1427                     // nothing to do...check will ensure proper type.
1428                 }
1429             }
1430 
1431             // handle lookups by determining if static field or normal function
1432             // call on a looked up property
1433             else if (expr instanceof Lookup) {
1434                 expr = checkForTypeExpression(expr);
1435                 if (expr == null) {
1436                     return false;
1437                 }
1438 
1439                 node.setExpression(expr);
1440             }
1441 
1442             // check expression if available
1443             if (expr != null) {
1444                 check(expr);
1445             }
1446 
1447             // check params list
1448             ExpressionList params = node.getParams();
1449             check(params);
1450 
1451             // check initializer
1452             Statement init = node.getInitializer();
1453             if (init != null) {
1454                 check(init);
1455             }
1456 
1457             // handle substitution block
1458             Block subParam = node.getSubstitutionParam();
1459             if (subParam != null) {
1460                 // Enter a new scope because variable declarations local
1461                 // to a substitution block might never get a value assigned
1462                 // because there is no guarantee it will ever execute.
1463                 Scope subScope = enterScope();
1464                 check(subParam);
1465                 exitScope();
1466 
1467                 Variable[] vars = subScope.promote();
1468 
1469                 if (vars.length > 0) {
1470                     // Since the subParam can be executed multiple times,
1471                     // variables must be of the correct type before entering
1472                     // the scope.
1473                     node.setInitializer
1474                         (createConversions(mScope, vars,node.getSourceInfo()));
1475 
1476                     // Apply any conversions at the end of the subParam as well
1477                     // in order for the variables to be of the correct type
1478                     // when accessed again at the start of the body.
1479                     subParam.setFinalizer
1480                         (createConversions

1533         protected Expression checkForTypeExpression(Expression expr) {
1534             // determine if the expression is a variable reference and determine
1535             // if the variable reference is actually a type reference
1536             // NOTE: variables take precedence, so ensure no active variable.
1537             if (expr instanceof VariableRef) {
1538                 String name = ((VariableRef) expr).getName();
1539                 if (mScope.getDeclaredVariable(name) == null) {
1540                     // determine if variable is a class and return as type expr
1541                     Class<?> type = lookupType(name, 0, null);
1542                     if (type != null) {
1543                         SourceInfo info = expr.getSourceInfo();
1544                         TypeName typeName = new TypeName(info, type.getName());
1545                         return new TypeExpression(info, typeName);
1546                     }
1547 
1548                     // otherwise, fail with unknown variable
1549                     error("variable.not.declared", expr);
1550                     return null;
1551                 }
1552             }
1553 
1554             // otherwise, determine if the expression is a series of lookups
1555             // bound to a variable reference that may be a fully-qualified
1556             // reference.
1557             else if (expr instanceof Lookup) {
1558 
1559                 // walk each lookup section to determine fully-qualified path
1560                 Expression parent = expr;
1561                 List<Lookup> lookups = new ArrayList<Lookup>(10);
1562                 while (parent instanceof Lookup) {
1563                     lookups.add(0, (Lookup) parent);
1564                     parent = ((Lookup) parent).getExpression();
1565                 }
1566 
1567                 // first node must be a variable/type reference to be able to
1568                 // convert to a fully-qualified path
1569                 if (parent instanceof VariableRef) {
1570 
1571                     // check if variable is referenced and handle as normal
1572                     // lookup and do not handle as a type expression as
1573                     // variables always have precedence
1574                     String name = ((VariableRef) parent).getName();
1575                     if (mScope.getDeclaredVariable(name) != null) {
1576                         return expr;
1577                     }
1578 
1579                     // otherwise, check if the first token references a class
1580                     // and handle as a type expression
1581                     Class<?> type = lookupType(name, 0, null);
1582                     if (type != null) {
1583                         SourceInfo info = parent.getSourceInfo();
1584                         TypeName typeName = new TypeName(info, type.getName());
1585                         TypeExpression typeExpr =
1586                             new TypeExpression(info, typeName);
1587                         lookups.get(0).setExpression(typeExpr);
1588                         return expr;
1589                     }
1590 
1591                     // otherwise, check if the first token references a package
1592                     // and walk the package tree for the static class lookup
1593                     String fqName = name;
1594                     int size = lookups.size();
1595                     for (int i = 0; i < size; i++) {
1596                         Lookup lookup = lookups.get(i);
1597                         String section = lookup.getLookupName().getName();
1598                         fqName = fqName.concat(".".concat(section));
1599 
1600                         // determine if fully-qualified class name and
1601                         // attempt static field lookup
1602                         type = lookupType(fqName, 0, null);
1603                         if (type != null) {
1604                             SourceInfo info = lookup.getSourceInfo();
1605                             TypeName typeName = new TypeName(info, type.getName());
1606                             TypeExpression typeExpr =
1607                                 new TypeExpression(info, typeName);
1608 
1609                             if (i + 1 < size) {
1610                                 lookups.get(i + 1).setExpression(typeExpr);
1611                                 return expr;
1612                             }
1613                             else { return typeExpr; }
1614                         }
1615                     }
1616 
1617                     // otherwise, fail with unknown variable
1618                     error("variable.not.declared", expr);
1619                     return null;
1620                 }
1621             }
1622 
1623             // unhandled, so handle normally
1624             return expr;
1625         }

2003             node.setType(Type.BOOLEAN_TYPE);

2027                     if (String.class.isAssignableFrom(leftClass)) {

2034                         type = leftType.getCompatibleType(rightType);

2038                     type = leftType.getCompatibleType(rightType);

2061                             rightType = type;
