47  /**
48   * The BasicOptimizer only performs two optimizations: constant
49   * folding and dead code elimination.
50   *
51   * <p>Expressions that contain known values at compile time can
52   * be replaced with a constant expression. This basic optimization is
53   * known as constant folding. It improves runtime performance and reduces
54   * the size of generated code.
55   *
56   * <p>Statements that are known at compile time to be unreachable (sometimes
57   * as a result of constant folding) can be removed. This is dead code
58   * elimination, and it mainly reduces the size of generated code. It can also
59   * improve runtime performance.
60   *
61   * @author Brian S O'Neill
62   */
63  public class BasicOptimizer {
64      private Node mTree;
65  
66      public BasicOptimizer(Node tree) {
67          mTree = tree;
68      }
69  
70      public Node optimize() {
71          return (Node)mTree.accept(new Visitor());
72      }
73  
74      private static class Visitor extends TreeMutator {
75          public Object visit(Statement node) {
76              return null;
77          }
78  
79          public Object visit(StatementList node) {
80              Statement[] stmts = optimizeStatements(node.getStatements());
81              if (stmts == null || stmts.length == 0) {
82                  return null;
83              }
84  
85              if (stmts.length == 1) {
86                  return stmts[0];
87              }
88  
89              return new StatementList(node.getSourceInfo(), stmts);
90          }
91  
92          public Object visit(Block node) {
93              Statement[] stmts = optimizeStatements(node.getStatements());
94  
95              Statement init = node.getInitializer();
96              if (init != null) {
97                  init = (Statement)init.accept(this);
98              }
99  
100             Statement fin = node.getFinalizer();
101             if (fin != null) {
102                 fin = (Statement)fin.accept(this);
103             }
104 
105             if (stmts == null || stmts.length == 0) {
106                 if (init == null && fin == null) {
107                     return null;
108                 }
109                 else {
110                     node.setStatements(new Statement[0]);
111                 }
112             }
113             else {
114                 node.setStatements(stmts);
115             }
116 
117             node.setInitializer(init);
118             node.setFinalizer(fin);
119 
120             return node;
121         }

123         private Statement[] optimizeStatements(Statement[] stmts) {
124             if (stmts == null) {
125                 return null;
126             }
127 
128             int length = stmts.length;
129             Vector<Statement> v = new Vector<Statement>(length);
130 
131             for (int i=0; i<length; i++) {
132                 Statement stmt = (Statement)stmts[i].accept(this);
133                 if (stmt != null) {
134                     v.addElement(stmt);
135                 }
136             }
137 
138             Statement[] newStmts = new Statement[v.size()];
139             v.copyInto(newStmts);
140 
141             return newStmts;
142         }

152         public Object visit(ForeachStatement node) {
153             node = (ForeachStatement)super.visit(node);
154 
155             Expression range = node.getRange();
156             Expression endRange = node.getEndRange();
157             if (endRange != null &&
158                 range.isValueKnown() && endRange.isValueKnown()) {
159 
160                 Object rangeValue = range.getValue();
161                 Object endRangeValue = endRange.getValue();
162 
163                 if (rangeValue instanceof Number &&
164                     endRangeValue instanceof Number) {
165 
166                     if (((Number)rangeValue).intValue() >
167                         ((Number)endRangeValue).intValue()) {
168 
169                         // Will loop zero times. Set the body to null
170                         // instead of returning null because the loop
171                         // variable still should be set to the first value.
172                         // It is possible that the loop variable is
173                         // accessed outside the loop, so it must be
174                         // set to the correct value.
175 
176                         node.setBody(null);
177                     }
178                 }
179             }
180 
181             return node;
182         }

184         public Object visit(IfStatement node) {
185             Expression condition = visitExpression(node.getCondition());
186 
187             Block thenPart;
188             Block elsePart;
189 
190             // If the condition is a constant boolean...
191             if (condition.isValueKnown() &&
192                 condition.getType().getObjectClass() == Boolean.class) {
193                 // ... and the condition's value is true...
194                 if ( ((Boolean)condition.getValue()).booleanValue() ) {
195                     // ... then the then part will always execute.
196                     thenPart = node.getThenPart();
197                     if (thenPart != null) {
198                         thenPart = (Block)thenPart.accept(this);
199                     }
200 
201                     return thenPart;
202                 }
203                 else {
204                     // ... else the else part will always execute.
205                     elsePart = node.getElsePart();
206                     if (elsePart != null) {
207                         elsePart = (Block)elsePart.accept(this);
208                     }
209 
210                     return elsePart;
211                 }
212             }
213 
214             thenPart = node.getThenPart();
215             if (thenPart != null) {
216                 thenPart = (Block)thenPart.accept(this);
217             }
218 
219             elsePart = node.getElsePart();
220             if (elsePart != null) {
221                 elsePart = (Block)elsePart.accept(this);
222             }
223 
224             // if there is no then part, but there is an else part,
225             // invert the condition and make the else part the then part.
226             if (thenPart == null && elsePart != null) {
227                 thenPart = elsePart;
228                 elsePart = null;
229                 condition.convertTo(Type.BOOLEAN_TYPE);
230                 condition = new NotExpression(condition.getSourceInfo(),
231                                               condition);
232                 condition.convertTo(Type.BOOLEAN_TYPE);
233             }
234 
235             node.setCondition(condition);
236             node.setThenPart(thenPart);
237             node.setElsePart(elsePart);
238 
239             return node;
240         }

246         public Object visit(NegateExpression node) {
247             SourceInfo info = node.getSourceInfo();
248             Expression expr = visitExpression(node.getExpression());
249 
250             if (expr.isValueKnown()) {
251                 Object value = expr.getValue();
252 
253                 if (value instanceof Number) {
254                     Number number = (Number)value;
255 
256                     if (value instanceof Integer) {
257                         return new NumberLiteral(info, -number.intValue());
258                     }
259                     else if (value instanceof Long) {
260                         return new NumberLiteral(info, -number.longValue());
261                     }
262                     else if (value instanceof Float) {
263                         return new NumberLiteral(info, -number.floatValue());
264                     }
265                     else if (value instanceof Double) {
266                         return new NumberLiteral(info, -number.doubleValue());
267                     }
268                 }
269             }
270             else if (expr instanceof NegateExpression) {
271                 return ((NegateExpression)expr).getExpression();
272             }
273 
274             node.setExpression(expr);
275             return node;
276         }

278         public Object visit(NotExpression node) {
279             SourceInfo info = node.getSourceInfo();
280             Expression expr = visitExpression(node.getExpression());
281 
282             if (expr.isValueKnown()) {
283                 Object value = expr.getValue();
284 
285                 if (value instanceof Boolean) {
286                     boolean bv = ((Boolean)value).booleanValue();
287                     return new BooleanLiteral(info, !bv);
288                 }
289             }
290             else if (expr instanceof NotExpression) {
291                 return ((NotExpression)expr).getExpression();
292             }
293 
294             node.setExpression(expr);
295             return node;
296         }

298         public Object visit(ConcatenateExpression node) {
299             SourceInfo info = node.getSourceInfo();
300 
301             Expression left = visitExpression(node.getLeftExpression());
302             Expression right = visitExpression(node.getRightExpression());
303 
304             if (left.isValueKnown() && left.getValue() instanceof String) {
305                 String leftValue = (String)left.getValue();
306                 if (right.isValueKnown() &&
307                     right.getValue() instanceof String) {
308 
309                     String rightValue = (String)right.getValue();
310                     return new StringLiteral(info, leftValue + rightValue);
311                 }
312 
313                 if (leftValue.length() == 0) {
314                     return right;
315                 }
316             }
317             else if (right.isValueKnown() &&
318                      right.getValue() instanceof String) {
319 
320                 if (((String)right.getValue()).length() == 0) {
321                     return left;
322                 }
323             }
324 
325             node.setLeftExpression(left);
326             node.setRightExpression(right);
327             return node;
328         }

330         public strictfp Object visit(ArithmeticExpression node) {
331             SourceInfo info = node.getSourceInfo();
332             Token operator = node.getOperator();
333 
334             Expression left = visitExpression(node.getLeftExpression());
335             Expression right = visitExpression(node.getRightExpression());
336 
337             if (node.getType() != null &&
338                 left.isValueKnown() && right.isValueKnown()) {
339                 int ID = operator.getID();
340                 Object leftValue = left.getValue();
341                 Object rightValue = right.getValue();
342 
343                 Type type = left.getType();
344 
345                 if (leftValue instanceof Number &&
346                     rightValue instanceof Number &&
347                     type.equals(right.getType())) {
348 
349                     Class<?> clazz = type.getObjectClass();
350 
351                     Number lv = (Number)leftValue;
352                     Number rv = (Number)rightValue;
353 
354                     try {
355                         if (clazz == Integer.class) {
356                             int i1 = lv.intValue();
357                             int i2 = rv.intValue();
358 
359                             switch (ID) {
360                             case Token.PLUS:
361                                 return new NumberLiteral(info, i1 + i2);
362                             case Token.MINUS:
363                                 return new NumberLiteral(info, i1 - i2);
364                             case Token.MULT:
365                                 return new NumberLiteral(info, i1 * i2);
366                             case Token.DIV:
367                                 return new NumberLiteral(info, i1 / i2);
368                             case Token.MOD:
369                                 return new NumberLiteral(info, i1 % i2);
370                             }
371                         }
372                         else if (clazz == Float.class) {
373                             float f1 = lv.floatValue();
374                             float f2 = rv.floatValue();
375 
376                             switch (ID) {
377                             case Token.PLUS:
378                                 return new NumberLiteral(info, f1 + f2);
379                             case Token.MINUS:
380                                 return new NumberLiteral(info, f1 - f2);
381                             case Token.MULT:
382                                 return new NumberLiteral(info, f1 * f2);
383                             case Token.DIV:
384                                 return new NumberLiteral(info, f1 / f2);
385                             case Token.MOD:
386                                 return new NumberLiteral(info, f1 % f2);
387                             }
388                         }
389                         else if (clazz == Long.class) {
390                             long L1 = lv.longValue();
391                             long L2 = rv.longValue();
392 
393                             switch (ID) {
394                             case Token.PLUS:
395                                 return new NumberLiteral(info, L1 + L2);
396                             case Token.MINUS:
397                                 return new NumberLiteral(info, L1 - L2);
398                             case Token.MULT:
399                                 return new NumberLiteral(info, L1 * L2);
400                             case Token.DIV:
401                                 return new NumberLiteral(info, L1 / L2);
402                             case Token.MOD:
403                                 return new NumberLiteral(info, L1 % L2);
404                             }
405                         }
406                         else if (clazz == Double.class) {
407                             double d1 = lv.doubleValue();
408                             double d2 = rv.doubleValue();
409 
410                             switch (ID) {
411                             case Token.PLUS:
412                                 return new NumberLiteral(info, d1 + d2);
413                             case Token.MINUS:
414                                 return new NumberLiteral(info, d1 - d2);
415                             case Token.MULT:
416                                 return new NumberLiteral(info, d1 * d2);
417                             case Token.DIV:
418                                 return new NumberLiteral(info, d1 / d2);
419                             case Token.MOD:
420                                 return new NumberLiteral(info, d1 % d2);
421                             }
422                         }
423                     }
424                     catch (ArithmeticException e) {
425                         // A divide or mod by 0 is ignored, and expression
426                         // is left as is. Runtime will detect and throw
427                         // divide by zero. The compiler is not required
428                         // to detect divide by zero, because it can't
429                         // always perform this check.
430                     }
431                 }
432             }
433 
434             node.setLeftExpression(left);
435             node.setRightExpression(right);
436             return node;
437         }

439         public Object visit(RelationalExpression node) {
440             SourceInfo info = node.getSourceInfo();
441             Token operator = node.getOperator();
442             int ID = operator.getID();
443 
444             Expression left = visitExpression(node.getLeftExpression());
445             Type leftType = left.getType();
446             Object leftValue = left.getValue();
447 
448             if (ID == Token.ISA) {
449                 Type rightType = node.getIsaTypeName().getType();
450 
451                 if (rightType.getObjectClass().isAssignableFrom
452                     (leftType.getObjectClass())) {
453                     // Widening case. i.e. (5 isa Number) is always true.
454                     return new BooleanLiteral(info, true);
455                 }
456 
457                 node.setLeftExpression(left);
458                 return node;
459             }
460 
461             Expression right = visitExpression(node.getRightExpression());
462             Type rightType = right.getType();
463             Object rightValue = right.getValue();
464 
465             if (node.getType() != null &&
466                 left.isValueKnown() && right.isValueKnown() &&
467                 leftValue != null && rightValue != null) {
468 
469                 Type type = leftType;
470 
471                 // TODO: support JDK1.2 Comparable interface
472 
473                 if (leftValue instanceof Number &&
474                     rightValue instanceof Number &&
475                     type.equals(rightType)) {
476 
477                     Class<?> clazz = type.getObjectClass();
478 
479                     Number lv = (Number)leftValue;
480                     Number rv = (Number)rightValue;
481 
482                     if (clazz == Integer.class) {
483                         int i1 = lv.intValue();
484                         int i2 = rv.intValue();
485 
486                         switch (ID) {
487                         case Token.EQ:
488                             return new BooleanLiteral(info, i1 == i2);
489                         case Token.NE:
490                             return new BooleanLiteral(info, i1 != i2);
491                         case Token.LT:
492                             return new BooleanLiteral(info, i1 < i2);
493                         case Token.GT:
494                             return new BooleanLiteral(info, i1 > i2);
495                         case Token.LE:
496                             return new BooleanLiteral(info, i1 <= i2);
497                         case Token.GE:
498                             return new BooleanLiteral(info, i1 >= i2);
499                         }
500                     }
501                     else if (clazz == Float.class) {
502                         float f1 = lv.floatValue();
503                         float f2 = rv.floatValue();
504 
505                         switch (ID) {
506                         case Token.EQ:
507                             return new BooleanLiteral(info, f1 == f2);
508                         case Token.NE:
509                             return new BooleanLiteral(info, f1 != f2);
510                         case Token.LT:
511                             return new BooleanLiteral(info, f1 < f2);
512                         case Token.GT:
513                             return new BooleanLiteral(info, f1 > f2);
514                         case Token.LE:
515                             return new BooleanLiteral(info, f1 <= f2);
516                         case Token.GE:
517                             return new BooleanLiteral(info, f1 >= f2);
518                         }
519                     }
520                     else if (clazz == Long.class) {
521                         long L1 = lv.longValue();
522                         long L2 = rv.longValue();
523 
524                         switch (ID) {
525                         case Token.EQ:
526                             return new BooleanLiteral(info, L1 == L2);
527                         case Token.NE:
528                             return new BooleanLiteral(info, L1 != L2);
529                         case Token.LT:
530                             return new BooleanLiteral(info, L1 < L2);
531                         case Token.GT:
532                             return new BooleanLiteral(info, L1 > L2);
533                         case Token.LE:
534                             return new BooleanLiteral(info, L1 <= L2);
535                         case Token.GE:
536                             return new BooleanLiteral(info, L1 >= L2);
537                         }
538                     }
539                     else if (clazz == Double.class) {
540                         double d1 = lv.doubleValue();
541                         double d2 = rv.doubleValue();
542 
543                         switch (ID) {
544                         case Token.EQ:
545                             return new BooleanLiteral(info, d1 == d2);
546                         case Token.NE:
547                             return new BooleanLiteral(info, d1 != d2);
548                         case Token.LT:
549                             return new BooleanLiteral(info, d1 < d2);
550                         case Token.GT:
551                             return new BooleanLiteral(info, d1 > d2);
552                         case Token.LE:
553                             return new BooleanLiteral(info, d1 <= d2);
554                         case Token.GE:
555                             return new BooleanLiteral(info, d1 >= d2);
556                         }
557                     }
558                 }
559                 else if (leftValue instanceof String &&
560                          rightValue instanceof String) {
561 
562                     int result =
563                         ((String)leftValue).compareTo((String)rightValue);
564 
565                     switch (ID) {
566                     case Token.EQ:
567                         return new BooleanLiteral(info, result == 0);
568                     case Token.NE:
569                         return new BooleanLiteral(info, result != 0);
570                     case Token.LT:
571                         return new BooleanLiteral(info, result < 0);
572                     case Token.GT:
573                         return new BooleanLiteral(info, result > 0);
574                     case Token.LE:
575                         return new BooleanLiteral(info, result <= 0);
576                     case Token.GE:
577                         return new BooleanLiteral(info, result >= 0);
578                     }
579                 }
580             }
581 
582             if (leftType.isNonNull() && rightType.isNonNull() &&
583                 leftType.getObjectClass() == Boolean.class &&
584                 rightType.getObjectClass() == Boolean.class) {
585 
586                 if (left.isValueKnown() && leftValue != null) {
587                     boolean lv = ((Boolean)leftValue).booleanValue();
588 
589                     if (right.isValueKnown() && rightValue != null) {
590                         boolean rv = ((Boolean)rightValue).booleanValue();
591 
592                         if (ID == Token.EQ) {
593                             return new BooleanLiteral(info, lv == rv);
594                         }
595                         else if (ID == Token.NE) {
596                             return new BooleanLiteral(info, lv != rv);
597                         }
598                     }
599                     else {
600                         if (lv) {
601                             // (true == (right)) is always (right)
602                             if (ID == Token.EQ) {
603                                 return right;
604                             }
605                             // (true != (right)) is always !(right)
606                             else if (ID == Token.NE) {
607                                 right.convertTo(Type.BOOLEAN_TYPE);
608                                 right = new NotExpression(info, right);
609                                 right.convertTo(Type.BOOLEAN_TYPE);
610                                 return right;
611                             }
612                         }
613                         else {
614                             // (false == (right)) is always !(right)
615                             if (ID == Token.EQ) {
616                                 right.convertTo(Type.BOOLEAN_TYPE);
617                                 right = new NotExpression(info, right);
618                                 right.convertTo(Type.BOOLEAN_TYPE);
619                                 return right;
620                             }
621                             // (false != (right)) is always (right)
622                             else if (ID == Token.NE) {
623                                 return right;
624                             }
625                         }
626                     }
627                 }
628                 else if (right.isValueKnown() && rightValue != null) {
629                     boolean rv = ((Boolean)rightValue).booleanValue();
630 
631                     if (rv) {
632                         // ((left) == true) is always (left)
633                         if (ID == Token.EQ) {
634                             return left;
635                         }
636                         // ((left) != true) is always !(left)
637                         else if (ID == Token.NE) {
638                             left.convertTo(Type.BOOLEAN_TYPE);
639                             left = new NotExpression(info, left);
640                             left.setType(Type.BOOLEAN_TYPE);
641                             return left;
642                         }
643                     }
644                     else {
645                         // ((left) == false) is always !(left)
646                         if (ID == Token.EQ) {
647                             left.convertTo(Type.BOOLEAN_TYPE);
648                             left = new NotExpression(info, left);
649                             left.setType(Type.BOOLEAN_TYPE);
650                             return left;
651                         }
652                         // ((left) != false) is always (left)
653                         else if (ID == Token.NE) {
654                             return left;
655                         }
656                     }
657                 }
658             }
659 
660             // Optimize tests against null.
661 
662             boolean leftIsNull = left.isValueKnown() && leftValue == null;
663             boolean rightIsNull = right.isValueKnown() && rightValue == null;
664 
665             if (leftIsNull && rightIsNull) {
666                 if (ID == Token.EQ) {
667                     return new BooleanLiteral(info, true);
668                 }
669                 else if (ID == Token.NE) {
670                     return new BooleanLiteral(info, false);
671                 }
672             }
673             /* This optimization may eliminate expressions with side effects.
674             else if (leftIsNull && rightType.isNonNull() ||
675                      rightIsNull && leftType.isNonNull()) {
676             */
677             else if (leftIsNull && rightType.isNonNull() &&
678                      (right instanceof Literal ||
679                       right instanceof VariableRef) ||
680                      rightIsNull && leftType.isNonNull() &&
681                      (left instanceof Literal ||
682                       left instanceof VariableRef)) {
683 
684                 if (ID == Token.EQ) {
685                     return new BooleanLiteral(info, false);
686                 }
687                 else if (ID == Token.NE) {
688                     return new BooleanLiteral(info, true);
689                 }
690             }
691 
692             node.setLeftExpression(left);
693             node.setRightExpression(right);
694             return node;
695         }

697         public Object visit(AndExpression node) {
698             SourceInfo info = node.getSourceInfo();
699 
700             Expression left = visitExpression(node.getLeftExpression());
701             Expression right = visitExpression(node.getRightExpression());
702 
703             if (left.isValueKnown()) {
704                 Object leftValue = left.getValue();
705                 if (leftValue instanceof Boolean) {
706                     if ( ((Boolean)leftValue).booleanValue() ) {
707                         // "And"ing with true has no effect. Simply return the
708                         // right expression.
709                         return right;
710                     }
711                     else {
712                         // Short circuit result if value is false.
713                         return new BooleanLiteral(info, false);
714                     }
715                 }
716             }
717 
718             if (right.isValueKnown()) {
719                 Object rightValue = right.getValue();
720                 if (rightValue instanceof Boolean) {
721                     if ( ((Boolean)rightValue).booleanValue() ) {
722                         // "And"ing with true has no effect. Simply return the
723                         // left expression.
724                         return left;
725                     }
726                     else {
727                         // Don't short cicuit in this case.
728                     }
729                 }
730             }
731 
732             node.setLeftExpression(left);
733             node.setRightExpression(right);
734             return node;
735         }

737         public Object visit(OrExpression node) {
738             SourceInfo info = node.getSourceInfo();
739 
740             Expression left = visitExpression(node.getLeftExpression());
741             Expression right = visitExpression(node.getRightExpression());
742 
743             if (left.isValueKnown()) {
744                 Object leftValue = left.getValue();
745                 if (leftValue instanceof Boolean) {
746                     if ( ((Boolean)leftValue).booleanValue() ) {
747                         // Short circuit result if value is true.

760                 if (rightValue instanceof Boolean) {

763                     }
764                     else {
765                         // "Or"ing with false has no effect. Simply return the
766                         // left expression.
767                         return left;
768                     }
769                 }
770             }

773             node.setRightExpression(right);

780             Expression thenPart;

782

784             if (condition.isValueKnown() &&
785                 condition.getType().getObjectClass() == Boolean.class) {
786                 // ... and the condition's value is true...
787                 if ( ((Boolean)condition.getValue()).booleanValue() ) {

789                     thenPart = node.getThenPart();

792                     }

795                 }
796                 else {
797                     // ... else the else part will always execute.
798                     elsePart = node.getElsePart();
799                     if (elsePart != null) {
800                         elsePart = (Expression)elsePart.accept(this);
801                     }

804                 }
805             }

807             thenPart = node.getThenPart();

809                 thenPart = (Expression)thenPart.accept(this);
810             }
811 
812             elsePart = node.getElsePart();
813             if (elsePart != null) {
814                 elsePart = (Expression)elsePart.accept(this);
815             }
816 
817             // if there is no then part, but there is an else part,
818             // invert the condition and make the else part the then part.
819             if (thenPart == null && elsePart != null) {
820                 thenPart = elsePart;
821                 elsePart = null;
822                 condition.convertTo(Type.BOOLEAN_TYPE);
823                 condition = new NotExpression(condition.getSourceInfo(),
824                                               condition);
825                 condition.convertTo(Type.BOOLEAN_TYPE);
