17  package org.teatrove.trove.net;
18  
19  import java.net.*;
20  import java.util.*;
21  import java.lang.ref.*;
22  import org.teatrove.trove.util.SoftHashMap;
23  
24  /**
25   * Allows network host names to resolve to inet addresses via event
26   * notification.
27   * <p>
28   * Note: This class makes use of the java.util.Timer class, available only in
29   * Java 2, version 1.3. 
30   *
31   * @author Brian S O'Neill
32   */
33  public class InetAddressResolver {
34      // Wait for 10 minutes before trying to resolve again.
35      private static final long RESOLVE_PERIOD = 10 * 60 * 1000;
36      private static Timer cResolveTimer;
37  
38      private static Map cResolvers = new SoftHashMap();
39  
40      private static Timer getResolveTimer() {
41          Timer timer = cResolveTimer;
42          if (timer == null) {
43              timer = new Timer(true);
44              timer.schedule(new TimerTask() {
45                  public void run() {
46                      Thread.currentThread().setName("InetAddressResolver");
47                  }
48              }, 0);
49              cResolveTimer = timer;
50          }
51          return timer;
52      }
53  
54      /**
55       * Resolve the host name into InetAddresses and listen for changes.
56       * The caller must save a reference to the resolver to prevent it from
57       * being reclaimed by the garbage collector.
58       *
59       * @param host host to resolve into InetAddresses
60       * @param listener listens for InetAddresses
61       */    
62      public static InetAddressResolver listenFor(String host,
63                                                  InetAddressListener listener) {
64          synchronized (cResolvers) {
65              InetAddressResolver resolver = 
66                  (InetAddressResolver)cResolvers.get(host);
67              if (resolver == null) {
68                  resolver = new InetAddressResolver(host);
69                  cResolvers.put(host, resolver);
70              }
71              resolver.addListener(listener);
72              return resolver;
73          }
74      }
75  
76      private final String mHost;
77  
78      private List mListeners = new ArrayList();
79      
80      // Is either an instance of UnknownHostException or InetAddress[].
81      private Object mResolutionResults;
82  
83      private InetAddressResolver(String host) {
84          mHost = host;
85  
86          // Initial delay is random so that requests against other hosts are
87          // staggered.
88          long delay = (long)(Math.random() * RESOLVE_PERIOD);
89          getResolveTimer().schedule(new Resolver(this), delay, RESOLVE_PERIOD);
90      }
91  
92      private synchronized void addListener(InetAddressListener listener) {
93          mListeners.add(listener);
94          if (!resolveAddresses()) {
95              // Ensure that new listener is called the first time.
96              notifyListener(listener);
97          }
98      }
99  
100     private synchronized void notifyListener(InetAddressListener listener) {
101         if (mResolutionResults instanceof UnknownHostException) {
102             listener.unknown((UnknownHostException)mResolutionResults);
103         }
104         else {
105             InetAddress[] addresses = (InetAddress[])mResolutionResults;
106             addresses = (InetAddress[])addresses.clone();
107             listener.resolved(addresses);
108         }
109     }
110 
111     // Returns true if anything changed from the last time this was invoked.
112     private synchronized boolean resolveAddresses() {
113         boolean changed;
114         try {
115             InetAddress[] addresses = InetAddress.getAllByName(mHost);
116             if (mResolutionResults instanceof UnknownHostException) {
117                 changed = true;
118             }
119             else {
120                 // Results may be ordered differently, so keep them sorted.
121                 Arrays.sort(addresses, new Comparator() {
122                     public int compare(Object a, Object b) {
123                         return ((InetAddress)a).getHostAddress()
124                             .compareTo(((InetAddress)b).getHostAddress());
125                     }
126                 });
127 
128                 changed = !Arrays.equals
129                     (addresses, (InetAddress[])mResolutionResults);
130             }
131             mResolutionResults = addresses;
132         }
133         catch (UnknownHostException e) {
134             changed = !(mResolutionResults instanceof UnknownHostException);
135             mResolutionResults = e;
136         }
137 
138         if (changed) {
139             int size = mListeners.size();
140             for (int i=0; i<size; i++) {
141                 notifyListener((InetAddressListener)mListeners.get(i));
142             }
143         }
144 
145         return changed;
146     }
147 
148     private class Resolver extends TimerTask {
149         // Weakly references owner so that the timer won't prevent it from
150         // being garbage collected.
151         private final Reference mOwner;
152 
153         public Resolver(InetAddressResolver owner) {
154             mOwner = new WeakReference(owner);
155         }
156 
157         public void run() {
158             InetAddressResolver owner = (InetAddressResolver)mOwner.get();
159             if (owner == null) {
160                 cancel();
161             }
162             Thread t = Thread.currentThread();
163             String originalName = t.getName();
164             t.setName("InetAddressResolver:" + owner.mHost);
165             try {
166                 owner.resolveAddresses();
167             }
168             finally {
169                 t.setName(originalName);
170             }
171         }
172     }
