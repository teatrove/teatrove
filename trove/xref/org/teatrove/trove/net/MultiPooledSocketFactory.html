22  /**
23   * Pooled SocketFactory implementation that connects to multiple hosts that may
24   * resolve to multiple InetAddresses. If running under Java 2, version 1.3,
25   * changes in the address resolution are automatically detected using
26   * {@link InetAddressResolver}.
27   * <p>
28   * Consider wrapping with a {@link LazySocketFactory} for automatic checking
29   * against socket factories that may be dead.
30   * 
31   * @author Brian S O'Neill
32   */
33  public class MultiPooledSocketFactory extends DistributedSocketFactory {
34      final InetAddress mLocalAddr;
35      final int mLocalPort;
36  
37      // Just references InetAddressResolvers to keep them from going away.
38      private Object[] mResolvers;
39  
40      /**
41       * @param hosts hosts to connect to; length matches ports
42       * @param ports ports to connect to; length matches hosts
43       * @param timeout Maximum time to wait (in milliseconds) for new
44       * connections to be established before throwing an exception
45       */
46      public MultiPooledSocketFactory(String[] hosts,
47                                      int[] ports,
48                                      long timeout) {
49          this(hosts, ports, null, 0, timeout, null);

86      /**
87       * @param hosts hosts to connect to; length matches ports
88       * @param ports ports to connect to; length matches hosts
89       * @param localAddr Local address to bind new sockets to
90       * @param localPort Local port to bind new sockets to, 0 for any
91       * @param timeout Maximum time to wait (in milliseconds) for new
92       * connections to be established before throwing an exception
93       * @param rnd Optional random number generator if random distribution is
94       * desired.
95       */
96      public MultiPooledSocketFactory(String[] hosts,
97                                      int[] ports,
98                                      InetAddress localAddr,
99                                      int localPort,
100                                     long timeout,
101                                     Random rnd)
102     {
103         super(timeout, rnd);
104 
105         mLocalAddr = localAddr;
106         mLocalPort = localPort;
107 
108         try {
109             mResolvers = new InetAddressResolver[hosts.length];
110             for (int i=0; i<hosts.length; i++) {
111                 Listener listener = new Listener(ports[i], timeout);
112                 mResolvers[i] =
113                     InetAddressResolver.listenFor(hosts[i], listener);
114             }
115         }
116         catch (NoClassDefFoundError e) {
117             // Timer class probably wasn't found, so InetAddressResolver
118             // cannot be used.
119             mResolvers = null;
120             for (int i=0; i<hosts.length; i++) {
121                 Listener listener = new Listener(ports[i], timeout);
122                 try {
123                     listener.resolved(InetAddress.getAllByName(hosts[i]));
124                 }
125                 catch (UnknownHostException e2) {
126                     listener.unknown(e2);
127                 }
128             }
129         }
130     }

132     /**
133      * Create socket factories for newly resolved addresses. Default
134      * implementation returns a LazySocketFactory wrapping a
135      * PooledSocketFactory wrapping a PlainSocketFactory.
136      */
137     protected SocketFactory createSocketFactory(InetAddress address,
138                                                 int port,
139                                                 InetAddress localAddr,
140                                                 int localPort,
141                                                 long timeout)
142     {
143         SocketFactory factory;
144         factory = new PlainSocketFactory
145             (address, port, localAddr, localPort, timeout);
146         factory = new PooledSocketFactory(factory);
147         factory = new LazySocketFactory(factory);
148         return factory;
149     }
150 
151     private class Listener implements InetAddressListener {
152         private final int mPort;
153         private final long mTimeout;
154 
155         // Maps InetAddress objects to SocketFactories.
156         private Map mAddressedFactories;
157     
158         public Listener(int port, long timeout) {
159             mPort = port;
160             mTimeout = timeout;
161             mAddressedFactories = new HashMap();
162         }
163 
164         public void unknown(UnknownHostException e) {
165             Thread t = Thread.currentThread();
166             t.getThreadGroup().uncaughtException(t, e);
167 
168             // Remove all the addressed factories.
169             Iterator it = mAddressedFactories.keySet().iterator();
170             while (it.hasNext()) {
171                 SocketFactory factory = 
172                     (SocketFactory)mAddressedFactories.get(it.next());
173                 removeSocketFactory(factory);
174             }
175         }
176         
177         public void resolved(InetAddress[] addresses) {
178             // Add newly discovered addresses.
179             for (int i=0; i<addresses.length; i++) {
180                 InetAddress address = addresses[i];
181                 if (!mAddressedFactories.containsKey(address)) {
182                     SocketFactory factory = createSocketFactory
183                         (address, mPort, mLocalAddr, mLocalPort, mTimeout);
184                     mAddressedFactories.put(address, factory);
185                     addSocketFactory(factory);
186                 }
187             }
188             
189             // Remove addresses no longer being routed to.
190             Iterator it = mAddressedFactories.keySet().iterator();
191             mainLoop:
192             while (it.hasNext()) {
193                 InetAddress address = (InetAddress)it.next();
194                 for (int i=0; i<addresses.length; i++) {
195                     if (addresses[i].equals(address)) {
196                         continue mainLoop;
197                     }
198                 }
199                 SocketFactory factory =
200                     (SocketFactory)mAddressedFactories.get(address);
201                 removeSocketFactory(factory);
202             }
203         }
204     }
