17  package org.teatrove.trove.file;
18  
19  import java.io.IOException;
20  import java.io.InterruptedIOException;
21  import java.io.FileNotFoundException;
22  import org.teatrove.trove.util.ReadWriteLock;
23  import org.teatrove.trove.util.SecureReadWriteLock;
24  
25  /**
26   * 
27   * @author Brian S O'Neill
28   */
29  public class MultiplexFileRepository implements FileRepository {
30      private final MultiplexFile mMF;
31  
32      private final FileBuffer mFreeIds;
33      // Set bits indicate that the file exists.
34      private final Bitlist mFreeIdBitlist;
35      private final FileBufferInputStream mFreeIdsIn;
36      private final FileBufferOutputStream mFreeIdsOut;
37      private final ReadWriteLock mLock;
38  
39      /**
40       * @param mf MultiplexFile to store files
41       */
42      public MultiplexFileRepository(MultiplexFile mf) throws IOException {
43          this(mf, 1);
44      }
45  
46      /**
47       * @param mf MultiplexFile to store files
48       * @param firstId First id in MultiplexFile to use
49       */
50      public MultiplexFileRepository(MultiplexFile mf, int firstId)
51          throws IOException
52      {
53          mMF = mf;
54          if (firstId <= 0) {
55              // Never use file zero.
56              firstId = 1;
57          }
58          mFreeIds = mf.openFile(firstId);
59          mFreeIdBitlist = new Bitlist(mf.openFile(firstId + 1));
60          mFreeIdsIn = new FileBufferInputStream(mFreeIds);
61          mFreeIdsOut = new FileBufferOutputStream(mFreeIds);
62          mLock = mFreeIds.lock();
63      }
64  
65      public long fileCount() throws IOException {
66          try {
67              mLock.acquireReadLock();
68              return mFreeIdBitlist.size() - mFreeIds.size() / 8;
69          }
70          catch (InterruptedException e) {
71              throw new InterruptedIOException();
72          }
73          finally {
74              mLock.releaseLock();
75          }
76      }
77  
78      public Iterator fileIds() throws IOException {
79          return new Iter();
80      }
81  
82      public boolean fileExists(long id) throws IOException {
83          try {
84              mLock.acquireReadLock();
85              return mFreeIdBitlist.get(id);
86          }
87          catch (InterruptedException e) {
88              throw new InterruptedIOException();
89          }
90          finally {
91              mLock.releaseLock();
92          }
93      }
94  
95      public FileBuffer openFile(long id)
96          throws IOException, FileNotFoundException
97      {
98          try {
99              mLock.acquireReadLock();
100             if (!mFreeIdBitlist.get(id)) {
101                 throw new FileNotFoundException(String.valueOf(id));
102             }
103             return mMF.openFile(id);
104         }
105         catch (InterruptedException e) {
106             throw new InterruptedIOException();
107         }
108         finally {
109             mLock.releaseLock();
110         }
111     }
112 
113     public long createFile() throws IOException {
114         try {
115             mLock.acquireWriteLock();
116             long id;
117             if (mFreeIds.size() < 8) {
118                 id = mMF.getFileCount();
119                 // This creates the file.
120                 mMF.openFile(id).close();
121             }
122             else {
123                 long pos = mFreeIds.size() - 8;
124                 mFreeIdsIn.position(pos);
125                 id = mFreeIdsIn.readLong();
126                 mFreeIds.truncate(pos);
127             }
128             mFreeIdBitlist.set(id);
129             return id;
130         }
131         catch (InterruptedException e) {
132             throw new InterruptedIOException();
133         }
134         finally {
135             mLock.releaseLock();
136         }
137     }
138 
139     public boolean deleteFile(long id) throws IOException {
140         try {
141             mLock.acquireUpgradableLock();
142             if (mFreeIdBitlist.get(id)) {
143                 mLock.acquireWriteLock();
144                 try {
145                     mMF.deleteFile(id);
146                     mFreeIdBitlist.clear(id);
147                     long pos = mFreeIds.size();
148                     mFreeIdsOut.position(pos);
149                     mFreeIdsOut.writeLong(id);
150                 }
151                 finally {
152                     mLock.releaseLock();
153                 }
154                 return true;
155             }
156         }
157         catch (InterruptedException e) {
158             throw new InterruptedIOException();
159         }
160         finally {
161             mLock.releaseLock();
162         }
163         return false;
164     }
165 
166     public ReadWriteLock lock() {
167         return mLock;
168     }
169 
170     public int getBlockSize() {
171         return mMF.getBlockSize();
172     }
173 
174     public int getBlockIdScale() {
175         return mMF.getBlockIdScale();
176     }
177 
178     public int getLengthScale() {
179         return mMF.getLengthScale();
180     }
181 
182     public long getMaximumFileLength() {
183         return mMF.getMaximumFileLength();
184     }
185 
186     private class Iter implements Iterator {
187         private byte[] mTemp;
188         private long mIndex;
189 
190         Iter() throws IOException {
191             mTemp = new byte[32];
192             mIndex = mFreeIdBitlist.findFirstSet(0, mTemp);
193         }
194 
195         public boolean hasNext() throws IOException {
196             return mIndex >= 0;
197         }
198 
199         public long next() throws IOException {
200             if (mIndex < 0) {
201                 throw new java.util.NoSuchElementException();
202             }
203             long index = mIndex;
204             mIndex = mFreeIdBitlist.findFirstSet(index + 1, mTemp);
205             return index;
206         }
207     }
