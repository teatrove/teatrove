29  /**
30   * Provides a simple and efficient means of reading and writing bean
31   * properties. BeanPropertyAccessor auto-generates code, eliminating the
32   * need to invoke methods via reflection. Bean access methods are bound-to
33   * directly, using a special hash/switch design pattern.
34   *
35   * @author Brian S O'Neill
36   */
37  public abstract class BeanPropertyAccessor {
38      private static final boolean DEBUG;
39  
40      // Maps classes to BeanPropertyAccessors.
41      private static Map cAccessors = new IdentityMap();
42  
43      static {
44          DEBUG = Boolean.getBoolean
45              ("org.teatrove.trove.util.BeanPropertyAccessor.DEBUG");
46      }

48      /**
49       * Returns a new or cached BeanPropertyAccessor for the given class.
50       */
51      public static BeanPropertyAccessor forClass(Class clazz) {
52          synchronized (cAccessors) {
53              BeanPropertyAccessor bpa =
54                  (BeanPropertyAccessor)cAccessors.get(clazz);
55              if (bpa != null) {
56                  return bpa;
57              }
58              bpa = generate(clazz);
59              cAccessors.put(clazz, bpa);
60              return bpa;
61          }
62      }

64      private static BeanPropertyAccessor generate(Class beanType) {
65          ClassInjector injector = new ClassInjector
66              (beanType.getClassLoader(), (File)null, null);
67  
68          int id = beanType.hashCode();
69  
70          String baseName = BeanPropertyAccessor.class.getName() + '$';
71          String className = baseName;
72          try {
73              while (true) {
74                  className = baseName + (id & 0xffffffffL);
75                  try {
76                      injector.loadClass(className);
77                  }
78                  catch (LinkageError e) {
79                  }
80                  id++;
81              }
82          }
83          catch (ClassNotFoundException e) {
84          }
85  
86          ClassFile cf = generateClassFile(className, beanType);
87  
88          if (DEBUG) {
89              try {
90                  String name = cf.getClassName();
91                  name = name.substring(name.lastIndexOf('.') + 1) + ".class";
92                  System.out.println(name);
93                  java.io.FileOutputStream out =
94                      new java.io.FileOutputStream(name);
95                  cf.writeTo(out);
96                  out.close();
97              }
98              catch (Exception e) {
99                  e.printStackTrace();
100             }
101         }
102 
103         try {
104             OutputStream stream = injector.getStream(cf.getClassName());
105             cf.writeTo(stream);
106             stream.close();
107         }
108         catch (IOException e) {
109             throw new InternalError(e.toString());
110         }
111 
112         try {
113             Class clazz = injector.loadClass(cf.getClassName());
114             return (BeanPropertyAccessor)clazz.newInstance();
115         }
116         catch (ClassNotFoundException e) {
117             throw new InternalError(e.toString());
118         }
119         catch (InstantiationException e) {
120             throw new InternalError(e.toString());
121         }
122         catch (IllegalAccessException e) {
123             throw new InternalError(e.toString());
124         }
125     }

127     private static ClassFile generateClassFile(String className,
128                                                Class beanType)
129     {
130         PropertyDescriptor[][] props = getBeanProperties(beanType);
131 
132         ClassFile cf = new ClassFile(className, BeanPropertyAccessor.class);
133         cf.markSynthetic();
134 
135         Modifiers publicAccess = new Modifiers();
136         publicAccess.setPublic(true);
137 
138         MethodInfo ctor = cf.addConstructor(publicAccess);
139         ctor.markSynthetic();
140         CodeBuilder builder = new CodeBuilder(ctor);
141 
142         builder.loadThis();
143         builder.invokeSuperConstructor();
144         builder.returnVoid();
145 
146         generateMethod(cf, beanType, props[0], true);
147         generateMethod(cf, beanType, props[1], false);
148 
149         return cf;
150     }

152     private static void generateMethod(ClassFile cf,
153                                        Class beanType,
154                                        PropertyDescriptor[] properties,
155                                        boolean forRead)
156     {
157         Modifiers publicAccess = new Modifiers();
158         publicAccess.setPublic(true);
159 
160         TypeDesc objectType = TypeDesc.OBJECT;
161         TypeDesc stringType = TypeDesc.STRING;
162         TypeDesc intType = TypeDesc.INT;
163         TypeDesc booleanType = TypeDesc.BOOLEAN;
164         TypeDesc exceptionType =
165             TypeDesc.forClass(NoSuchPropertyException.class);
166 
167         MethodInfo mi;
168         if (forRead) {
169             TypeDesc[] params = {objectType, stringType};
170             mi = cf.addMethod
171                 (publicAccess, "getPropertyValue", objectType, params);
172         }
173         else {
174             TypeDesc[] params = new TypeDesc[] {
175                 objectType, stringType, objectType
176             };
177             mi = cf.addMethod
178                 (publicAccess, "setPropertyValue", null, params);
179         }
180 
181         mi.markSynthetic();
182         CodeBuilder builder = new CodeBuilder(mi);
183 
184         LocalVariable beanVar = builder.getParameters()[0];
185         LocalVariable propertyVar = builder.getParameters()[1];
186         LocalVariable valueVar;
187         if (forRead) {
188             valueVar = null;
189         }
190         else {
191             valueVar = builder.getParameters()[2];
192         }
193 
194         builder.loadLocal(beanVar);
195         builder.checkCast(TypeDesc.forClass(beanType));
196         builder.storeLocal(beanVar);
197 
198         if (properties.length > 0) {
199             int[] cases = new int[hashCapacity(properties.length)];
200             int caseCount = cases.length;
201             for (int i=0; i<caseCount; i++) {
202                 cases[i] = i;
203             }
204 
205             Label[] switchLabels = new Label[caseCount];
206             Label noMatch = builder.createLabel();
207             List[] caseMethods = caseMethods(caseCount, properties);
208 
209             for (int i=0; i<caseCount; i++) {
210                 List matches = caseMethods[i];
211                 if (matches == null || matches.size() == 0) {
212                     switchLabels[i] = noMatch;
213                 }
214                 else {
215                     switchLabels[i] = builder.createLabel();
216                 }
217             }
218 
219             if (properties.length > 1) {
220                 builder.loadLocal(propertyVar);
221                 builder.invokeVirtual(String.class.getName(),
222                                       "hashCode", intType);
223                 builder.loadConstant(0x7fffffff);
224                 builder.math(Opcode.IAND);
225                 builder.loadConstant(caseCount);
226                 builder.math(Opcode.IREM);
227 
228                 builder.switchBranch(cases, switchLabels, noMatch);
229             }
230 
231             // Params to invoke String.equals.
232             TypeDesc[] params = {objectType};
233 
234             for (int i=0; i<caseCount; i++) {
235                 List matches = caseMethods[i];
236                 if (matches == null || matches.size() == 0) {
237                     continue;
238                 }
239 
240                 switchLabels[i].setLocation();
241 
242                 int matchCount = matches.size();
243                 for (int j=0; j<matchCount; j++) {
244                     PropertyDescriptor pd = (PropertyDescriptor)matches.get(j);
245 
246                     // Test against name to find exact match.
247 
248                     builder.loadConstant(pd.getName());
249                     builder.loadLocal(propertyVar);
250                     builder.invokeVirtual(String.class.getName(),
251                                           "equals", booleanType, params);
252 
253                     Label notEqual;
254 
255                     if (j == matchCount - 1) {
256                         notEqual = null;
257                         builder.ifZeroComparisonBranch(noMatch, "==");
258                     }
259                     else {
260                         notEqual = builder.createLabel();
261                         builder.ifZeroComparisonBranch(notEqual, "==");
262                     }
263 
264                     if (forRead) {
265                         loadPropertyAsObject(builder, beanVar,
266                                              pd.getReadMethod());
267                         builder.returnValue(TypeDesc.OBJECT);
268                     }
269                     else {
270                         savePropertyFromObject(builder, beanVar, valueVar,
271                                                pd.getWriteMethod());
272                         builder.returnVoid();
273                     }
274 
275                     if (notEqual != null) {
276                         notEqual.setLocation();
277                     }
278                 }
279             }
280 
281             noMatch.setLocation();
282         }
283 
284         builder.newObject(exceptionType);
285         builder.dup();
286         builder.loadLocal(propertyVar);
287         builder.loadConstant(forRead);
288 
289         // Params to invoke NoSuchPropertyException.<init>.
290         TypeDesc[] params = {stringType, booleanType};
291 
292         builder.invokeConstructor
293             (NoSuchPropertyException.class.getName(), params);
294         builder.throwObject();
295     }

297     private static void loadPropertyAsObject(CodeBuilder builder,
298                                              LocalVariable beanVar,
299                                              Method m) {
300         Class returnType = m.getReturnType();
301         Type genericType = m.getGenericReturnType();
302 
303         if (!returnType.isPrimitive()) {
304             builder.loadLocal(beanVar);
305             builder.invoke(m);
306             return;
307         }
308 
309         if (returnType == boolean.class) {
310             TypeDesc td = TypeDesc.BOOLEAN.toObjectType();
311             Label falseLabel = builder.createLabel();
312             Label endLabel = builder.createLabel();
313             builder.loadLocal(beanVar);
314             builder.invoke(m);
315             builder.ifZeroComparisonBranch(falseLabel, "==");
316             builder.loadStaticField("java.lang.Boolean", "TRUE", td);
317             builder.branch(endLabel);
318             falseLabel.setLocation();
319             builder.loadStaticField("java.lang.Boolean", "FALSE", td);
320             endLabel.setLocation();
321             return;
322         }
323 
324         TypeDesc objectType =
325             TypeDesc.forClass(returnType, genericType).toObjectType();
326         TypeDesc[] params = {objectType.toPrimitiveType()};
327 
328         builder.newObject(objectType);
329         builder.dup();
330         builder.loadLocal(beanVar);
331         builder.invoke(m);
332         builder.invokeConstructor(objectType.getRootName(), params);
333     }

335     private static void savePropertyFromObject(CodeBuilder builder,
336                                                LocalVariable beanVar,
337                                                LocalVariable valueVar,
338                                                Method m) {
339         Class valueType = m.getParameterTypes()[0];
340 
341         if (!valueType.isPrimitive()) {
342             builder.loadLocal(beanVar);
343             builder.loadLocal(valueVar);
344             builder.checkCast(TypeDesc.forClass(valueType));
345             builder.invoke(m);
346             return;
347         }
348 
349         String methodName;
350         String objectName;
351 
352         String name = valueType.getName();
353         methodName = name + "Value";
354 
355         if (valueType == int.class) {
356             objectName = Integer.class.getName();
357         }
358         else if (valueType == char.class) {
359             objectName = Character.class.getName();
360         }
361         else {
362             objectName = "java.lang." + Character.toUpperCase(name.charAt(0)) +
363                 name.substring(1);
364         }
365 
366         builder.loadLocal(beanVar);
367         builder.loadLocal(valueVar);
368         builder.checkCast(TypeDesc.forClass(objectName));
369         builder.invokeVirtual(objectName, methodName,
370                               TypeDesc.forClass(valueType));
371         builder.invoke(m);
372     }

374     /**
375      * Returns a prime number, at least twice as large as needed. This should
376      * minimize hash collisions. Since all the hash keys are known up front,
377      * the capacity could be tweaked until there are no collisions, but this
378      * technique is easier and deterministic.
379      */
380     private static int hashCapacity(int min) {
381         BigInteger capacity = BigInteger.valueOf(min * 2 + 1);
382         while (!capacity.isProbablePrime(100)) {
383             capacity = capacity.add(BigInteger.valueOf(2));
384         }
385         return capacity.intValue();
386     }

388     /**
389      * Returns an array of Lists of PropertyDescriptors. The first index
390      * matches a switch case, the second index provides a list of all the
391      * PropertyDescriptors whose name hash matched on the case.
392      */
393     private static List[] caseMethods(int caseCount,
394                                       PropertyDescriptor[] props) {
395         List[] cases = new List[caseCount];
396 
397         for (int i=0; i<props.length; i++) {
398             PropertyDescriptor prop = props[i];
399             int hashCode = prop.getName().hashCode();
400             int caseValue = (hashCode & 0x7fffffff) % caseCount;
401             List matches = cases[caseValue];
402             if (matches == null) {
403                 matches = cases[caseValue] = new ArrayList();
404             }
405             matches.add(prop);
406         }
407 
408         return cases;
409     }

411     /**
412      * Returns two arrays of PropertyDescriptors. Array 0 has contains read
413      * PropertyDescriptors, array 1 contains the write PropertyDescriptors.
414      */
415     private static PropertyDescriptor[][] getBeanProperties(Class beanType) {
416         List readProperties = new ArrayList();
417         List writeProperties = new ArrayList();
418 
419         try {
420             Map map = CompleteIntrospector.getAllProperties(beanType);
421 
422             Iterator it = map.values().iterator();
423             while (it.hasNext()) {
424                 PropertyDescriptor pd = (PropertyDescriptor)it.next();
425                 if (pd.getReadMethod() != null) {
426                     readProperties.add(pd);
427                 }
428                 if (pd.getWriteMethod() != null) {
429                     writeProperties.add(pd);
430                 }
431             }
432         }
433         catch (IntrospectionException e) {
434             throw new RuntimeException(e.toString());
435         }
436 
437         PropertyDescriptor[][] props = new PropertyDescriptor[2][];
438 
439         props[0] = new PropertyDescriptor[readProperties.size()];
440         readProperties.toArray(props[0]);
441         props[1] = new PropertyDescriptor[writeProperties.size()];
442         writeProperties.toArray(props[1]);
443 
444         return props;
445     }
