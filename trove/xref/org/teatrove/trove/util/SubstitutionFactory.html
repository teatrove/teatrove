53      @SuppressWarnings("unchecked")
54      public static PropertyMap getSubstitutions(PropertyMap properties,
55                                                 ResourceFactory resourceFactory)
56          throws IOException {
57          
58          PropertyMap subs = new PropertyMap();
59          
60          // add environment variables
61          if (properties == null || properties.getBoolean("env", true)) {
62              subs.putAll(System.getenv());
63          }
64          
65          // add system properties
66          if (properties == null || properties.getBoolean("system", true)) {
67              for (Map.Entry<?, ?> entry : System.getProperties().entrySet()) {
68                  subs.put(entry.getKey().toString(), entry.getValue().toString());
69              }
70          }
71  
72          // add file
73          if (properties != null) {
74              String file = properties.getString("file");
75              if (file != null) {
76                  PropertyMap props = 
77                      resourceFactory.getResourceAsProperties(file);
78                  if (props != null) {
79                      subs.putAll(props);
80                  }
81              }
82          }
83          
84          // add factory class
85          if (properties != null && resourceFactory != null) {
86              PropertyMap factoryMap = properties.subMap("factory");
87              if (factoryMap != null && factoryMap.size() > 0) {
88                  try { subs.putAll(loadProperties(factoryMap)); }
89                  catch (Exception exception) {
90                      throw new IOException("unable to load substitution factory", 
91                                            exception);
92                  }
93              }
94          }
95          
96          // add provided properties
97          if (properties != null) {
98              PropertyMap props = properties.subMap("properties");
99              if (props != null) {
100                 subs.putAll(props);
101             }
102         }
103         
104         // return subs
105         return subs;
106     }

112     public static String substitute(String value, PropertyMap vars) {
113         // ignore if empty string
114         if (value == null || value.isEmpty()) { return value; }
115 
116         // iterate while we have variables
117         int idx = 0, current = 0, marker = 0;
118         StringBuilder sb = new StringBuilder(value.length() + 32);
119 
120         while (idx >= 0) {
121             idx = value.indexOf("${", current);
122             if (idx >= 0) {
123                 // append prefix up to marker
124                 sb.append(value.substring(current, idx));
125 
126                 // update marker to end of variable
127                 current = value.indexOf('}', idx) + 1;
128                 marker = value.indexOf(':', idx) + 1;
129                 if (marker >= current) {
130                     marker = -1;
131                 }
132 
133                 // lookup the variable
134                 String type = null, dflt = null;
135                 if (marker > 0) {
136                     dflt = value.substring(marker, current - 1);
137                     type = value.substring(idx + 2, marker - 1);
138                 }
139                 else {
140                     type = value.substring(idx + 2, current - 1);
141                 }
142                 
143                 // attempt to match the value or set the default
144                 String match = null;
145                 if (vars.containsKey(type)) {
146                     match = (String) vars.get(type);
147                 }
148                 else if (dflt != null) {
149                     match = dflt;
150                 }
151                 
152                 // append value if valid, otherwise append match
153                 if (match == null) {
154                     sb.append("${").append(type).append('}');
155                 }
156                 else { sb.append(match); }
157             }
158         }
159 
160         // append remaining content if applicable
161         if (current < value.length()) {
162             sb.append(value.substring(current));
163         }
164 
165         // return replaced string
166         return sb.toString();
167     }

169     protected String _substitute(String value, Map<String, String> vars) {
170 
171         // ignore if empty string
172         if (value == null || value.isEmpty()) { return value; }
173 
174         // create matcher for url
175         Matcher matcher = PARSER_PATTERN.matcher(value);
176 
177         // loop through matches, substituting as necessary
178         int idx = 0;
179         StringBuilder sb = new StringBuilder(value.length() + 32);
180         while (matcher.find()) {
181             // append non-matching portion
182             sb.append(value.substring(idx, matcher.start()));
183 
184             // lookup type via variables
185             String type = matcher.group(1);
186             String match = vars.get(type);
187 
188             // TODO: add support for default value via colon matching per group2
189             
190             // append value if valid, otherwise append match
191             if (match == null) { sb.append(matcher.group(0)); }
192             else { sb.append(match); }
193 
194             // update index
195             idx = matcher.end();
196         }
197 
198         // append remaining unmatched portion
199         if (idx < value.length()) {
200             sb.append(value.substring(idx));
201         }
202 
203         // return parsed result
204         return sb.toString();
205     }

207     private static PropertyMap loadProperties(PropertyMap factoryProps) 
208         throws Exception {
209         
210         String className = null;
211         PropertyMapFactory factory = null;
212         if (factoryProps != null && factoryProps.size() > 0
213             && (className = factoryProps.getString("class")) != null) {
214         
215             // Load and use custom PropertyMapFactory.
216             Class<?> factoryClass = Class.forName(className);
217             java.lang.reflect.Constructor<?> ctor =
218                 factoryClass.getConstructor(new Class[]{Map.class});
219             factory = (PropertyMapFactory)
220                 ctor.newInstance(new Object[]{factoryProps.subMap("init")});
221         }
222         
223         // return properties
224         return (factory == null ? null : factory.createProperties());
225     }
