26  /**
27   * A highly customizable, high-performance Comparator, designed specifically
28   * for advanced sorting of JavaBeans. BeanComparators contain dynamically
29   * auto-generated code and perform as well as hand written Comparators.
30   * <p>
31   * BeanComparator instances are immutable; order customization methods
32   * return new BeanComparators with refined rules. Calls to customizers can
33   * be chained together to read like a formula. The following example produces
34   * a Comparator that orders Threads by name, thread group name, and reverse
35   * priority.
36   *
37   * <pre>
38   * Comparator c = BeanComparator.forClass(Thread.class)
39   *     .orderBy("name")
40   *     .orderBy("threadGroup.name")
41   *     .orderBy("priority")
42   *     .reverse();
43   * </pre>
44   *
45   * The results of sorting Threads using this Comparator and displaying the
46   * results in a table may look like this:
47   *
48   * <p><table border="2">
49   * <tr><th>name</th><th>threadGroup.name</th><th>priority</th></tr>
50   * <tr><td>daemon</td><td>appGroup</td><td>9</td></tr>
51   * <tr><td>main</td><td>main</td><td>5</td></tr>
52   * <tr><td>main</td><td>secureGroup</td><td>5</td></tr>
53   * <tr><td>sweeper</td><td>main</td><td>1</td></tr>
54   * <tr><td>Thread-0</td><td>main</td><td>5</td></tr>
55   * <tr><td>Thread-1</td><td>main</td><td>5</td></tr>
56   * <tr><td>worker</td><td>appGroup</td><td>8</td></tr>
57   * <tr><td>worker</td><td>appGroup</td><td>5</td></tr>
58   * <tr><td>worker</td><td>secureGroup</td><td>8</td></tr>
59   * <tr><td>worker</td><td>secureGroup</td><td>5</td></tr>
60   * </table><p>
61   *
62   * An equivalent Thread ordering Comparator may be specified as:
63   *
64   * <pre>
65   * Comparator c = BeanComparator.forClass(Thread.class)
66   *     .orderBy("name")
67   *     .orderBy("threadGroup")
68   *     .using(BeanComparator.forClass(ThreadGroup.class).orderBy("name"))
69   *     .orderBy("priority")
70   *     .reverse();
71   * </pre>
72   *
73   * The current implementation of BeanComparator has been optimized for fast
74   * construction and execution of BeanComparators. For maximum performance,
75   * however, save and re-use BeanComparators wherever possible.
76   * <p>
77   * Even though BeanComparator makes use of auto-generated code, instances are
78   * fully Serializable, as long as all passed in Comparators are also
79   * Serializable.
80   *
81   * @author Brian S O'Neill
82   * @see java.beans.Introspector
83   * @see CompleteIntrospector
84   * @see java.util.Collections.sort
85   * @see java.util.Arrays.sort
86   */
87  public class BeanComparator implements Comparator, Serializable {

162     /**
163      * Add an order-by property to produce a more refined Comparator. If the
164      * property does not return a {@link Comparable} object when
165      * {@link #compare compare} is called on the returned comparator, the
166      * property is ignored. Call {@link #using using} on the returned
167      * BeanComparator to specify a Comparator to use for this property instead.
168      * <p>
169      * The specified propery name may refer to sub-properties using a dot
170      * notation. For example, if the bean being compared contains a property
171      * named "info" of type "Information", and "Information" contains a
172      * property named "text", then ordering by the info text can be specified
173      * by "info.text". Sub-properties of sub-properties may be refered to as
174      * well, a.b.c.d.e etc.
175      * <p>
176      * If property type is a primitive, ordering is the same as for its
177      * Comparable object peer. Primitive booleans are ordered false low, true
178      * high. Floating point primitves are ordered exactly the same way as
179      * {@link Float#compareTo(Float) Float.compareTo} and
180      * {@link Double#compareTo(Double) Double.compareTo}.
181      * <p>
182      * Any {@link #reverse reverse-order}, {@link #nullHigh null-order} and 
183      * {@link #caseSensitive case-sensitive} settings are not carried over,
184      * and are reset to the defaults for this order-by property.
185      *
186      * @throws IllegalArgumentException when property doesn't exist or cannot
187      * be read.
188      */
189     public BeanComparator orderBy(String propertyName)
190         throws IllegalArgumentException
191     {
192         int dot = propertyName.indexOf('.');
193         String subName;
194         if (dot < 0) {
195             subName = null;
196         }
197         else {
198             subName = propertyName.substring(dot + 1);
199             propertyName = propertyName.substring(0, dot);
200         }
201 
202         PropertyDescriptor desc =
203             (PropertyDescriptor)getProperties().get(propertyName);
204 
205         if (desc == null) {
206             throw new IllegalArgumentException
207                 ("Property '" + propertyName + "' doesn't exist in '" +
208                  mBeanClass.getName() + '\'');
209         }
210 
211         if (desc.getPropertyType() == null) {
212             throw new IllegalArgumentException
213                 ("Property '" + propertyName + "' is only an indexed type");
214         }
215 
216         if (desc.getReadMethod() == null) {
217             throw new IllegalArgumentException
218                 ("Property '" + propertyName + "' cannot be read");
219         }
220 
221         if (propertyName.equals(mOrderByName)) {
222             // Make String unique so that properties can be specified in
223             // consecutive order-by calls without being eliminated by
224             // reduceRules. A secondary order-by may wish to further refine an
225             // ambiguous comparison using a Comparator.
226             propertyName = new String(propertyName);
227         }
228 
229         BeanComparator bc = new BeanComparator(this);
230         bc.mOrderByName = propertyName;
231 
232         if (subName != null) {
233             BeanComparator subOrder = forClass(desc.getPropertyType());
234             subOrder.mCollator = mCollator;
235             bc = bc.using(subOrder.orderBy(subName));
236         }
237 
238         return bc;
239     }

241     /**
242      * Specifiy a Comparator to use on just the last {@link #orderBy order-by}
243      * property. This is good for comparing properties that are not
244      * {@link Comparable} or for applying special ordering rules for a
245      * property. If no order-by properties have been specified, then Comparator
246      * is applied to the compared beans.
247      * <p>
248      * Any String {@link #caseSensitive case-sensitive} or 
249      * {@link #collate collator} settings are overridden by this Comparator.
250      * If property values being compared are primitive, they are converted to
251      * their object peers before being passed to the Comparator.
252      *
253      * @param c Comparator to use on the last order-by property. Passing null
254      * restores the default comparison for the last order-by property.
255      */
256     public BeanComparator using(Comparator c) {
257         BeanComparator bc = new BeanComparator(this);
258         bc.mOrderByName = mOrderByName;
259         bc.mUsingComparator = c;
260         bc.mFlags = mFlags;
261         return bc;
262     }
263 
264     /**
265      * Toggle reverse-order option on just the last {@link #orderBy order-by}
266      * property. By default, order is ascending. If no order-by properties have
267      * been specified, then reverse order is applied to the compared beans.
268      */
269     public BeanComparator reverse() {
270         BeanComparator bc = new BeanComparator(this);
271         bc.mOrderByName = mOrderByName;
272         bc.mUsingComparator = mUsingComparator;
273         bc.mFlags = mFlags ^ 0x01;
274         return bc;
275     }
276 
277     /**
278      * Set the order of comparisons against null as being high (the default)
279      * on just the last {@link #orderBy order-by} property. If no order-by
280      * properties have been specified, then null high order is applied to the
281      * compared beans. Null high order is the default for consistency with the
282      * high ordering of {@link Float#NaN NaN} by
283      * {@link Float#compareTo(Float) Float}.
284      * <p>
285      * Calling 'nullHigh, reverse' is equivalent to calling 'reverse, nullLow'.
286      */
287     public BeanComparator nullHigh() {
288         BeanComparator bc = new BeanComparator(this);
289         bc.mOrderByName = mOrderByName;
290         bc.mUsingComparator = mUsingComparator;
291         bc.mFlags = mFlags ^ ((mFlags & 0x01) << 1);
292         return bc;
293     }
294 
295     /**
296      * Set the order of comparisons against null as being low
297      * on just the last {@link #orderBy order-by} property. If no order-by
298      * properties have been specified, then null low order is applied to the
299      * compared beans.
300      * <p>
301      * Calling 'reverse, nullLow' is equivalent to calling 'nullHigh, reverse'.
302      */
303     public BeanComparator nullLow() {
304         BeanComparator bc = new BeanComparator(this);
305         bc.mOrderByName = mOrderByName;
306         bc.mUsingComparator = mUsingComparator;
307         bc.mFlags = mFlags ^ ((~mFlags & 0x01) << 1);
308         return bc;
309     }
310 
311     /**
312      * Override the collator and compare just the last order-by property using
313      * {@link String#compareTo(String) String.compareTo}, if it is of type
314      * String. If no order-by properties have been specified then this call is
315      * ineffective.
316      * <p>
317      * A {@link #using using} Comparator disables this setting. Passing null to
318      * the using method will re-enable a case-sensitive setting.
319      */
320     public BeanComparator caseSensitive() {
321         if ((mFlags & 0x04) != 0) {
322             // Already case-sensitive.
323             return this;
324         }
325         BeanComparator bc = new BeanComparator(this);
326         bc.mOrderByName = mOrderByName;
327         bc.mUsingComparator = mUsingComparator;
328         bc.mFlags = mFlags | 0x04;
329         return bc;
330     }

332     /**
333      * Set a Comparator for ordering Strings, which is passed on to all
334      * BeanComparators derived from this one. By default, String are compared
335      * using {@link String#CASE_INSENSITIVE_ORDER}. Passing null for a collator
336      * will cause all String comparisons to use
337      * {@link String#compareTo(String) String.compareTo}.
338      * <p>
339      * A {@link #using using} Comparator disables this setting. Passing null
340      * to the using method will re-enable a collator.
341      *
342      * @param c Comparator to use for ordering all Strings. Passing null
343      * causes all Strings to be ordered by
344      * {@link String#compareTo(String) String.compareTo}.
345      */
346     public BeanComparator collate(Comparator c) {
347         BeanComparator bc = new BeanComparator(this);
348         bc.mOrderByName = mOrderByName;
349         bc.mUsingComparator = mUsingComparator;
350         bc.mFlags = mFlags & ~0x04;
351         bc.mCollator = c;
352         return bc;
353     }

355     public int compare(Object obj1, Object obj2) throws ClassCastException {
356         Comparator c = mComparator;
357         if (c == null) {
358             c = mComparator = generateComparator();
359         }
360         return c.compare(obj1, obj2);
361     }

375     /**
376      * Compares BeanComparators for equality based on their imposed ordering.
377      * Returns true only if the given object is a BeanComparater and it can be
378      * determined without a doubt that the ordering is identical. Because
379      * equality testing is dependent on the behavior of the equals methods of
380      * any 'using' Comparators and/or collators, false may be returned even
381      * though ordering is in fact identical.
382      */
383     public boolean equals(Object obj) {
384         if (obj instanceof BeanComparator) {
385             BeanComparator bc = (BeanComparator)obj;
386          
387             return mFlags == bc.mFlags &&
388                 equalTest(mBeanClass, bc.mBeanClass) &&
389                 equalTest(mOrderByName, bc.mOrderByName) &&
390                 equalTest(mUsingComparator, bc.mUsingComparator) &&
391                 equalTest(mCollator, bc.mCollator) &&
392                 equalTest(mParent, bc.mParent);
393         }
394         else {
395             return false;
396         }
397     }

412     private Comparator generateComparator() {
413         Rules rules = new Rules(this);
414 
415         if (!mHasHashCode) {
416             setHashCode(rules);
417         }
418 
419         Class clazz;
420 
421         synchronized (cGeneratedComparatorCache) {
422             Object c = cGeneratedComparatorCache.get(rules);
423             
424             if (c == null) {
425                 clazz = generateComparatorClass(rules);
426                 cGeneratedComparatorCache.put(rules, clazz);
427             }
428             else if (c instanceof Comparator) {
429                 return (Comparator)c;
430             }
431             else {
432                 clazz = (Class)c;
433             }
434 
435             BeanComparator[] ruleParts = rules.getRuleParts();
436             Comparator[] collators = new Comparator[ruleParts.length];
437             Comparator[] usingComparators = new Comparator[ruleParts.length];
438             boolean singleton = true;
439 
440             for (int i=0; i<ruleParts.length; i++) {
441                 BeanComparator rp = ruleParts[i];
442                 Comparator c2 = rp.mCollator;
443                 if ((collators[i] = c2) != null) {
444                     if (c2 != String.CASE_INSENSITIVE_ORDER) {
445                         singleton = false;
446                     }
447                 }
448                 if ((usingComparators[i] = rp.mUsingComparator) != null) {
449                     singleton = false;
450                 }
451             }
452             
453             try {
454                 Constructor ctor = clazz.getDeclaredConstructor
455                     (new Class[] {Comparator[].class, Comparator[].class});
456                 c = (Comparator)ctor.newInstance
457                     (new Object[] {collators, usingComparators});
458             }
459             catch (NoSuchMethodException e) {
460                 throw new InternalError(e.toString());
461             }
462             catch (InstantiationException e) {
463                 throw new InternalError(e.toString());
464             }
465             catch (IllegalAccessException e) {
466                 throw new InternalError(e.toString());
467             }
468             catch (IllegalArgumentException e) {
469                 throw new InternalError(e.toString());
470             }
471             catch (InvocationTargetException e) {
472                 throw new InternalError(e.getTargetException().toString());
473             }
474             
475             if (singleton) {
476                 // Can save and re-use instance since it obeys the requirements
477                 // for a singleton.
478                 cGeneratedComparatorCache.put(rules, c);
479             }
480 
481             return (Comparator)c;
482         }
483     }

485     private Class generateComparatorClass(Rules rules) {
486         ClassInjector injector =
487             ClassInjector.getInstance(mBeanClass.getClassLoader());
488         
489         int id = rules.hashCode();
490             
491         String baseName = this.getClass().getName() + '$';
492         String className = baseName;
493         try {
494             while (true) {
495                 className = baseName + (id & 0xffffffffL);
496                 try {
497                     injector.loadClass(className);
498                 }
499                 catch (LinkageError e) {
500                 }
501                 id++;
502             }
503         }
504         catch (ClassNotFoundException e) {
505         }
506         
507         ClassFile cf = generateClassFile(className, rules);
508             
509         /*
510         try {
511             String name = cf.getClassName();
512             name = name.substring(name.lastIndexOf('.') + 1) + ".class";
513             System.out.println(name);
514             java.io.FileOutputStream out =
515                 new java.io.FileOutputStream(name);
516             cf.writeTo(out);
517             out.close();
518         }
519         catch (Exception e) {
520             e.printStackTrace();
521         }
522         */
523         
524         try {
525             OutputStream stream = injector.getStream(cf.getClassName());
526             cf.writeTo(stream);
527             stream.close();
528         }
529         catch (IOException e) {
530             throw new InternalError(e.toString());
531         }
532         
533         try {
534             return injector.loadClass(cf.getClassName());
535         }
536         catch (ClassNotFoundException e) {
537             throw new InternalError(e.toString());
538         }
539     }

541     private static ClassFile generateClassFile(String className, Rules rules) {
542         ClassFile cf = new ClassFile(className);
543         cf.markSynthetic();
544 
545         cf.addInterface(Comparator.class);
546         cf.addInterface(Serializable.class);
547 
548         Modifiers privateAccess = new Modifiers();
549         privateAccess.setPrivate(true);
550         Modifiers publicAccess = new Modifiers();
551         publicAccess.setPublic(true);
552 
553         // Define fields to hold usage comparator and collator.
554         TypeDesc comparatorType = TypeDesc.forClass(Comparator.class);
555         TypeDesc comparatorArrayType = comparatorType.toArrayType();
556         cf.addField(privateAccess, 
557                     "mCollators", comparatorArrayType).markSynthetic();
558         cf.addField(privateAccess, 
559                     "mUsingComparators", comparatorArrayType).markSynthetic();
560 
561         // Create constructor to initialize fields.
562         TypeDesc[] paramTypes = {
563             comparatorArrayType, comparatorArrayType
564         };
565         MethodInfo ctor = cf.addConstructor(publicAccess, paramTypes);
566         ctor.markSynthetic();
567         CodeBuilder builder = new CodeBuilder(ctor);
568 
569         builder.loadThis();
570         builder.invokeSuperConstructor();
571         builder.loadThis();
572         builder.loadLocal(builder.getParameters()[0]);
573         builder.storeField("mCollators", comparatorArrayType);
574         builder.loadThis();
575         builder.loadLocal(builder.getParameters()[1]);
576         builder.storeField("mUsingComparators", comparatorArrayType);
577         builder.returnVoid();
578 
579         // Create the all-important compare method.
580         Method compareMethod, compareToMethod;
581         try {
582             compareMethod = Comparator.class.getMethod
583                 ("compare", new Class[] {Object.class, Object.class});
584             compareToMethod = Comparable.class.getMethod
585                 ("compareTo", new Class[] {Object.class});
586         }
587         catch (NoSuchMethodException e) {
588             throw new InternalError(e.toString());
589         }
590 
591         MethodInfo mi = cf.addMethod(compareMethod);
592         mi.markSynthetic();
593         builder = new CodeBuilder(mi);
594 
595         Label endLabel = builder.createLabel();
596         LocalVariable obj1 = builder.getParameters()[0];
597         LocalVariable obj2 = builder.getParameters()[1];
598 
599         // The first rule always applies to the beans directly. All others
600         // apply to properties.
601 
602         BeanComparator[] ruleParts = rules.getRuleParts();
603         BeanComparator bc = ruleParts[0];
604 
605         if ((bc.mFlags & 0x01) != 0) {
606             // Reverse beans.
607             LocalVariable temp = obj1;
608             obj1 = obj2;
609             obj2 = temp;
610         }
611 
612         // Handle the case when obj1 and obj2 are the same (or both null)
613         builder.loadLocal(obj1);
614         builder.loadLocal(obj2);
615         builder.ifEqualBranch(endLabel, true);
616 
617         // Do null order checks for beans.
618         boolean nullHigh = (bc.mFlags & 0x02) == 0;
619         Label label = builder.createLabel();
620         builder.loadLocal(obj1);
621         builder.ifNullBranch(label, false);
622         builder.loadConstant(nullHigh ? 1 : -1);
623         builder.returnValue(TypeDesc.INT);
624         label.setLocation();
625         label = builder.createLabel();
626         builder.loadLocal(obj2);
627         builder.ifNullBranch(label, false);
628         builder.loadConstant(nullHigh ? -1 : 1);
629         builder.returnValue(TypeDesc.INT);
630         label.setLocation();
631 
632         // Call 'using' Comparator if one is provided.
633         LocalVariable result =
634             builder.createLocalVariable("result", TypeDesc.INT);
635         if (bc.mUsingComparator != null) {
636             builder.loadThis();
637             builder.loadField("mUsingComparators", comparatorArrayType);
638             builder.loadConstant(0);
639             builder.loadFromArray(TypeDesc.forClass(Comparator.class));
640             builder.loadLocal(obj1);
641             builder.loadLocal(obj2);
642             builder.invoke(compareMethod);
643             builder.storeLocal(result);
644             builder.loadLocal(result);
645             label = builder.createLabel();
646             builder.ifZeroComparisonBranch(label, "==");
647             builder.loadLocal(result);
648             builder.returnValue(TypeDesc.INT);
649             label.setLocation();
650         }
651 
652         // Cast bean parameters to correct types so that properties may be
653         // accessed.
654         TypeDesc type = TypeDesc.forClass(bc.mBeanClass);
655         builder.loadLocal(obj1);
656         builder.checkCast(type);
657         builder.storeLocal(obj1);
658         builder.loadLocal(obj2);
659         builder.checkCast(type);
660         builder.storeLocal(obj2);
661 
662         // Generate code to perform comparisons against each property.
663         for (int i=1; i<ruleParts.length; i++) {
664             bc = ruleParts[i];
665 
666             PropertyDescriptor desc = 
667                 (PropertyDescriptor)bc.getProperties().get(bc.mOrderByName);
668             Class propertyClass = desc.getPropertyType();
669             TypeDesc propertyType = TypeDesc.forClass(propertyClass);
670             
671             // Create local variable to hold property values.
672             LocalVariable p1 = builder.createLocalVariable("p1", propertyType);
673             LocalVariable p2 = builder.createLocalVariable("p2", propertyType);
674 
675             // Access properties and store in local variables.
676             builder.loadLocal(obj1);
677             builder.invoke(desc.getReadMethod());
678             builder.storeLocal(p1);
679             builder.loadLocal(obj2);
680             builder.invoke(desc.getReadMethod());
681             builder.storeLocal(p2);
682 
683             if ((bc.mFlags & 0x01) != 0) {
684                 // Reverse properties.
685                 LocalVariable temp = p1;
686                 p1 = p2;
687                 p2 = temp;
688             }
689 
690             Label nextLabel = builder.createLabel();
691 
692             // Handle the case when p1 and p2 are the same (or both null)
693             if (!propertyClass.isPrimitive()) {
694                 builder.loadLocal(p1);
695                 builder.loadLocal(p2);
696                 builder.ifEqualBranch(nextLabel, true);
697 
698                 // Do null order checks for properties.
699                 nullHigh = (bc.mFlags & 0x02) == 0;
700                 label = builder.createLabel();
701                 builder.loadLocal(p1);
702                 builder.ifNullBranch(label, false);
703                 builder.loadConstant(nullHigh ? 1 : -1);
704                 builder.returnValue(TypeDesc.INT);
705                 label.setLocation();
706                 label = builder.createLabel();
707                 builder.loadLocal(p2);
708                 builder.ifNullBranch(label, false);
709                 builder.loadConstant(nullHigh ? -1 : 1);
710                 builder.returnValue(TypeDesc.INT);
711                 label.setLocation();
712             }
713 
714             // Call 'using' Comparator if one is provided, else assume
715             // Comparable.
716             if (bc.mUsingComparator != null) {
717                 builder.loadThis();
718                 builder.loadField("mUsingComparators", comparatorArrayType);
719                 builder.loadConstant(i);
720                 builder.loadFromArray(TypeDesc.forClass(Comparator.class));
721                 loadAsObject(builder, propertyClass, p1);
722                 loadAsObject(builder, propertyClass, p2);
723                 builder.invoke(compareMethod);
724             }
725             else {
726                 // If case-sensitive is off and a collator is provided and
727                 // property could be a String, apply collator.
728                 if ((bc.mFlags & 0x04) == 0 && bc.mCollator != null &&
729                     propertyClass.isAssignableFrom(String.class)) {
730 
731                     Label resultLabel = builder.createLabel();
732 
733                     if (!String.class.isAssignableFrom(propertyClass)) {
734                         // Check if both property values are strings at
735                         // runtime. If they aren't, cast to Comparable and call
736                         // compareTo.
737 
738                         TypeDesc stringType = TypeDesc.STRING;
739 
740                         builder.loadLocal(p1);
741                         builder.instanceOf(stringType);
742                         Label notString = builder.createLabel();
743                         builder.ifZeroComparisonBranch(notString, "==");
744                         builder.loadLocal(p2);
745                         builder.instanceOf(stringType);
746                         Label isString = builder.createLabel();
747                         builder.ifZeroComparisonBranch(isString, "!=");
748 
749                         notString.setLocation();
750                         generateComparableCompareTo
751                             (builder, propertyClass, compareToMethod,
752                              resultLabel, nextLabel, p1, p2);
753 
754                         isString.setLocation();
755                     }
756 
757                     builder.loadThis();
758                     builder.loadField("mCollators", comparatorArrayType);
759                     builder.loadConstant(i);
760                     builder.loadFromArray(TypeDesc.forClass(Comparator.class));
761                     builder.loadLocal(p1);
762                     builder.loadLocal(p2);
763                     builder.invoke(compareMethod);
764 
765                     resultLabel.setLocation();
766                 }
767                 else if (propertyClass.isPrimitive()) {
768                     generatePrimitiveComparison(builder, propertyClass, p1,p2);
769                 }
770                 else {
771                     // Assume properties are instances of Comparable.
772                     generateComparableCompareTo
773                         (builder, propertyClass, compareToMethod,
774                          null, nextLabel, p1, p2);
775                 }
776             }
777 
778             if (i < (ruleParts.length - 1)) {
779                 builder.storeLocal(result);
780                 builder.loadLocal(result);
781                 builder.ifZeroComparisonBranch(nextLabel, "==");
782                 builder.loadLocal(result);
783             }
784             builder.returnValue(TypeDesc.INT);
785 
786             // The next property comparison will start here.
787             nextLabel.setLocation();
788         }
789 
790         endLabel.setLocation();
791         builder.loadConstant(0);
792         builder.returnValue(TypeDesc.INT);
793 
794         return cf;
795     }

797     private static void loadAsObject(CodeBuilder builder,
798                                      Class type, LocalVariable v)
799     {
800         if (!type.isPrimitive()) {
801             builder.loadLocal(v);
802             return;
803         }
804 
805         if (type == boolean.class) {
806             TypeDesc td = TypeDesc.BOOLEAN;
807             Label falseLabel = builder.createLabel();
808             Label endLabel = builder.createLabel();
809             builder.loadLocal(v);
810             builder.ifZeroComparisonBranch(falseLabel, "==");
811             builder.loadStaticField("java.lang.Boolean", "TRUE", td);
812             builder.branch(endLabel);
813             falseLabel.setLocation();
814             builder.loadStaticField("java.lang.Boolean", "FALSE", td);
815             endLabel.setLocation();
816             return;
817         }

820         TypeDesc[] params = {objectType.toPrimitiveType()};
821 
822         builder.newObject(objectType);
823         builder.dup();
824         builder.loadLocal(v);
825         builder.invokeConstructor(objectType.getRootName(), params);

829                                                     Class type,
830                                                     LocalVariable a,
831                                                     LocalVariable b)
832     {
833         if (type == float.class) {
834             // Comparison is same as for Float.compareTo(Float).
835             Label done = builder.createLabel();
836 
837             builder.loadLocal(a);
838             builder.loadLocal(b);
839             builder.math(Opcode.FCMPG);
840             Label label = builder.createLabel();
841             builder.ifZeroComparisonBranch(label, ">=");
842             builder.loadConstant(-1);
843             builder.branch(done);
844 
845             label.setLocation();
846             builder.loadLocal(a);
847             builder.loadLocal(b);
848             builder.math(Opcode.FCMPL);
849             label = builder.createLabel();
850             builder.ifZeroComparisonBranch(label, "<=");
851             builder.loadConstant(1);
852             builder.branch(done);
853 
854             Method floatToIntBits;
855             try {
856                 floatToIntBits = Float.class.getMethod
857                     ("floatToIntBits", new Class[] {float.class});
858             }
859             catch (NoSuchMethodException e) {
860                 throw new InternalError(e.toString());
861             }
862 
863             label.setLocation();
864             builder.loadLocal(a);
865             builder.invoke(floatToIntBits);
866             builder.convert(TypeDesc.INT, TypeDesc.LONG);
867             builder.loadLocal(b);
868             builder.invoke(floatToIntBits);
869             builder.convert(TypeDesc.INT, TypeDesc.LONG);
870             builder.math(Opcode.LCMP);
871 
872             done.setLocation();
873         }
874         else if (type == double.class) {
875             // Comparison is same as for Double.compareTo(Double).
876             Label done = builder.createLabel();
877 
878             builder.loadLocal(a);
879             builder.loadLocal(b);
880             done = builder.createLabel();
881             builder.math(Opcode.DCMPG);
882             Label label = builder.createLabel();
883             builder.ifZeroComparisonBranch(label, ">=");
884             builder.loadConstant(-1);
885             builder.branch(done);
886 
887             label.setLocation();
888             builder.loadLocal(a);
889             builder.loadLocal(b);
890             builder.math(Opcode.DCMPL);
891             label = builder.createLabel();
892             builder.ifZeroComparisonBranch(label, "<=");
893             builder.loadConstant(1);
894             builder.branch(done);
895 
896             Method doubleToLongBits;
897             try {
898                 doubleToLongBits = Double.class.getMethod
899                     ("doubleToLongBits", new Class[] {double.class});
900             }
901             catch (NoSuchMethodException e) {
902                 throw new InternalError(e.toString());
903             }
904 
905             label.setLocation();
906             builder.loadLocal(a);
907             builder.invoke(doubleToLongBits);
908             builder.loadLocal(b);
909             builder.invoke(doubleToLongBits);
910             builder.math(Opcode.LCMP);
911 
912             done.setLocation();
913         }
914         else if (type == long.class) {
915             builder.loadLocal(a);
916             builder.loadLocal(b);
917             builder.math(Opcode.LCMP);
918         }
919         else if (type == int.class) {
920             builder.loadLocal(a);
921             builder.convert(TypeDesc.INT, TypeDesc.LONG);
922             builder.loadLocal(b);
923             builder.convert(TypeDesc.INT, TypeDesc.LONG);
924             builder.math(Opcode.LCMP);
925         }
926         else {
927             builder.loadLocal(a);
928             builder.loadLocal(b);
929             builder.math(Opcode.ISUB);
930         }

933     private static void generateComparableCompareTo(CodeBuilder builder,
934                                                     Class type,
935                                                     Method compareToMethod,
936                                                     Label goodLabel,
937                                                     Label nextLabel,
938                                                     LocalVariable a,
939                                                     LocalVariable b)
940     {
941         if (Comparable.class.isAssignableFrom(type)) {
942             builder.loadLocal(a);
943             builder.loadLocal(b);
944             builder.invoke(compareToMethod);
945             if (goodLabel != null) {
946                 builder.branch(goodLabel);
947             }
948         }
949         else {
950             // Cast each property to Comparable only if needed.
951             TypeDesc comparableType = TypeDesc.forClass(Comparable.class);
952 
953             boolean locateGoodLabel = false;
954             if (goodLabel == null) {
955                 goodLabel = builder.createLabel();
956                 locateGoodLabel = true;
957             }
958 
959             Label tryStart = builder.createLabel().setLocation();
960             builder.loadLocal(a);
961             builder.checkCast(comparableType);
962             builder.loadLocal(b);
963             builder.checkCast(comparableType);
964             Label tryEnd = builder.createLabel().setLocation();
965             builder.invoke(compareToMethod);
966             builder.branch(goodLabel);
967 
968             builder.exceptionHandler(tryStart, tryEnd,
969                                      ClassCastException.class.getName());
970             // One of the properties is not Comparable, so just go to next.
971             // Discard the exception.
972             builder.pop();
973             if (nextLabel == null) {
974                 builder.loadConstant(0);
975             }
976             else {
977                 builder.branch(nextLabel);
978             }
979 
980             if (locateGoodLabel) {
981                 goodLabel.setLocation();
982             }
983         }

986     // A key that uniquely describes the rules of a BeanComparator.
987     private static class Rules {
988         private BeanComparator[] mRuleParts;
989         private int mHashCode;
990 
991         public Rules(BeanComparator bc) {
992             mRuleParts = reduceRules(bc);
993 
994             // Compute hashCode.
995             int hash = 0;
996 
997             for (int i = mRuleParts.length - 1; i >= 0; i--) {
998                 bc = mRuleParts[i];
999                 hash = 31 * hash;
1000 
1001                 hash += bc.mFlags << 4;
1002 
1003                 Object obj = bc.mBeanClass;
1004                 if (obj != null) {
1005                     hash += obj.hashCode();
1006                 }
1007                 obj = bc.mOrderByName;
1008                 if (obj != null) {
1009                     hash += obj.hashCode();
1010                 }
1011                 obj = bc.mUsingComparator;
1012                 if (obj != null) {
1013                     hash += obj.getClass().hashCode();
1014                 }
1015                 obj = bc.mCollator;
1016                 if (obj != null) {
1017                     hash += obj.getClass().hashCode();
1018                 }
1019             }
1020 
1021             mHashCode = hash;
1022         }
1023 
1024         public BeanComparator[] getRuleParts() {
1025             return mRuleParts;
1026         }
1027 
1028         public int hashCode() {
1029             return mHashCode;
1030         }

1032         /**
1033          * Equality test determines if rules produce an identical
1034          * auto-generated Comparator.
1035          */
1036         public boolean equals(Object obj) {
1037             if (!(obj instanceof Rules)) {
1038                 return false;
1039             }
1040 
1041             BeanComparator[] ruleParts1 = getRuleParts();
1042             BeanComparator[] ruleParts2 = ((Rules)obj).getRuleParts();
1043 
1044             if (ruleParts1.length != ruleParts2.length) {
1045                 return false;
1046             }
1047             
1048             for (int i=0; i<ruleParts1.length; i++) {
1049                 BeanComparator bc1 = ruleParts1[i];
1050                 BeanComparator bc2 = ruleParts2[i];
1051 
1052                 if (bc1.mFlags != bc2.mFlags) {
1053                     return false;
1054                 }
1055                 if (!equalTest(bc1.mBeanClass, bc2.mBeanClass)) {
1056                     return false;
1057                 }
1058                 if (!equalTest(bc1.mOrderByName, bc2.mOrderByName)) {
1059                     return false;
1060                 }
1061                 if ((bc1.mUsingComparator == null) !=
1062                     (bc2.mUsingComparator == null)) {
1063                     return false;
1064                 }
1065                 if ((bc1.mCollator == null) != (bc2.mCollator == null)) {
1066                     return false;
1067                 }
1068             }
1069 
1070             return true;
1071         }

1073         private BeanComparator[] reduceRules(BeanComparator bc) {
1074             // Reduce the ordering rules by returning BeanComparators
1075             // that are at the end of the chain or before an order-by rule.
1076             List rules = new ArrayList();
1077             
1078             rules.add(bc);
1079             String name = bc.mOrderByName;
1080 
1081             while ((bc = bc.mParent) != null) {
1082                 // Don't perform string comparison using equals method.
1083                 if (name != bc.mOrderByName) {
1084                     rules.add(bc);
1085                     name = bc.mOrderByName;
1086                 }
1087             }
1088             
1089             int size = rules.size();
1090             BeanComparator[] bcs = new BeanComparator[size];
1091             // Reverse rules so that they are in forward order.
1092             for (int i=0; i<size; i++) {
1093                 bcs[size - i - 1] = (BeanComparator)rules.get(i);
1094             }
1095             
1096             return bcs;
1097         }
