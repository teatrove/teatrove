19  /**
20   * SecureReadWriteLock prevents threads from accidentally releasing locks on
21   * behalf of other threads. A thread can only release as many locks as it has
22   * acquired. In addition, IllegalStateExceptions are thrown when a thread
23   * attempts to acquire a lock in a fashion that is deadlock prone.
24   * <p>
25   * Threads requesting write locks are given priority over those requesting
26   * weaker locks. Without this priority, many read threads can easily starve the
27   * threads requesting write locks.
28   * <p>
29   * SecureReadWriteLock is fully re-entrant. A thread may request a lock of
30   * equal or lesser strength as many times as it likes, provided that a release
31   * is performed as many times.
32   *
33   * <pre>
34   * lock.acquireWriteLock();
35   * lock.acquireReadLock();  // works because thread already has stronger lock
36   * ...
37   * lock.releaseLock(); // read lock released, original write lock still kept
38   * lock.releaseLock(); // release original write lock
39   * </pre>
40   *
41   * @author Brian S O'Neill
42   */
43  public final class SecureReadWriteLock implements ReadWriteLock {
44      private final long mTimeout;
45  
46      // The count of threads holding read locks.
47      private int mReadLocks;
48

50      private Thread mUpgradableLockHeld;
51  
52      // The thread (if any) that holds the write lock.
53      private Thread mWriteLockHeld;
54  
55      // The count of threads attempting to acquire write locks.
56      private int mWriteLockAttempts;

82      /**
83       * A read lock is obtained when no threads currently hold a write lock.
84       * When a thread has a read lock, it only blocks threads that wish to
85       * acquire a write lock.
86       * <p>
87       * A read lock is the weakest form of lock.
88       *
89       * @param timeout milliseconds to wait for lock acquisition. If negative,
90       * timeout is infinite.
91       * @return true if the lock was acquired.
92       */
93      public boolean acquireReadLock(long timeout) throws InterruptedException {
94          LockInfo info = (LockInfo)mLockInfoRef.get();
95  
96          if (info.mType == NONE) {
97              synchronized (this) {
98                  if (!readLockAvailable()) {
99                      if (timeout < 0) {
100                         while (true) {
101                             wait();
102                             if (readLockAvailable()) {
103                                 break;
104                             }
105                         }
106                     }
107                     else if (timeout > 0) {
108                         long expire = System.currentTimeMillis() + timeout;
109                         while (true) {
110                             wait(timeout);
111                             if (readLockAvailable()) {
112                                 break;
113                             }
114                             timeout = expire - System.currentTimeMillis();
115                             if (timeout <= 0) {
116                                 return false;
117                             }
118                         };
119                     }
120                     else {
121                         return false;
122                     }
123                 }
124                 mReadLocks++;
125             }
126             info.mType = READ;
127         }     
128         
129         info.mCount++;
130         return true;
131     }

142     /**
143      * An upgradable lock is obtained when no threads currently hold write or
144      * upgradable locks. When a thread has an upgradable lock, it blocks 
145      * threads that wish to acquire upgradable or write locks.
146      * <p>
147      * Upgradable locks are to be used when a thread needs a read lock, but
148      * while that read lock is held, it may need to be upgraded to a write
149      * lock. If a thread acquired a read lock and then attempted to
150      * acquire a write lock while the first lock was held, the thread
151      * would be deadlocked with itself.
152      * <p>
153      * To prevent deadlock, threads that may need to upgrade a read lock
154      * to a write lock should acquire an upgradable lock instead of a read 
155      * lock. Upgradable locks will not block threads that wish to only read.
156      * <p>
157      * To perform an upgrade, call acquireWriteLock while the upgradable 
158      * lock is still held.
159      *
160      * @param timeout milliseconds to wait for lock acquisition. If negative,
161      * timeout is infinite.
162      * @return true if the lock was acquired.
163      * @throws IllegalStateException if thread holds a read lock.
164      */
165     public boolean acquireUpgradableLock(long timeout)
166         throws InterruptedException, IllegalStateException
167     {
168         LockInfo info = (LockInfo)mLockInfoRef.get();
169         int type = info.mType;
170 
171         if (type == READ) {
172             throw new IllegalStateException
173                 ("Cannot acquire an upgradable lock while thread holds " +
174                  "only a read lock.");
175         }
176 
177         if (type == NONE) {
178             synchronized (this) {
179                 if (!upgradableLockAvailable()) {
180                     if (timeout < 0) {
181                         while (true) {
182                             wait();
183                             if (upgradableLockAvailable()) {
184                                 break;
185                             }
186                         }
187                     }
188                     else if (timeout > 0) {
189                         long expire = System.currentTimeMillis() + timeout;
190                         while (true) {
191                             wait(timeout);
192                             if (upgradableLockAvailable()) {
193                                 break;
194                             }
195                             timeout = expire - System.currentTimeMillis();
196                             if (timeout <= 0) {
197                                 return false;
198                             }
199                         };
200                     }
201                     else {
202                         return false;
203                     }
204                 }
205                 mUpgradableLockHeld = Thread.currentThread();
206             }
207             info.mType = UPGRADABLE;
208         }     
209 
210         info.mCount++;
211         return true;
212     }

223     /**
224      * A write lock is obtained only when there are no read, upgradable or 
225      * write locks held by any other thread. When a thread has a write lock,
226      * it blocks any thread that wishes to acquire any kind of lock.
227      * <p>
228      * A write lock is the strongest form of lock. When a write lock is
229      * held by a thread, then that thread alone has a lock. Requests for
230      * write locks are granted the highest priority.
231      *
232      * @param timeout milliseconds to wait for lock acquisition. If negative,
233      * timeout is infinite.
234      * @return true if the lock was acquired.
235      * @throws IllegalStateException if thread holds a read lock.
236      */
237     public boolean acquireWriteLock(long timeout)
238         throws InterruptedException, IllegalStateException
239     {
240         LockInfo info = (LockInfo)mLockInfoRef.get();
241         int type = info.mType;
242 
243         if (type == READ) {
244             throw new IllegalStateException
245                 ("Cannot acquire a write lock while thread holds " + 
246                  "only a read lock. " +
247                  "Use an upgradable lock instead of a read lock.");
248         }
249 
250         if (type != WRITE) {
251             synchronized (this) {
252                 if (type == UPGRADABLE) {
253                     // Save the count before upgrade, which will be used when
254                     // the lock is released.
255                     info.mUpgradeCount = info.mCount;
256                 }
257                 
258                 if (!writeLockAvailable(type)) {
259                     if (timeout < 0) {
260                         mWriteLockAttempts++;
261                         try {
262                             while (true) {
263                                 wait();
264                                 if (writeLockAvailable(type)) {
265                                     break;
266                                 }
267                             }
268                         }
269                         catch (InterruptedException e) {
270                             if (type == UPGRADABLE) {
271                                 info.mUpgradeCount = 0;
272                             }
273                             throw e;
274                         }
275                         finally {
276                             mWriteLockAttempts--;
277                         }
278                     }
279                     else if (timeout > 0) {
280                         mWriteLockAttempts++;
281                         long expire = System.currentTimeMillis() + timeout;
282                         try {
283                             while (true) {
284                                 wait(timeout);
285                                 if (writeLockAvailable(type)) {
286                                     break;
287                                 }
288                             }
289                             timeout = expire - System.currentTimeMillis();
290                             if (timeout <= 0) {
291                                 return false;
292                             }
293                         }
294                         catch (InterruptedException e) {
295                             if (type == UPGRADABLE) {
296                                 info.mUpgradeCount = 0;
297                             }
298                             throw e;
299                         }
300                         finally {
301                             mWriteLockAttempts--;
302                         }
303                     }
304                     else {
305                         return false;
306                     }
307                 }
308                 mWriteLockHeld = Thread.currentThread();
309             }
310             info.mType = WRITE;
311         }
312         
313         info.mCount++;
314         return true;
315     }

317     /**
318      * Release the lock held by the current thread.
319      *
320      * @return false if this thread doesn't hold a lock.
321      */
322     public boolean releaseLock() {
323         LockInfo info = (LockInfo)mLockInfoRef.get();
324         int type = info.mType;
325         int count = info.mCount;
326         if (count > 0) {
327             count--;
328         }
329         else {
330             info.mType = type = NONE;
331             count = 0;
332         }
333 
334         if ((info.mCount = count) == 0) {
335             switch (type) {
336             case NONE: default:
337                 return false;
338 
339             case READ:
340                 synchronized (this) {
341                     mReadLocks--;
342                     notifyAll();
343                 }
344                 break;
345 
346             case UPGRADABLE:
347                 synchronized (this) {
348                     mUpgradableLockHeld = null;
349                     notifyAll();
350                 }
351                 break;
352 
353             case WRITE:
354                 synchronized (this) {
355                     mWriteLockHeld = null;
356                     notifyAll();
357                 }
358                 break;
359             }
360 
361             info.mType = NONE;
362         }
363         else if (type == WRITE && info.mUpgradeCount == count) {
364             // Convert write lock back into an upgradable lock.
365             info.mType = UPGRADABLE;
366             info.mUpgradeCount = 0;
367             synchronized (this) {
368                 mWriteLockHeld = null;
369                 notifyAll();
370             }
371         }
372 
373         return true;
374     }

429     // Caller must be synchronized.
430     private boolean readLockAvailable() {
431         return mWriteLockHeld == null && mWriteLockAttempts == 0;
432     }
433 
434     // Caller must be synchronized.
435     private boolean upgradableLockAvailable() {
436         return readLockAvailable() && mUpgradableLockHeld == null;
437     }
438 
439     // Caller must be synchronized.
440     private boolean writeLockAvailable(int type) {
441         if (mReadLocks == 0 && mWriteLockHeld == null) {
442             return (type == UPGRADABLE) ? true : mUpgradableLockHeld == null;
443         }
444         return false;
445     }
