22  /**
23   * A Map that accepts int or Integer keys only. Most of the implementation for
24   * this class is ripped off from java.util.HashMap.
25   *
26   * @author Brian S O'Neill
27   * @see java.util.HashMap
28   */
29  public class IntHashMap extends AbstractMap 
30      implements Map, Cloneable, Serializable {

32      /**
33       * The hash table data.
34       */
35      private transient Entry table[];

42      /**
43       * The table is rehashed when its size exceeds this threshold.  (The
44       * value of this field is (int)(capacity * loadFactor).)

55      private float loadFactor;

57      /**
58       * The number of times this IntHashMap has been structurally modified
59       * Structural modifications are those that change the number of mappings in
60       * the IntHashMap or otherwise modify its internal structure (e.g.,
61       * rehash).  This field is used to make iterators on Collection-views of
62       * the IntHashMap fail-fast.  (See ConcurrentModificationException).
63       */
64      private transient int modCount = 0;

66      /**
67       * Constructs a new, empty map with the specified initial 
68       * capacity and the specified load factor. 
69       *
70       * @param      initialCapacity   the initial capacity of the IntHashMap.
71       * @param      loadFactor        the load factor of the IntHashMap
72       * @throws     IllegalArgumentException  if the initial capacity is less
73       *               than zero, or if the load factor is nonpositive.
74       */
75      public IntHashMap(int initialCapacity, float loadFactor) {
76          if (initialCapacity < 0) {
77              throw new IllegalArgumentException("Illegal Initial Capacity: "+
78                                                 initialCapacity);
79          }
80  
81          if (loadFactor <= 0) {
82              throw new IllegalArgumentException("Illegal Load factor: "+
83                                                 loadFactor);
84          }
85  
86          if (initialCapacity==0) {
87              initialCapacity = 1;
88          }
89  
90          this.loadFactor = loadFactor;
91          table = new Entry[initialCapacity];
92          threshold = (int)(initialCapacity * loadFactor);
93      }

144     /**
145      * Returns <tt>true</tt> if this map maps one or more keys to the
146      * specified value.
147      *
148      * @param value value whose presence in this map is to be tested.
149      * @return <tt>true</tt> if this map maps one or more keys to the
150      *         specified value.
151      */
152     public boolean containsValue(Object value) {
153         Entry tab[] = table;
154         
155         if (value==null) {
156             for (int i = tab.length ; i-- > 0 ;) {
157                 for (Entry e = tab[i] ; e != null ; e = e.next) {
158                     if (e.value == null) {
159                         return true;
160                     }
161                 }
162             }
163         } 
164         else {
165             for (int i = tab.length ; i-- > 0 ;) {
166                 for (Entry e = tab[i] ; e != null ; e = e.next) {
167                     if (value.equals(e.value)) {
168                         return true;
169                     }
170                 }
171             }
172         }
173         
174         return false;
175     }

208                 return true;

244      * @param key key whose associated value is to be returned.

247         Entry tab[] = table;
248         
249         int index = (key & 0x7FFFFFFF) % tab.length;
250         for (Entry e = tab[index]; e != null; e = e.next) {
251             if (e.key == key) {
252                 return e.value;
253             }
254         }
255         
256         return null;

259     /**
260      * Rehashes the contents of this map into a new <tt>IntHashMap</tt> instance
261      * with a larger capacity. This method is called automatically when the
262      * number of keys in this map exceeds its capacity and load factor.
263      */
264     private void rehash() {
265         int oldCapacity = table.length;
266         Entry oldMap[] = table;
267         
268         int newCapacity = oldCapacity * 2 + 1;
269         Entry newMap[] = new Entry[newCapacity];
270         
271         modCount++;
272         threshold = (int)(newCapacity * loadFactor);
273         table = newMap;
274         
275         for (int i = oldCapacity ; i-- > 0 ;) {
276             for (Entry old = oldMap[i] ; old != null ; ) {
277                 Entry e = old;
278                 old = old.next;
279                 
280                 int index = (e.key & 0x7FFFFFFF) % newCapacity;
281                 e.next = newMap[index];
282                 newMap[index] = e;
283             }
284         }
285     }

287     /**
288      * Associates the specified value with the specified key in this map.
289      * If the map previously contained a mapping for this key, the old
290      * value is replaced.
291      *
292      * @param key key with which the specified value is to be associated.
293      * @param value value to be associated with the specified key.
294      * @return previous value associated with specified key, or <tt>null</tt>
295      *         if there was no mapping for key.  A <tt>null</tt> return can
296      *         also indicate that the IntHashMap previously associated
297      *         <tt>null</tt> with the specified key.
298      */
299     public Object put(Object key, Object value) {
300         if (key instanceof Number) {
301             return put( ((Number)key).intValue(), value );
302         }
303         else {
304             throw new UnsupportedOperationException
305                 ("IntHashMap key must be a number");
306         }
307     }

309     /**
310      * Associates the specified value with the specified key in this map.
311      * If the map previously contained a mapping for this key, the old
312      * value is replaced.
313      *
314      * @param key key with which the specified value is to be associated.
315      * @param value value to be associated with the specified key.
316      * @return previous value associated with specified key, or <tt>null</tt>
317      *         if there was no mapping for key.  A <tt>null</tt> return can
318      *         also indicate that the IntHashMap previously associated
319      *         <tt>null</tt> with the specified key.
320      */
321     public Object put(int key, Object value) {
322         // Makes sure the key is not already in the IntHashMap.
323         Entry tab[] = table;
324         int index = 0;
325         
326         index = (key & 0x7FFFFFFF) % tab.length;
327         for (Entry e = tab[index] ; e != null ; e = e.next) {
328             if (e.key == key) {
329                 Object old = e.value;
330                 e.value = value;
331                 return old;
332             }
333         }
334 
335         modCount++;
336         if (count >= threshold) {
337             // Rehash the table if the threshold is exceeded
338             rehash();
339             
340             tab = table;
341             index = (key & 0x7FFFFFFF) % tab.length;
342         }
343         
344         // Creates the new entry.
345         Entry e = new Entry(key, value, tab[index]);
346         tab[index] = e;
347         count++;
348         return null;
349     }

369     /**
370      * Removes the mapping for this key from this map if present.
371      *
372      * @param key key whose mapping is to be removed from the map.
373      * @return previous value associated with specified key, or <tt>null</tt>
374      *         if there was no mapping for key.  A <tt>null</tt> return can
375      *         also indicate that the map previously associated <tt>null</tt>
376      *         with the specified key.
377      */
378     public Object remove(int key) {
379         Entry tab[] = table;
380         
381         int index = (key & 0x7FFFFFFF) % tab.length;
382         
383         for (Entry e = tab[index], prev = null; e != null; 
384              prev = e, e = e.next) {
385 
386             if (e.key == key) {
387                 modCount++;
388                 if (prev != null) {
389                     prev.next = e.next;
390                 }
391                 else {
392                     tab[index] = e.next;
393                 }
394                 
395                 count--;
396                 Object oldValue = e.value;
397                 e.value = null;
398                 return oldValue;
399             }
400         }
401         
402         return null;
403     }

405     /**
406      * Copies all of the mappings from the specified map to this one.
407      * 
408      * These mappings replace any mappings that this map had for any of the
409      * keys currently in the specified Map.
410      *
411      * @param t Mappings to be stored in this map.
412      */
413     public void putAll(Map t) {
414         Iterator i = t.entrySet().iterator();
415         while (i.hasNext()) {
416             Map.Entry e = (Map.Entry) i.next();
417             put(e.getKey(), e.getValue());
418         }
419     }

421     /**
422      * Removes all mappings from this map.
423      */
424     public void clear() {
425         Entry tab[] = table;
426         modCount++;
427         for (int index = tab.length; --index >= 0; ) {
428             tab[index] = null;
429         }
430         count = 0;
431     }

433     /**
434      * Returns a shallow copy of this <tt>IntHashMap</tt> instance: the keys and
435      * values themselves are not cloned.
436      *
437      * @return a shallow copy of this map.
438      */
439     public Object clone() {
440         try { 
441             IntHashMap t = (IntHashMap)super.clone();
442             t.table = new Entry[table.length];
443             for (int i = table.length ; i-- > 0 ; ) {
444                 t.table[i] = (table[i] != null) 
445                     ? (Entry)table[i].clone() : null;
446             }
447             t.keySet = null;
448             t.entrySet = null;
449             t.values = null;
450             t.modCount = 0;
451             return t;
452         } 
453         catch (CloneNotSupportedException e) { 
454             // this shouldn't happen, since we are Cloneable
455             throw new InternalError();
456         }
457     }

530     /**
531      * Returns a collection view of the mappings contained in this map.  Each
532      * element in the returned collection is a <tt>Map.Entry</tt>.  The
533      * collection is backed by the map, so changes to the map are reflected in
534      * the collection, and vice-versa.  The collection supports element
535      * removal, which removes the corresponding mapping from the map, via the
536      * <tt>Iterator.remove</tt>, <tt>Collection.remove</tt>,
537      * <tt>removeAll</tt>, <tt>retainAll</tt>, and <tt>clear</tt> operations.
538      * It does not support the <tt>add</tt> or <tt>addAll</tt> operations.
539      *
540      * @return a collection view of the mappings contained in this map.
541      * @see Map.Entry
542      */
543     public Set entrySet() {
544         if (entrySet==null) {
545             entrySet = new AbstractSet() {
546                 public Iterator iterator() {
547                     return new IntHashIterator(ENTRIES);
548                 }
549                 
550                 public boolean contains(Object o) {
551                     if (!(o instanceof Map.Entry)) {
552                         return false;
553                     }
554                     Map.Entry entry = (Map.Entry)o;
555                     Object key = entry.getKey();
556                     Entry tab[] = table;
557                     int hash = (key==null ? 0 : key.hashCode());
558                     int index = (hash & 0x7FFFFFFF) % tab.length;
559                     
560                     for (Entry e = tab[index]; e != null; e = e.next) {
561                         if (e.key == hash && e.equals(entry)) {
562                             return true;
563                         }
564                     }
565                     return false;
566                 }
567                 
568                 public boolean remove(Object o) {
569                     if (!(o instanceof Map.Entry))
570                     return false;
571                     Map.Entry entry = (Map.Entry)o;
572                     Object key = entry.getKey();
573                     Entry tab[] = table;
574                     int hash = (key==null ? 0 : key.hashCode());
575                     int index = (hash & 0x7FFFFFFF) % tab.length;
576                     
577                     for (Entry e = tab[index], prev = null; e != null;
578                          prev = e, e = e.next) {
579                              if (e.key == hash && e.equals(entry)) {
580                                  modCount++;
581                                  if (prev != null) {
582                                      prev.next = e.next;
583                                  }
584                                  else {
585                                      tab[index] = e.next;
586                                  }
587                                  
588                                  count--;
589                                  e.value = null;
590                                  return true;
591                              }
592                          }
593                     return false;
594                 }
595                 
596                 public int size() {
597                     return count;
598                 }
599                 
600                 public void clear() {
601                     IntHashMap.this.clear();
602                 }
603             };
604         }
605         
606         return entrySet;
607     }

609     /**
610      * IntHashMap collision list entry.
611      */
612     private static class Entry implements Map.Entry {
613         int key;
614         Object value;
615         Entry next;
616         private Integer objectKey;
617         
618         Entry(int key, Object value, Entry next) {
619             this.key = key;
620             this.value = value;
621             this.next = next;
622         }
623         
624         protected Object clone() {
625             return new Entry(key, value,
626                              (next==null ? null : (Entry)next.clone()));
627         }
628         
629         // Map.Entry Ops 
630         
631         public Object getKey() {
632             return (objectKey != null) ? objectKey : 
633                 (objectKey = new Integer(key));
634         }
635         
636         public Object getValue() {
637             return value;
638         }
639         
640         public Object setValue(Object value) {
641             Object oldValue = this.value;
642             this.value = value;
643             return oldValue;
644         }
645         
646         public boolean equals(Object o) {
647             if (!(o instanceof Map.Entry)) {
648                 return false;
649             }
650             Map.Entry e = (Map.Entry)o;
651             
652             return (getKey().equals(e.getKey())) &&
653                 (value==null ? e.getValue()==null : value.equals(e.getValue()));
654         }
655         
656         public int hashCode() {
657             return key ^ (value==null ? 0 : value.hashCode());
658         }
659         
660         public String toString() {
661             return String.valueOf(key) + "=" + value;
662         }
663     }

665     // Types of Iterators
666     private static final int KEYS = 0;
667     private static final int VALUES = 1;
668     private static final int ENTRIES = 2;
669     
670     private class IntHashIterator implements Iterator {
671         Entry[] table = IntHashMap.this.table;
672         int index = table.length;
673         Entry entry = null;
674         Entry lastReturned = null;
675         int type;
676         
677         /**
678          * The modCount value that the iterator believes that the backing
679          * List should have.  If this expectation is violated, the iterator
680          * has detected concurrent modification.
681          */
682         private int expectedModCount = modCount;
683         
684         IntHashIterator(int type) {
685             this.type = type;
686         }
687         
688         public boolean hasNext() {
689             while (entry == null && index > 0) {
690                 entry = table[--index];
691             }
692             
693             return entry != null;
694         }
695         
696         public Object next() {
697             if (modCount != expectedModCount) {
698                 throw new ConcurrentModificationException();
699             }
700             
701             while (entry == null && index > 0) {
702                 entry = table[--index];
703             }
704             
705             if (entry != null) {
706                 Entry e = lastReturned = entry;
707                 entry = e.next;
708                 return type == KEYS ? e.getKey() : 
709                     (type == VALUES ? e.value : e);
710             }
711             throw new NoSuchElementException();
712         }
713         
714         public void remove() {
715             if (lastReturned == null) {
716                 throw new IllegalStateException();
717             }
718             if (modCount != expectedModCount) {
719                 throw new ConcurrentModificationException();
720             }
721             
722             Entry[] tab = IntHashMap.this.table;
723             int index = (lastReturned.key & 0x7FFFFFFF) % tab.length;
724             
725             for (Entry e = tab[index], prev = null; e != null;
726                  prev = e, e = e.next) {
727                 if (e == lastReturned) {
728                     modCount++;
729                     expectedModCount++;
730                     if (prev == null) {
731                         tab[index] = e.next;
732                     }
733                     else {
734                         prev.next = e.next;
735                     }
736                     count--;
737                     lastReturned = null;
738                     return;
739                 }
740             }
741             throw new ConcurrentModificationException();
742         }
743     }

745     /**
746      * Save the state of the <tt>IntHashMap</tt> instance to a stream (i.e.,
747      * serialize it).
748      *
749      * @serialData The <i>capacity</i> of the IntHashMap (the length of the
750      *         bucket array) is emitted (int), followed  by the
751      *         <i>size</i> of the IntHashMap (the number of key-value
752      *         mappings), followed by the key (Object) and value (Object)
753      *         for each key-value mapping represented by the IntHashMap
754      * The key-value mappings are emitted in no particular order.
755      */
756     private void writeObject(java.io.ObjectOutputStream s)
757         throws IOException
758     {
759         // Write out the threshold, loadfactor, and any hidden stuff
760         s.defaultWriteObject();
761         
762         // Write out number of buckets
763         s.writeInt(table.length);
764         
765         // Write out size (number of Mappings)
766         s.writeInt(count);
767         
768         // Write out keys and values (alternating)
