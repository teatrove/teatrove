28  /**
29   * Parses a properties file similar to how {@link java.util.Properties} does,
30   * except:
31   *
32   * <ul>
33   * <li>Values have trailing whitespace trimmed.
34   * <li>Quotation marks ( " or ' ) can be used to define keys and values that
35   * have embedded spaces.
36   * <li>Quotation marks can also be used to define multi-line keys and values
37   * without having to use continuation characters.
38   * <li>Properties may be nested using braces '{' and '}'.
39   * </ul>
40   *
41   * Just like Properties, comment lines start with optional whitespace followed
42   * by a '#' or '!'. Keys and values may have an optional '=' or ':' as a
43   * separator, unicode escapes are supported as well as other common escapes.
44   * A line may end in a backslash so that it continues to the next line.
45   * Escapes for brace characters '{' and '}' are also supported.
46   *
47   * Example:
48   *
49   * <pre>
50   * # Properties file
51   *
52   * foo = bar
53   * foo.sub = blink
54   * empty
55   *
56   * block {
57   *     inner {
58   *         foo = bar
59   *         item
60   *     }
61   *     next.item = "true"
62   * }
63   *
64   * section = test {
65   *     level = 4
66   *     message = "Message: "
67   * }
68   * </pre>
69   *
70   * is equivalent to
71   *
72   * <pre>
73   * # Properties file
74   *
75   * foo = bar
76   * foo.sub = blink
77   * empty
78   *
79   * block.inner.foo = bar
80   * block.inner.item
81   * block.next.item = true
82   *
83   * section = test
84   * section.level = 4
85   * section.message = Message: 
86   * </pre>
87   *
88   * @author Brian S O'Neill
89   */
90  public class PropertyParser {

92      //
93      // Properties   ::= { PropertyList }
94      // PropertyList ::= { Property | COMMENT }
95      // Property     ::= KEY [ VALUE ] [ Block ]
96      // Block        ::= LBRACE PropertyList RBRACE

161     /**
162      * Parses properties from the given reader and stores them in the Map. To
163      * capture any parsing errors, call addErrorListener prior to parsing.
164      */    
165     public void parse(Reader reader) throws IOException {
166         mScanner = new Scanner(reader);
167 
168         mScanner.addErrorListener(new ErrorListener() {
169             public void parseError(ErrorEvent e) {
170                 dispatchParseError(e);
171             }
172         });
173 
174         try {
175             parseProperties();
176         }
177         finally {
178             mScanner.close();
179         }
180     }
181 
182     private void parseProperties() throws IOException {
183         Token token;
184         while ((token = peek()).getId() != Token.EOF) {
185             switch (token.getId()) {
186 
187             case Token.KEY:
188             case Token.LBRACE:  
189             case Token.COMMENT:
190                 parsePropertyList(null);
191                 break;
192 
193             case Token.RBRACE:
194                 token = read();
195                 error("No matching left brace", token);
196                 break;
197 
198             default:
199                 token = read();
200                 error("Unexpected token: " + token.getValue(), token);
201                 break;
202             }
203         }
204     }
205 
206     private void parsePropertyList(String keyPrefix) throws IOException {
207         Token token;
208 
209     loop:
210         while ((token = peek()).getId() != Token.EOF) {
211             switch (token.getId()) {
212 
213             case Token.KEY:
214                 token = read();
215                 parseProperty(keyPrefix, token);
216                 break;
217                 
218             case Token.COMMENT:
219                 read();
220                 break;
221 
222             case Token.LBRACE:
223                 read();
224                 error("Nested properties must have a base name", token);
225                 parseBlock(keyPrefix);
226                 break;
227                 
228             default:
229                 break loop;
230             }
231         }
232     }
233 
234     private void parseProperty(String keyPrefix, Token token)
235         throws IOException {
236 
237         String key = token.getValue();
238         if (keyPrefix != null) {
239             key = keyPrefix + key;
240         }
241 
242         String value = null;
243 
244         if (peek().getId() == Token.VALUE) {
245             token = read();
246             value = token.getValue();
247         }
248 
249         if (peek().getId() == Token.LBRACE) {
250             read();
251             parseBlock(key + '.');
252         }
253         else if (value == null) {
254             value = "";
255         }
256 
257         if (value != null) {
258             putProperty(key, value, token);
259             propertyChange(key, value, token.getSourceInfo());
260         }
261     }
262 
263     // When this is called, the LBRACE token has already been read.
264     private void parseBlock(String keyPrefix) throws IOException {
265         parsePropertyList(keyPrefix);
266             
267         Token token;
268         if ((token = peek()).getId() == Token.RBRACE) {
269             read();
270         }
271         else {
272             error("Right brace expected", token);
273         }
274     }
275 
276     private void putProperty(String key, String value, Token token) {
277         if (mMap.containsKey(key)) {
278             error("Property \"" + key + "\" already defined", token);
279         }
280         mMap.put(key, value);
281     }

366     private static class Token implements java.io.Serializable {
367         public final static int UNKNOWN = 0;
368         public final static int EOF = 1;
369         
370         public final static int COMMENT = 2;
371         public final static int KEY = 3;
372         public final static int VALUE = 4;
373         
374         public final static int LBRACE = 5;
375         public final static int RBRACE = 6;

377         private final static int LAST_ID = 6;
378     
379         private int mTokenId;
380         private String mValue;
381         private SourceInfo mInfo;
382 
383         Token(int sourceLine,
384               int sourceStartPos, 
385               int sourceEndPos,
386               int tokenId,
387               String value) {
388             
389             mTokenId = tokenId;
390             mValue = value;
391             
392             if (tokenId > LAST_ID) {
393                 throw new IllegalArgumentException("Token Id out of range: " +
394                                                    tokenId);
395             }
396             
397             mInfo = new SourceInfo(sourceLine, sourceStartPos, sourceEndPos);
398             
399             if (sourceStartPos > sourceEndPos) {
400                 // This is an internal error.
401                 throw new IllegalArgumentException
402                     ("Token start position greater than " + 
403                      "end position at line: " + sourceLine);
404             }
405         }

438         public String toString() {
439             StringBuffer buf = new StringBuffer(10);
440 
441             String image = getCode();
442             
443             if (image != null) {
444                 buf.append(image);
445             }
446             
447             String str = getValue();
448             
449             if (str != null) {
450                 if (image != null) {
451                     buf.append(' ');

545             }

563         private Token scanToken() throws IOException {
564             if (mSource.isClosed()) {
565                 if (mEOFToken == null) {
566                     mEOFToken = makeToken(Token.EOF, null);
567                 }
568                 
569                 return mEOFToken;
570             }
571             
572             int c;
573             
574             while ( (c = mSource.read()) != -1 ) {
575                 switch (c) {
576 
577                 case SourceReader.ENTER_CODE:
578                 case SourceReader.ENTER_TEXT:
579                     continue;
580                     
581                 case '#':
582                 case '!':
583                     mScanKey = true;
584                     return scanComment();
585 
586                 case '{':
587                     mScanKey = true;
588                     return makeToken(Token.LBRACE, "{");
589                 case '}':
590                     mScanKey = true;
591                     return makeToken(Token.RBRACE, "}");
592                 
593                 case '0': case '1': case '2': case '3': case '4': 
594                 case '5': case '6': case '7': case '8': case '9':
595                 case 'a': case 'b': case 'c': case 'd': case 'e':
596                 case 'f': case 'g': case 'h': case 'i': case 'j':
597                 case 'k': case 'l': case 'm': case 'n': case 'o':
598                 case 'p': case 'q': case 'r': case 's': case 't':
599                 case 'u': case 'v': case 'w': case 'x': case 'y':
600                 case 'z': case '.':
601                 case 'A': case 'B': case 'C': case 'D': case 'E':
602                 case 'F': case 'G': case 'H': case 'I': case 'J':
603                 case 'K': case 'L': case 'M': case 'N': case 'O':
604                 case 'P': case 'Q': case 'R': case 'S': case 'T':
605                 case 'U': case 'V': case 'W': case 'X': case 'Y':
606                 case 'Z': case '_':
607                     mSource.unread();
608                     return scanKeyOrValue();
609 
610                 case '\n':
611                     mScanKey = true;
612                     // fall through
613                 case ' ': 
614                 case '\0':
615                 case '\t':
616                     continue;
617 
618                 default:
619                     if (Character.isWhitespace((char)c)) {
620                         continue;
621                     }
622                     else {
623                         mSource.unread();
624                         return scanKeyOrValue();
625                     }
626                 }
627             }
628             
629             if (mEOFToken == null) {
630                 mEOFToken = makeToken(Token.EOF, null);
631             }
632             
633             return mEOFToken;
634         }

636         private Token scanKeyOrValue() throws IOException { 
637             StringBuffer buf = new StringBuffer(40);
638             boolean trim = true;
639 
640             int startLine = mSource.getLineNumber();
641             int startPos = mSource.getStartPosition();
642             int endPos = mSource.getEndPosition();
643 
644             boolean skipWhitespace = true;
645             boolean skipSeparator = true;
646 
647             int c;
648         loop:
649             while ( (c = mSource.read()) != -1 ) {
650                 switch (c) {
651 
652                 case '\n':
653                     mSource.unread();
654                     break loop;
655                 
656                 case '\\':
657                     int next = mSource.read();
658                     if (next == -1 || next == '\n') {
659                         // line continuation
660                         skipWhitespace = true;
661                         continue;
662                     }
663 
664                     c = processEscape(c, next);
665                     skipWhitespace = false;
666                     break;
667 
668                 case '{':
669                 case '}':
670                     mSource.unread();
671                     break loop;
672                 
673                 case '=':
674                 case ':':
675                     if (mScanKey) {
676                         mSource.unread();
677                         break loop;
678                     }
679                     else if (skipSeparator) {
680                         skipSeparator = false;
681                         continue;
682                     }
683                     skipWhitespace = false;
684                     break;
685 
686                 case '\'':
687                 case '"':
688                     if (buf.length() == 0) {
689                         scanStringLiteral(c, buf);
690                         endPos = mSource.getEndPosition();
691                         trim = false;
692                         break loop;
693                     }
694                     // fall through
695                 case '0': case '1': case '2': case '3': case '4': 
696                 case '5': case '6': case '7': case '8': case '9':
697                 case 'a': case 'b': case 'c': case 'd': case 'e':
698                 case 'f': case 'g': case 'h': case 'i': case 'j':
699                 case 'k': case 'l': case 'm': case 'n': case 'o':
700                 case 'p': case 'q': case 'r': case 's': case 't':
701                 case 'u': case 'v': case 'w': case 'x': case 'y':
702                 case 'z': case '.':
703                 case 'A': case 'B': case 'C': case 'D': case 'E':
704                 case 'F': case 'G': case 'H': case 'I': case 'J':
705                 case 'K': case 'L': case 'M': case 'N': case 'O':
706                 case 'P': case 'Q': case 'R': case 'S': case 'T':
707                 case 'U': case 'V': case 'W': case 'X': case 'Y':
708                 case 'Z': case '_':
709                     skipWhitespace = false;
710                     break;
711 
712                 case ' ': 
713                 case '\0':
714                 case '\t':
715                     if (skipWhitespace) {
716                         continue;
717                     }
718                     if (mScanKey) {
719                         break loop;
720                     }
721                     break;
722 
723                 default:
724                     if (Character.isWhitespace((char)c)) {
725                         if (skipWhitespace) {
726                             continue;
727                         }
728                         if (mScanKey) {
729                             break loop;
730                         }
731                     }
732                     else {
733                         skipWhitespace = false;
734                     }
735                     break;
736                 }
737 
738                 buf.append((char)c);
739                 endPos = mSource.getEndPosition();
740                 skipSeparator = false;
741             }
742 
743             int tokenId;
744             if (mScanKey) {
745                 tokenId = Token.KEY;
746                 mScanKey = false;
747             }
748             else {
749                 tokenId = Token.VALUE;
750                 mScanKey = true;
751             }
752 
753             String value = buf.toString();
754 
755             if (trim) {
756                 value = value.trim();
757             }
758 
759             return new Token(startLine, startPos, endPos, tokenId, value);
760         }

762         private Token scanComment() throws IOException {
763             StringBuffer buf = new StringBuffer(40);
764 
765             int startLine = mSource.getLineNumber();
766             int startPos = mSource.getStartPosition();
767             int endPos = mSource.getEndPosition();
768 
769             int c;
770             while ( (c = mSource.peek()) != -1 ) {
771                 if (c == '\n') {
772                     break;
773                 }
774                 
775                 mSource.read();
776                 buf.append((char)c);
777                 
778                 endPos = mSource.getEndPosition();
779             }
780 
781             return new Token(startLine, startPos, endPos,
782                              Token.COMMENT, buf.toString());
783         }

785         private void scanStringLiteral(int quote, StringBuffer buf)
786             throws IOException {
787 
788             int c;
789             while ( (c = mSource.read()) != -1 ) {
790                 if (c == quote) {
791                     return;
792                 }
793 
794                 if (c == '\\') {
795                     int next = mSource.read();
796                     if (next == -1 || next == '\n') {
797                         // line continuation
798                         continue;
799                     }
800                     c = processEscape(c, next);
801                 }
802 
803                 buf.append((char)c);
804             }
805         }
806 
807         private int processEscape(int c, int next) {
808             switch (next) {
809             case '0':
810                 return '\0';
811             case 't':
812                 return '\t';
813             case 'n':
814                 return '\n';
815             case 'f':
816                 return '\f';
817             case 'r':
818                 return '\r';
819 
820             case '\\':
821             case '\'':
822             case '\"':
823             case '=':
824             case ':':
825             case '{':
826             case '}':
827                 return next;
828 
829             default:
830                 error("Invalid escape code: \\" + (char)next);
831                 return next;
832             }
833         }
