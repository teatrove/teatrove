19  /**
20   * CodeAssembler is a high-level interface for assembling Java Virtual Machine
21   * byte code. It can also be used as a visitor to a disassembler.
22   *
23   * @author Brian S O'Neill
24   */
25  public interface CodeAssembler {

27      /**
28       * Returns LocalVariable references for all the parameters passed into
29       * the method being assembled, not including any "this" reference.
30       * Returns a zero-length array if there are no passed in parameters.
31       * <p>
32       * The names of the LocalVariables returned by this method are initially
33       * set to null. It is encouraged that a name be provided.
34       */
35      public LocalVariable[] getParameters();

44       * store.

155     /**
156      * Generates code that loads a value from an array. An array
157      * reference followed by an index must be on the stack. The array 
158      * reference and index are replaced by the value retrieved from the array

160      * <p>
161      * The type doesn't need to be an exact match for objects.

176      * <p>
177      * The type doesn't need to be an exact match for objects.
178      * TypeDesc.OBJECT works fine for all objects. For primitive types, use
179      * the appropriate TypeDesc. For an int, the type is TypeDesc.INT.
180      *
181      * @param type The type of data stored in the array.

183     public void storeToArray(TypeDesc type);

576     /**
577      * Generates code for either a unary or binary math operation on one
578      * or two values pushed on the stack.
579      * <p>
580      * Pass in an opcode from the the Opcode class. The only valid math
581      * opcodes are:
582      *
583      * <pre>
584      * IADD, ISUB, IMUL, IDIV, IREM, INEG, IAND, IOR, IXOR, ISHL, ISHR, IUSHR
585      * LADD, LSUB, LMUL, LDIV, LREM, LNEG, LAND, LOR, LXOR, LSHL, LSHR, LUSHR
586      * FADD, FSUB, FMUL, FDIV, FREM, FNEG
587      * DADD, DSUB, DMUL, DDIV, DREM, DNEG
588      *
589      * LCMP
590      * FCMPG, FCMPL
591      * DCMPG, DCMPL
592      * </pre>
593      *
594      * A not operation (~) is performed by doing a loadConstant with either
595      * -1 or -1L followed by math(Opcode.IXOR) or math(Opcode.LXOR).
596      *
597      * @param opcode An opcode from the Opcode class.
598      * @exception IllegalArgumentException When the opcode selected is not
599      * a math operation.
600      * @see Opcode
601      */
602     public void math(byte opcode);
